package battleshipsGame;

    public static Board bd2=null;

                print("Illegal input!!!");
                continue;
            }
        }

    }

        while(!sc.nextLine().split(" ")[0].equals("")){
 
        }
    }

Class: Ships

package battleshipsGame;
import java.util.*;
 
public class Ships {
    private String type;
    private Point[] coors;
    public ArrayList<Point> survived = new ArrayList<Point>();
    public ArrayList<Point> destroyed = new ArrayList<Point>();
 
    private void checkAndAdd(Board bd){
        if (!checkValidity(coors, bd)){
            System.out.println("Illegal ship placement!!!");
        }
        else{
            for (Point p: coors){
                survived.add(p);
            }
            bd.ShipArray.add(this);
            System.out.println("Successfully added "+this.type+" on "+bd.toString());
        }
    }
    public Ships(Board bd, Point[] pts){
        coors = pts.clone();
        if (pts.length==1){
            type = "frigate";
        }
        else if (pts.length==2){
            type = "destroyer";
        }
        else if (pts.length==3){
            type = "submarine";
        }
        else if (pts.length==4){
            type = "cruiser";
        }
        else if (pts.length==5){
            type = "battleship";
        }
        checkAndAdd(bd);
    }
    public String getType(){return type;}
    public Point[] getCoors(){return coors;}
    private boolean checkValidity(Point[] pts, Board bd){
        if (pts.length>5) return false;
        if (pts.length>1){
            int difX = pts[1].getX()-pts[0].getX();
            int difY = pts[1].getY()-pts[0].getY();
            if (Math.abs(difX)>1 || Math.abs(difY)>1) return false;
            for (int i = 2;i<pts.length;i++){
                if ((pts[i].getX()-pts[i-1].getX())!=difX || (pts[i].getY()-pts[i-1].getY())!=difY)
                    return false;
            }
        }
        if (bd.ShipArray != null){
            for (Ships sp: bd.ShipArray){
                if (sp.getCoors().length==coors.length) return false;
                for (Point pt_sp: sp.getCoors()){
                    for (Point p: coors){
                        int dx = Math.abs(pt_sp.getX()-p.getX());
                        int dy = Math.abs(pt_sp.getY()-p.getY());
                        if ((dx==0 && dy==1)||(dx==1 && dy==0)){
                            return false;
                        }
                    }
                }
            }
        }
        return true;
    }
 
 
}
 

Class: Screen (Inherits JFrame)

package battleshipsGame;
 
import javax.swing.*;
import java.awt.Color;
import java.awt.Font;
import java.awt.Graphics;
import java.util.*;
 
public class Screen extends JFrame {
    private static final long serialVersionUID = 1L;
    private ArrayList<Ships> sps = null;
    private List<Point> shotArea;
    public int winState = -1;
    public boolean visibility = true;
 
    public Screen(String name, int wid, int hei, int locX, int locY) {
        setTitle(name);
        this.setLocation(locX, locY);
        this.setVisible(true);
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        this.setSize(wid, hei);
        this.getContentPane().setBackground(new Color(255, 255, 255));
        this.setAlwaysOnTop(true);
    }
 
    @SuppressWarnings("unchecked")
    public void setShips(ArrayList<Ships> sps) {
        this.sps = (ArrayList<Ships>) sps.clone();
    }
 
    public void setShotArea(List<Point> pts) {
        this.shotArea = pts;
    }
 
    public void paint(Graphics g) {
        super.paint(g);
 
        // resize controller
        if (this.getWidth() > this.getHeight())
            this.setSize(getHeight(), getHeight());
        else
            this.setSize(getWidth(), getWidth());
 
        //parameters set for drawing
        int addPara = (int) (this.getWidth() * 0.02);
        int leftBound = (int) (getWidth() * (1.0 / 9)) - addPara;
        int topBound = (int) (getHeight() * (1.5 / 9)) - addPara;
        int unit = (int) ((getHeight() - topBound - addPara) / 8.0);
        int unitShip = (int) (unit * 0.75);
        g.setFont(new Font("Calibri", Font.BOLD, (int)(getWidth()/25)));
 
        //draw the grid
        for (int i = 0; i < 8; i++) {
            for (int j = 0; j < 8; j++) {
                g.drawRect(leftBound + i * (unit), topBound + j * (unit), unit, unit);
            }
            g.drawString(String.valueOf(i), (int) (leftBound * 1.5 + unit * i), (int) (topBound * 0.85));
 
            g.drawString(String.valueOf(i), (int) (leftBound * 0.5), (int) (topBound * 1.45 + unit * i));
        }
 
        //draw the shot area (red blocks)
        if (shotArea != null) {
            g.setColor(Color.RED);
            for (Point pt : shotArea)
                g.fillRect(leftBound + pt.getX() * (unit), topBound + pt.getY() * (unit), unit, unit);
        }
 
        //draw the ship survived if visibility is true
        if (visibility) {
            if (sps != null) {
                for (Ships sp : sps) {
                    g.setColor(new Color(125, 255, 255));
                    for (Point p : sp.survived) {
                        g.fillOval(leftBound + p.getX() * unit + (unit - unitShip) / 2,
                                topBound + p.getY() * unit + (unit - unitShip) / 2, unitShip, unitShip);
                    }
                }
            }
        }
 
        //draw the destroyed ship
        g.setColor(Color.BLACK);
        if (sps != null) {
            for (Ships sp : sps) {
                for (Point p : sp.destroyed) {
                    g.fillOval(leftBound + p.getX() * unit + (unit - unitShip) / 2,
                            topBound + p.getY() * unit + (unit - unitShip) / 2, unitShip, unitShip);
                }
            }
        }
 
        //displays "you win" or "you lose" if one of the player wins
        g.setFont(new Font("Calibri", Font.BOLD, (int)(getHeight()/5)));
        g.setColor(Color.ORANGE);
        if (winState==1){
            g.drawString("You win!",getWidth()/10,getHeight()/2);
        }
        else if (winState==0){
            g.drawString("You lose!",getWidth()/10,getHeight()/2);
        }
 
    }
}
 


Class: Board

package battleshipsGame;
import java.util.*;
 
public class Board {
    private Screen sc;
    public ArrayList<Ships> ShipArray=new ArrayList<Ships>();
    public List<Point> shotArea=new ArrayList<Point>();
    private String name;
    public Board(String player, int locX, int locY){
         sc = new Screen(player,400,400, locX, locY);
         name=player;
    }
    public String getPlayerName(){return name;}
    public void win(){
        sc.winState=1;
        sc.repaint();
    }
    public void lose(){
        sc.winState=0;
        sc.repaint();
    }
    public void drawAllShips(){
        sc.setShips(ShipArray);
        sc.repaint();
    }
    public String toString(){
        return "Board of "+sc.getTitle();
    }
    public boolean shoot(int x, int y){
        if ((x<8 && y<8) &&(x>=0 && y>=0)){
            Point shotPoint = new Point(x,y);
            shotArea.add(shotPoint);
            boolean isHit = false;
 
            for (Ships sp : ShipArray){
                for (int i = 0;i<sp.survived.size();i++){
                    if (sp.survived.get(i).equals(shotPoint)){
                        sp.survived.remove(i);
                        sp.destroyed.add(shotPoint);
                        System.out.println("Hit!");
                        isHit = true;
                        if (sp.survived.size()==0){
                            System.out.println("You sank "+sc.getTitle()+"'s "+sp.getType());
                        }
                    }
                }
            }
 
            if (!isHit) System.out.println("Miss!");
            sc.setShotArea(shotArea);
            sc.repaint();
 
            int count = 0;
            for (Ships sp : ShipArray){
                count+=sp.survived.size();
            }
            if (count==0) return true;
 
        }
        else{
            System.out.println("Out of range!");
            throw new IllegalArgumentException();
        }
        return false;
    }
    public void setVisibility(boolean vis){
        sc.visibility=vis;
        sc.repaint();
    }
}
 



Class: Point

package battleshipsGame;
 
public class Point{
    private int x;
    private int y;
    public Point(){
        x=0;y=0;
    }
    public Point(int x1, int y1){
        x=x1;y=y1;
    }
    public int getX(){
        return x;
    }
    public int getY(){
        return y;
    }
    public boolean equals(Object another){
        return x==((Point)another).getX() && y==((Point)another).getY();
    }
 
}
