import matplotlib.pyplot as plt
import random
import time
 
 
def has_duplicates(list1):
    for item in list1:
        if list1.count(item) > 1:
            return True
    return False
 
 
def random_birthday_generator(number):
    birthday = []
    for i in range(number):
        birthday.append(random.randrange(1, 366))
    return birthday
 
 
def same_birthday_statistic(num):
    count = 0
    for i in range(1000):
        if has_duplicates(random_birthday_generator(num)):
            count +=1
    result = count/10
    print(result)
    return result
 
t0 = time.clock()
number_of_birthdays = list(range(1, 101))
probability = []
for num in number_of_birthdays:
    probability.append(same_birthday_statistic(num))
t1 = time.clock()
print(list(zip(number_of_birthdays, probability)))
print("Total time spent:", t1-t0)
plt.plot(number_of_birthdays, probability,"r*")
plt.title("The Relationship between Number of Students\nand the Probability of Having the Same Birthdays")
plt.ylabel("Probability / percentage")
plt.xlabel("Number of students")
plt.savefig("Statistic_Result.pdf")
plt.show()
