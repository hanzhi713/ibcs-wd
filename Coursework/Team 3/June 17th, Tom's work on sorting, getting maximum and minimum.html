            a.append(b)
        else:
            for z in range(0,len(a)+1):
                success=1
                if b > a[z]:
                    try:
                        a.remove(b)
                        success = 0
                    except: None
                    a.insert(z-1+success,b)
                    break
                elif b == a[z]:None
                else:
                    try:
                        a.remove(b)
                        success = 0

                    a.insert(z+success ,b)
    return "The maximum is "+ str(a[0]) + ". The minimum is "+ str(a[len(a)-1]) + ". The list is "+ str(a)
