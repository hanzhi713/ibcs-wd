public String withoutX(String str) {
          if (str.startsWith("x")){
            str=str.substring(1,str.length());
          }
          if (str.endsWith("x")){
              str=str.substring(0,str.length()-1);
          }
          return str;
        }
    public int countHi(String str) {
          int id = 0;
          int posid = str.indexOf("hi", id);
          int count = 0;
          while (posid!=-1){
              id = posid+1;
              count+=1;
              posid = str.indexOf("hi", id);
          }
          return count;
    }
    public String repeatSeparator(String word, String sep, int count) {
        String result = "";
          for (int i=0;i<count*2-1;i++){
              if (i%2==0){
                  result+=word;
              }
              else{
                  result+=sep;
              }
          }
          return result;
    }
    public boolean linearIn(int[] outer, int[] inner) {
        int idxin = 0;
        int idxout = 0;
        while (idxout<outer.length && idxin<inner.length){
            if (outer[idxout]==inner[idxin]){
                idxin+=1;
            }
            idxout+=1;
        }
        return idxin==inner.length;
    }
    public int[] seriesUp(int n) {
          int[] result = new int[n*(n + 1)/2];
          int idx = 0;
          int count = 0;
          while (idx<result.length){
              for (int num=1;num<count+2;num++){
                  result[idx]=num;
                  idx++;
              }
              count++;
          }
          return result;
    }
 
    public int matchUp(String[] a, String[] b) {
        int count = 0;
        for (int i = 0; i < a.length; i++) {
            if (a[i].length() == 0 || b[i].length() == 0)
                continue;
            if (a[i].startsWith(b[i].substring(0, 1)))
                count += 1;
        }
        return count;
    }
