Q2:

import random
number_drawn = []
 
 
def initialise_number_drawn():
    global number_drawn
    for i in range(50):
        number_drawn.append(False)
 
 
def main():
    initialise_number_drawn()
    generated = 0
    while generated < 6:
        next_number = int(random.uniform(1,51))
        if number_drawn[next_number] == False:
            print(next_number)
            number_drawn[next_number] = True
            generated += 1
 
 
main()

Binary search version:

message_string = "ATSEVEN"
import string
alphabet = string.ascii_uppercase
substitution = "PLFNOCQUDZVGIXMWJBKEAHSYRT"
encrypted = ""
for counter in range(0,len(message_string)):
    let = message_string[counter]
    lb = 0
    ub = 25
    while True:
        if lb == ub:
            None
        mid_index = (lb + ub) //2
        item_at_mid = alphabet[mid_index]
 
        if item_at_mid == let:
            encrypted += substitution[mid_index]
            break
        if item_at_mid < let:
            lb = mid_index + 1
        else:
            ub = mid_index
print(encrypted)

Most efficient version:

message_string = "ATSEVEN"
import string
alphabet = string.ascii_uppercase
substitution = "PLFNOCQUDZVGIXMWJBKEAHSYRT"
encrypted = ""
for letter in message_string:
    encrypted += substitution[ord(letter) - 65]
print(encrypted)
