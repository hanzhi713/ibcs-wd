Class BinaryTreeNode

package tree2;
 
public class BinaryTreeNode<K extends Comparable<K>, C> {
    private K key;
    private C cargo;
    private BinaryTreeNode<K, C> left, right;
 
    /**
     *
     * Each tree node consists of four components
     * @param a key of any class K which inherits Comparable<K> abstract class (and therefore has implementation for comparedTo(K) )
     * @param a cargo of any Class C
     * @param a pointer pointing its left child
     * @param a pointer pointing its right child
     * */
    public BinaryTreeNode(K key, C cargo, BinaryTreeNode<K, C> left, BinaryTreeNode<K, C> right){
        this.key = key;
        this.cargo = cargo;
        this.left = left;
        this.right = right;
    }
    public C getCargo(){
        return cargo;
    }
    public void setCargo(C cargo){
        this.cargo = cargo;
    }
    public void setKey(K key){
        this.key = key;
    }
    public K getKey(){
        return key;
    }
    public BinaryTreeNode<K, C> getLeft(){
        return left;
    }
    public BinaryTreeNode<K, C> getRight(){
        return right;
    }
    public void setRight(BinaryTreeNode<K, C> t){
        right = t;
    }
    public void setLeft(BinaryTreeNode<K, C> t){
        left = t;
    }
    public String toString(){
        return String.format("(%s, %s)", key, cargo);
    }
}
 

Class BinaryTree

package tree2;
 
import java.util.*;
 
public class BinaryTree<K extends Comparable<K>, C>{
    private BinaryTreeNode<K, C> root;
 
    public static BinaryTree<Integer, Integer> randIntTree(int numOfElements){
        ArrayList<Integer> a = new ArrayList<Integer>();
        BinaryTree<Integer, Integer> bt = new BinaryTree<Integer, Integer>();
        for (int i = 1; i < numOfElements; i++)
            a.add(i);
        Collections.shuffle(a);
        for (Integer i : a)
            bt.add(i, i);
        return bt;
    }
 
    public BinaryTree(){
        root = null;
    }
    public BinaryTree(BinaryTreeNode<K, C> t){
        root = t;
    }
    public BinaryTree(K key, C cargo){
        root = new BinaryTreeNode<K, C>(key, cargo, null, null);
    }
    public BinaryTreeNode<K, C> getRoot(){
        return root;
    }
    public void add(K key, C cargo){
        BinaryTreeNode<K, C> current = root;
        if (root==null){
            root = new BinaryTreeNode<K, C>(key, cargo, null, null);
            return;
        }
        while (true){
            if (current.getKey().compareTo(key)<0){
                if (current.getRight() != null)
                    current = current.getRight();
                else{
                    current.setRight(new BinaryTreeNode<K, C>(key, cargo, null, null));
                    return;
                }
            }
            else if (current.getKey().compareTo(key)>0){
                if (current.getLeft() != null)
                    current = current.getLeft();
                else{
                    current.setLeft(new BinaryTreeNode<K, C>(key, cargo, null, null));
                    return;
                }
            }
            else
                throw new IllegalArgumentException("Duplicated key!!!");
        }
 
    }
    public BinaryTreeNode<K, C> search(K key){
        BinaryTreeNode<K, C> current = root;
        while (current!=null){
            if (current.getKey().compareTo(key)<0)
                current = current.getRight();
            else if (current.getKey().compareTo(key)>0)
                current = current.getLeft();
            else
                return current;
        }
        return null;
    }
    public BinaryTreeNode<K, C> getParent(BinaryTreeNode<K, C> t){
        K key = t.getKey();
        BinaryTreeNode<K, C> current = root;
        BinaryTreeNode<K, C> last = root;
        while (current!=null){
            if (current.getKey().compareTo(key)<0){
                last = current;
                current = current.getRight();
            }
            else if (current.getKey().compareTo(key)>0){
                last = current;
                current = current.getLeft();
            }
            else
                return last;
        }
        return last;
    }
    public void remove(K key){
        root = remove(key, root);
    }
    public BinaryTreeNode<K, C> remove(K key, BinaryTreeNode<K, C> n){
       if (n == null)
           return n;
       int v = key.compareTo(n.getKey());
       if (v > 0)
           n.setRight(remove(key, n.getRight()));
       else if (v < 0)
           n.setLeft(remove(key, n.getLeft()));
       else if (n.getLeft() != null && n.getRight() != null){
           BinaryTreeNode<K, C> min = findMin(n.getRight());
           n.setCargo(min.getCargo());
           n.setKey(min.getKey());
           n.setRight(remove(n.getKey(), n.getRight()));
       }
       else
           n = (n.getLeft() != null) ? n.getLeft() : n.getRight();
       return n;
    }
    public BinaryTreeNode<K, C> findMax(){
        return findMax(root);
    }
    public BinaryTreeNode<K, C> findMax(BinaryTreeNode<K, C> t){
        if (t==null) return null;
        BinaryTreeNode<K, C> current = t.getRight();
        BinaryTreeNode<K, C> last = t;
        while (current!=null){
            last = current;
            current = current.getRight();
        }
        return last;
    }
    public BinaryTreeNode<K, C> findMin(BinaryTreeNode<K, C> t){
        if (t==null) return null;
        BinaryTreeNode<K, C> current = t.getLeft();
        BinaryTreeNode<K, C> last = t;
        while (current!=null){
            last = current;
            current = current.getLeft();
        }
        return last;
    }
    public List<BinaryTreeNode<K, C>> inOrder(BinaryTreeNode<K, C> root) {
        List<BinaryTreeNode<K, C>> result = new ArrayList<BinaryTreeNode<K, C>>();
        if(root !=null)
            inOrderHelper(root, result);
        return result;
    }
 
    private void inOrderHelper(BinaryTreeNode<K, C> p, List<BinaryTreeNode<K, C>> result){
        if(p.getLeft()!=null)
            inOrderHelper(p.getLeft(), result);
        result.add(p);
        if(p.getRight()!=null)
            inOrderHelper(p.getRight(), result);
    }
}

package tree2;
 
import java.awt.*;
import java.awt.event.*;
import javax.swing.*;
//import java.util.*;
 
public class VisualTree extends JPanel {
    private static final long serialVersionUID = 1L;
    BinaryTree<Integer, Integer> root = new BinaryTree<Integer, Integer>();
    TreeView treeView = new TreeView();
    private JTextField jtField = new JTextField(5);
    private JButton jbtInsert = new JButton("Insert");
    private JButton jbtDelete = new JButton("Delete");
    private JButton jbtGenerate = new JButton("Rand Tree");
    private JLabel jLab = new JLabel("Enter a key :");
    private static final Font GOODFONT = new Font("Arial", Font.BOLD, 15);
    public VisualTree(BinaryTree<Integer, Integer> root) {
        this.root = root;
        setUI();
    }
 
    private void setUI() {
        this.setLayout(new BorderLayout());
        add(treeView, BorderLayout.CENTER);
        JPanel panel = new JPanel();
        jbtInsert.setFont(GOODFONT);
        jbtDelete.setFont(GOODFONT);
        jbtGenerate.setFont(GOODFONT);
        jtField.setFont(GOODFONT);
        jLab.setFont(GOODFONT);
        panel.add(jLab);
        panel.add(jtField);
        panel.add(jbtInsert);
        panel.add(jbtDelete);
        panel.add(jbtGenerate);
        add(panel, BorderLayout.SOUTH);
        jbtInsert.addActionListener(new ActionListener() {
 
            @Override
            public void actionPerformed(ActionEvent arg0) {
                int key = Integer.parseInt(jtField.getText());
                if (root.search(key) != null) {
                    JOptionPane.showMessageDialog(null, key + " is already in the tree");
                } else {
                    root.add(key, key);
                    treeView.repaint();
                }
                jtField.setText("");
            }
        });
 
        jbtDelete.addActionListener(new ActionListener() {
 
            @Override
            public void actionPerformed(ActionEvent arg0) {
                int key = Integer.parseInt(jtField.getText());
                if (root.search(key) != null) {
                    root.remove(key);
                    treeView.repaint();
                }
                else
                    JOptionPane.showMessageDialog(null, key + " is not in the tree");
                jtField.setText("");
            }
        });
 
        jbtGenerate.addActionListener(new ActionListener(){
 
            @Override
            public void actionPerformed(ActionEvent arg0) {
                try {
                    root = BinaryTree.randIntTree(Integer.parseInt(
                            JOptionPane.showInputDialog(null, "Please enter the number of nodes",
                                    "Randomly generate a tree", JOptionPane.INFORMATION_MESSAGE)));
                    treeView.repaint();
                } catch (Exception e) {
                    e.printStackTrace();
                }
            }});
    }
    class TreeView extends JPanel {
        private static final long serialVersionUID = 1L;
        private int radius = 22;
        private int vGap = 50;
        private final Font GOODFONT = new Font("Arial", Font.BOLD, 15);
//        private Random rng = new Random();
        public TreeView(){
            this.setBackground(new Color(255, 255, 255));
        }
//        private Color rngColor(int lower, int upper){
//            int r = upper - lower;
//            return new Color(lower + rng.nextInt(r), lower + rng.nextInt(r), lower + rng.nextInt(r));
//        }
        public void paintComponent(Graphics g) {
            super.paintComponent(g);
            if (root.getRoot() != null) {
                diaplay(g, root.getRoot(), getWidth() / 2, 30, getWidth() / 4);
            }
        }
 
        private void diaplay(Graphics g, BinaryTreeNode<Integer, Integer> root, int x, int y, int hGap) {
            g.setColor(Color.ORANGE);
            g.setFont(GOODFONT);
            g.fillOval(x - radius, y - radius, 2 * radius, 2 * radius);
            g.setColor(Color.BLACK);
            int strWidth = g.getFontMetrics().stringWidth(root.getKey().toString());
            g.drawString(root.getKey().toString(), x-strWidth/2, y + 6);
            if (root.getLeft() != null) {
                connectLeftChild(g, x - hGap, y + vGap, x, y);
                diaplay(g, root.getLeft(), x - hGap, y + vGap, hGap / 2);
            }
            if (root.getRight() != null) {
                connectRightChild(g, x + hGap, y + vGap, x, y);
                diaplay(g, root.getRight(), x + hGap, y + vGap, hGap / 2);
            }
        }
 
        private void connectRightChild(Graphics g, int x1, int y1, int x2, int y2) {
            double d = Math.sqrt(vGap * vGap + (x2 - x1) * (x2 - x1));
            int x11 = (int) (x1 - radius * (x1 - x2) / d);
            int y11 = (int) (y1 - radius * vGap / d);
            int x21 = (int) (x2 + radius * (x1 - x2) / d);
            int y21 = (int) (y2 + radius * vGap / d);
            g.drawLine(x11, y11, x21, y21);
        }
        private void connectLeftChild(Graphics g, int x1, int y1, int x2, int y2) {
            double d = Math.sqrt(vGap * vGap + (x2 - x1) * (x2 - x1));
            int x11 = (int) (x1 + radius * (x2 - x1) / d);
            int y11 = (int) (y1 - radius * vGap / d);
            int x21 = (int) (x2 - radius * (x2 - x1) / d);
            int y21 = (int) (y2 + radius * vGap / d);
            g.drawLine(x11, y11, x21, y21);
        }
    }
}
