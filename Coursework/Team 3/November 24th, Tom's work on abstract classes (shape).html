Abstract class Shape

package shapes;
 
public abstract class Shape {
 
    public String name;
    private String type;
    public Shape(String shapeName, String typeName){
        name=shapeName;
        type=typeName;
    }
 
    public String getName(){
        return name;
    }
    public void changeName(String newName){
        name=newName;
    }
    public String getType(){
        return type;
    }
    public abstract double getPerimeter();
    public abstract void draw(Screen sc);
    public abstract String toString();
    public abstract Point[] getVertices();
 
}

package shapes;
 
public abstract class Polygons extends Shape{
    private int sides;
    public Polygons(int s, String name, String type){
        super(name, type);
        sides=s;
    }
 
    public int getSides(){
        return sides;
    }
    public abstract double[] getSidelength();
 
}

Subclass Regular_Polygon

package shapes;
 
public class Regular_Polygon extends Polygons{
    private double sidelength;
    private double radius;
    private double internalAngle;
    private double rotation;
    private Point center;
    private Point[] vertices;
 
    public Regular_Polygon(String name,Point center, int sides, double rad, double rt){
        super(sides, name, "Regular_Polygon");
        rotation = Math.toRadians(rt);
        internalAngle = Math.toRadians(360/sides);
        this.center=center;
        radius = rad;
        vertices = new Point[sides];
        for (int i = 0;i<sides;i++){
            vertices[i] = new Point(center.getX()+rad*Math.cos(i*internalAngle+rotation),
                    center.getY()+rad*Math.sin(i*internalAngle+rotation));
        }
        sidelength = vertices[0].getDistance(vertices[1]);
    }
    public double getPerimeter(){
        return super.getSides()*sidelength;
    }
    public void draw(Screen sc){
        int[] x = new int[vertices.length];
        int[] y = new int[vertices.length];
        for (int i = 0; i<vertices.length;i++){
            x[i]=(int)Math.round(vertices[i].getX());
            y[i]=(int)Math.round(vertices[i].getY());
        }
        sc.addPolygon(x,y,vertices.length);
        sc.repaint();
 
    }
    public double[] getSidelength(){
        return new double[] {sidelength};
    }
    public Point[] getVertices(){
        return vertices;
    }
    public double getArea(){
        return sidelength*radius*vertices.length/2;
    }
    public double getInternalAngle(){
        return internalAngle;
    }
    public Point getCenter(){
        return center;
    }
    public String toString(){
        return String.format("Regular polygon named %s of %d sides, center %s, radius %.1f and rotation %.1f degrees",
                super.getName(), vertices.length,center.toString(), radius, Math.toDegrees(rotation));
    }
 
}
 
 
 

Subclass Irregular_Polygon

package shapes;
 
public class Irregular_Polygon extends Polygons{
 
    private double[] sidelength;
    private Point[] vertices;
 
    public Irregular_Polygon(String name, Point[] pts){
        super(pts.length, name, "Irregular_Polygon");
        sidelength = new double[pts.length];
        for (int i = 0; i<pts.length-1; i++){
            sidelength[i]=pts[i].getDistance(pts[i+1]);
        }
        sidelength[sidelength.length-1]=pts[pts.length-1].getDistance(pts[0]);
        vertices = pts;
    }
    public double getPerimeter(){
        double peri = 0;
        for (double sl : sidelength){
            peri+= sl;
        }
        return peri;
    }
    public void draw(Screen sc){
        int[] x = new int[vertices.length];
        int[] y = new int[vertices.length];
        for (int i = 0; i<vertices.length;i++){
            x[i]=(int)Math.round(vertices[i].getX());
            y[i]=(int)Math.round(vertices[i].getY());
        }
        sc.addPolygon(x,y,vertices.length);
        sc.repaint();
    }
    public double[] getSidelength(){
        return sidelength;
    }
    public Point[] getVertices(){
        return vertices;
    }
    public String toString(){
        String vertiStr = "";
        for (Point v : vertices){
            vertiStr+= v.toString()+" ";
        }
        return String.format("Irregular polygon named %s of %d sides and vertices %s",
                super.getName(), vertices.length, vertiStr);
    }
 
}
 
 

Subclass Line

package shapes;
 
public class Line extends Shape {
    private Point start;
    private Point end;
    private double length;
    public Line(String name, Point st, Point ed){
        super(name, "Line");
        start=st;
        end=ed;
        length = Point.getDistance(start, end);
    }
 
    public void draw(Screen sc){
        sc.addLine(this);
        sc.repaint();
    }
    public Point[] getVertices(){
        return new Point[] {start, end};
    }
    public double getPerimeter(){
        return length;
    }
    public String toString(){
        return String.format("Line between %s and %s of length %.1f", start.toString(), end.toString(), length);
    }
}

Class Point

package shapes;
 
public class Point {
    private double x;
    private double y;
    private double z;
    private String type;
    public Point(){
        x=0;y=0;z=0;
        type="3D";
    }
    public Point(double x1, double y1){
        x=x1;y=y1;z=0;
        type="2D";
    }
    public Point(double x1, double y1, double z1){
        x=x1;y=y1;z=z1;
        type="3D";
    }
 
    public double getX(){
        return x;
    }
    public double getY(){
        return y;
    }
    public double getZ(){
        return z;
    }
    public static double getDistance(Point p, Point q){
        return Math.sqrt(Math.pow(p.getX()-q.getX(),2)+Math.pow(p.getY()-q.getY(),2)+Math.pow(p.getZ()-q.getZ(),2));
    }
 
    public double getDistance(Point p){
        return Math.sqrt(Math.pow(p.getX()-x,2)+Math.pow(p.getY()-y,2)+Math.pow(p.getZ()-z,2));
    }
 
    public double getDistance(){
        return Math.sqrt(x*x+y*y+z*z);
    }
    public String toString(){
        if (type.equals("2D"))
            return String.format("(%.1f,%.1f)", x,y);
        else
            return String.format("(%.1f,%.1f,%.1f)", x,y,z);
    }
 
}
 

Class Screen

package shapes;
import javax.swing.*;
import java.awt.Graphics;
import java.awt.Polygon;
import java.util.*;
 
public class Screen extends JFrame{
    private ArrayList<Polygon> PolygonList = new ArrayList<Polygon>();
    private ArrayList<Circle> CircleList = new ArrayList<Circle>();
    private ArrayList<Line> LineList = new ArrayList<Line>();
    private String name;
    public void addPolygon (int[] x, int[] y, int s){
        PolygonList.add(new Polygon(x, y, s));
    }
    public void addCircle(Circle c){
        CircleList.add(c);
    }
    public void addLine(Line c){
        LineList.add(c);
    }
    public Screen(String n, int wid, int hei) {
        name=n;
        setTitle(n);
        this.setVisible(true);
        this.setDefaultCloseOperation(JFrame.HIDE_ON_CLOSE);
        this.setSize(wid, hei);
    }
    public String getName(){
        return name;
    }
    public void changeName(String newName){
        name=newName;
    }
    public void paint(Graphics g){
        super.paint(g);
        for (Polygon py : PolygonList){
            g.drawPolygon(py);
        }
        for (Line l : LineList){
            g.drawLine((int)Math.round(l.getVertices()[0].getX()), (int)Math.round(l.getVertices()[0].getY()),
                    (int)Math.round(l.getVertices()[1].getX()), (int)Math.round(l.getVertices()[1].getY()));
        }
        for (Circle c : CircleList){
            g.drawOval((int)Math.round(c.getVertices()[0].getX()-c.getRadius()),
                    (int)Math.round(c.getVertices()[0].getY()-c.getRadius())
                    , (int)Math.round(c.getRadius()*2), (int)Math.round(c.getRadius()*2));
        }
    }
    public String toString(){
        return "Screen named "+getName()+" of width "+this.getWidth()+" and height "+this.getHeight();
    }
    public void show(){
        super.show();
    }
}
