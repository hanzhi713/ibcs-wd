def Serial_search_algorithm():
    handler = open("Book.txt")
    SearchBook = input("Enter the book title")
    IsFound = False
 
    while 1:
        NextBook = " ".join(handler.readline().split())
        if NextBook == SearchBook:
            IsFound = True
            print("Found")
            break
        if IsFound or len(NextBook) == 0:
            break
    if not IsFound:
        print("Book title was NOT FOUND")
 
 
def BinarySearch(ThisArray, FindValue, Low, High):
    if High < Low:
        return -1
    else:
        Middle = int((High + Low)/2)
        if ThisArray[Middle] > FindValue:
            return BinarySearch(ThisArray, FindValue, Low, Middle - 1)
        else:
            if ThisArray[Middle] < FindValue:
                return BinarySearch(ThisArray, FindValue, Middle + 1, High)
            else:
                return Middle
