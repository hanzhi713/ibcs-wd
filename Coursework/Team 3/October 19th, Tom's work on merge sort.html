Code for comparison of the time complexity of iterative merge sort and recursive merge sort

def split_and_merge(xs):
    xs[:] = [[item] for item in xs]
    while 1:
        index = 0
        while 1:
            xs[index] = merge(xs[index], xs[index + 1])
            del xs[index+1]
            index += 1
            if index >= len(xs)-1:
                break
        if len(xs) == 1:
            xs[:] = xs[0]
            return
 
 
def mergeSort(alist):
    if len(alist) > 1:
        mid = len(alist)//2
        lefthalf = alist[:mid]
        righthalf = alist[mid:]
        mergeSort(lefthalf)
        mergeSort(righthalf)
        alist[:] = merge(lefthalf, righthalf)
 
 
def merge(xs, ys):
    result = []
    xi = 0
    yi = 0
 
    while True:
        if xi >= len(xs):
            result.extend(ys[yi:])
            return result
 
        if yi >= len(ys):
            result.extend(xs[xi:])
            return result
 
        if xs[xi] <= ys[yi]:
            result.append(xs[xi])
            xi += 1
        else:
            result.append(ys[yi])
            yi += 1
 
 
import random
import time
start = 10000
end = 100000
step = 1000
sample_num = 3
time_iter = []
time_recur = []
for i in range(start, end+step, step):
    test_sample = list(range(i))
    time_samples_recur = []
    time_samples_iter = []
    for a in range(sample_num):
        random.shuffle(test_sample)
 
        t0 = time.clock()
        split_and_merge(test_sample[:])
        time_samples_iter.append(time.clock() - t0)
 
        t0 = time.clock()
        mergeSort(test_sample[:])
        time_samples_recur.append(time.clock() - t0)
 
    average_iter = round(sum(time_samples_iter) / sample_num, 5)
    average_recur = round(sum(time_samples_recur) / sample_num, 5)
    print("The list has {} items, it takes {} seconds in average by iterative merge sort.".format(i, average_iter))
    time_iter.append(average_iter)
    print("The list has {} items, it takes {} seconds in average by recursive merge sort.".format(i, average_recur))
    time_recur.append(average_recur)
 
import matplotlib.pyplot as plt
plt.figure()
plt.hold(True)
iter_graph, = plt.plot(list(range(start, end+step, step)), time_iter,'b', label='Iterative')
recur_graph, = plt.plot(list(range(start, end+step, step)), time_recur, 'y', label='Recursive')
plt.legend(handles=[iter_graph, recur_graph])
plt.show()
