import java.util.*;
import java.io.*;
public class Grade_Statistics {
 
    public static void main(String[] args) {
        Scanner data = null;
           char[] grade = {'A','B','C','D','E'};
           int[] num = new int[5];
        try {
                data = new Scanner(new File("C:/Users/Zhou Xuqi/Desktop/JavaProjects/Test/src/test.dat"));
        }
        catch ( IOException e) {
            System.out.println("Sorry but I was unable to open your data file");
            e.printStackTrace();
            System.exit(0);
        }
        while(data.hasNextInt()) {
            int gr = data.nextInt();
            if (90<=gr && gr<=100){
                num[0]+=1;
            }
            else if (80<=gr && 90>gr){
                num[1]+=1;
            }
 
            else if (70<=gr&& 80>gr){
                num[2]+=1;
            }
            else if (60<=gr&& 70>gr){
                num[3]+=1;
            }
            else if (0<=gr && 60>gr){
                num[4]+=1;
            }
 
        }
        for(int i=0;i<5;i++) {
            System.out.print(grade[i]+" |");
            for (int n=0;n<num[i];n++){
                System.out.print("*");
            }
            System.out.printf("  (%d)%n",num[i]);
        }
 
    }
 
}
