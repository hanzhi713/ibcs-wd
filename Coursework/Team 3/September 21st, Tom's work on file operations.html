Q2: Reverse every word in a file

def reverse_words(file_path):
    handler = open(file_path,"r")
    new_list_words = []
    while 1:
        tempstr = handler.readline()
        if len(tempstr) == 0:
            break
        tempstr = tempstr[::-1]
        new_list_words.append(tempstr)
    return new_list_words
 
def write_lines(lines,destin):
    handler = open(destin,"w")
    for line in lines:
        handler.write(line)
 
write_lines(reverse_words("vocab.txt"),"reversed_vocab")

def read_all_words(file_path):

def replace(word,target):

    while 1:
        index = new_.find("Alice")
        if index > -1:
            new_ = new_[:index] +word + new_[index+5:]
