import matplotlib.pyplot as plt
import random
import time
 
 
def insertionSort(alist):
    n = len(alist)
    for i in range(1,n):
        cur_value = alist[i]
        j = i
        while j > 0 and cur_value < alist[j-1]:
            alist[j] = alist[j-1]
            j -= 1
        alist[j] = cur_value
 
 
def selectionSort(alist):
    n = len(alist)
    for i in range(0, n-1):
        minsub = i
        for j in range(i+1, n):
            if alist[j] < alist[minsub]:
                minsub = j
        (alist[minsub], alist[i]) = (alist[i], alist[minsub])
 
 
def bubbleSort(alist):
    n = len(alist)
    for i in range(0, n-1):
        for j in range(0, n-i-1):
            if alist[j] > alist[j+1]:
                (alist[j], alist[j+1]) = (alist[j+1], alist[j])

def merge(xs, ys):
    """ merge sorted lists xs and ys. Return a sorted result """
    result = []
    xi = 0
    yi = 0
    while True:
        if xi >= len(xs):          # If xs list is finished,
            result.extend(ys[yi:]) # Add remaining items from ys
            return result          # And we're done.
        if yi >= len(ys):          # Same again, but swap roles
            result.extend(xs[xi:])
            return result
        # Both lists still have items, copy smaller item to result.
        if xs[xi] <= ys[yi]:
            result.append(xs[xi])
            xi += 1
        else:
            result.append(ys[yi])
            yi += 1
 
 
def CompRunningTimeWithPlot():
    rng = random.Random()
    timeSeqInsertion = []
    timeSeqSelection = []
    timeSeqBubble = []
    timeSeqMerge = []
    start = 1000
    end = 5000
    step = 500
    for i in range(start, end, step):
        alist = list(range(i))
        rng.shuffle(alist)
 
        t0 = time.clock()
        insertionSort(alist[:])
        t1 = time.clock()
        print("The list has {} items, it takes {} seconds by insertion sort.".format(i, t1-t0))
        timeSeqInsertion.append(t1-t0)
 
        t0 = time.clock()
        selectionSort(alist[:])
        t1 = time.clock()
        print("The list has {} items, it takes {} seconds by Selection sort.".format(i, t1 - t0))
        timeSeqSelection.append(t1-t0)
 
        t0 = time.clock()
        bubbleSort(alist[:])
        t1 = time.clock()
        print("The list has {} items, it takes {} seconds by Bubble sort.".format(i, t1 - t0))
        timeSeqBubble.append(t1-t0)
 
        t0 = time.clock()
        mergeSort(alist[:])
        t1 = time.clock()
        print("The list has {} items, it takes {} seconds by Merge sort.".format(i, t1 - t0))
        timeSeqMerge.append(t1 - t0)
 
    plt.figure()
    plt.hold(True)
    plt.plot(list(range(start, end, step)), timeSeqInsertion,'r')
    plt.plot(list(range(start, end, step)), timeSeqSelection,'g')
    plt.plot(list(range(start, end, step)), timeSeqBubble,'b')
    plt.plot(list(range(start, end, step)), timeSeqMerge, 'y')
    plt.show()
