import time
import random
def guess_number(lower,upper,truth):
    count = 0
    while 1:
        count += 1
        num = round((lower + upper) / 2)
        if num > truth:
            #print(num,"is too high")
            upper = num
        elif num < truth:
            #print(num, "is too low")
            lower = num
        else:
            #print("Correct! The number is {0}. {1} trials in total".format(num,count))
            return
