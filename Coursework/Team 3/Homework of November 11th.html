import java.util.*;
 
class Matrix{
 
    public static double[][] DiagonalMatrix(int n, int...args){
        int[] e = new int[2];
        if (args.length==1){
            e[0]=e[1]=args[0];
        }
        else if (args.length>=2){
            e[0]=args[0];
            e[1]=args[1];
        }
        double[][] matrix = new double[n][n];
        for (int x =0;x<n;x++){
            for (int y =0;y<x;y++){
                matrix[x][y]=e[0];
            }
            matrix[x][x]=x+1;
            for (int y =x+1;y<n;y++){
                matrix[x][y]=e[1];
            }
        }
        return matrix;
    }
    public static void printMatrix(double[][] mx, int len, int precision){
         String format_text="%"+(precision+len+2)+"."+precision+"f  ";
         for (double[] x : mx){
             for (double y: x){
                 System.out.printf(format_text,y);
             }
             System.out.println();
             System.out.println();
         }
    }
}
 
class GuessNumber{
 
    private static Random rng = new Random();
    private static Scanner sc = new Scanner(System.in);
 
    public static void Game(int lower, int upper){
        System.out.printf("Lower bound: %d\nUpper bound: %d\n", lower,upper);
        int num = rng.nextInt(upper-lower)+lower;
        System.out.println("Guess the number:");
        int user_input = sc.nextInt();
        int counter = 0;
        while (true){
            counter++;
            if (user_input == num){
                System.out.println("You win! Number of trials:"+counter);
                break;
            }
            else{
                if (user_input<num){
                    System.out.println("Too low!");
                }
                else{
                    System.out.println("Too high");
                }
            }
            System.out.println("Guess the number:");
            user_input = sc.nextInt();
        }
    }
}
public class Homework_11_11 {
 
    public static void main(String[] args) {
        Matrix.printMatrix(Matrix.DiagonalMatrix(10,0,-1),2,2);
        GuessNumber.Game(1,10);
    }
 
}
