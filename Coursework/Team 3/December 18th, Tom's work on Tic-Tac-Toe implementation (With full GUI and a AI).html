    private void reset(String whoFirst){
        placement = new int[3][3];
        currentTurn = whoFirst;
 
        currentCell = null;
        currentPoint = new Point(-1,-1);
        if (AI!=null){
            if (whoFirst.equals("○")){
                isAI = true;
                currentTurn = "×";
                next();
            }
            else{
                isAI = false;
 
            }
        }
        else
            isAI = false;
        repaint();
    }
    public void next(){
        int sumRow = 0;
        int sumColumn = 0;
        int sumMjDiagonal = 0;
        int sumMnDiagonal = 0;
        boolean isFulfilled = true;
 
        for (int i = 0;i<3;i++){
            for (int j = 0;j<3;j++){
                if (placement[i][j]==0) isFulfilled = false;
                sumRow += placement[i][j];
                sumColumn += placement[j][i];
                if (i==j) sumMjDiagonal += placement[i][j];
                if (i==Math.abs(j-2)) sumMnDiagonal += placement[i][j];
            }
            if (sumRow == 3 || sumColumn== 3) {
                currentTurn = "";
                repaint();
                JOptionPane.showMessageDialog(Grid.this, "○ wins!", "Tic-Tac-Toe", JOptionPane.INFORMATION_MESSAGE);
                return;
            }
            if (sumRow == -3 || sumColumn == -3){
                currentTurn = "";
                repaint();
                JOptionPane.showMessageDialog(Grid.this, "× wins!","Tic-Tac-Toe", JOptionPane.INFORMATION_MESSAGE);
                return;
            }
            sumRow = 0;
            sumColumn = 0;
        }
        if (sumMjDiagonal == -3 || sumMnDiagonal == -3){
            currentTurn = "";
            repaint();
            JOptionPane.showMessageDialog(Grid.this, "× wins!", "Tic-Tac-Toe", JOptionPane.INFORMATION_MESSAGE);
            return;
        }
        if (sumMjDiagonal == 3 || sumMnDiagonal == 3){
            currentTurn = "";
            repaint();
            JOptionPane.showMessageDialog(Grid.this, "○ wins!", "Tic-Tac-Toe", JOptionPane.INFORMATION_MESSAGE);
            return;
        }
        if (isFulfilled){
            currentTurn = "";
            repaint();
            JOptionPane.showMessageDialog(Grid.this, "draw!", "Tic-Tac-Toe", JOptionPane.INFORMATION_MESSAGE);
            return;
        }
 
        if (currentTurn.equals("×"))
            currentTurn = "○";
        else
            currentTurn = "×";
        if (AI != null && isAI == true){
                isAI = false;
                int[] AIplacement;
                AIplacement = AI.place(placement, -1);
                currentPoint.x = AIplacement[0];
                currentPoint.y = AIplacement[1];
                currentCell=null;
                place();
        }
 
    }
    public void place(){
        if (currentPoint.x>-1){
            if (currentTurn.equals("×"))
                placement[currentPoint.x][currentPoint.y] = -1;
            else
                placement[currentPoint.x][currentPoint.y] = 1;
            currentCell=null;
            currentPoint.x=-1;
            next();
            repaint();
        }
    }

Class TTT_AI —— Using Game Rule (Perfect and unbeatable)

package Tic_Tac_Toe;
 
public class TTT_AI {
 
    public int evaluate(int[][] placement, int i, int j, int who){
        int[] sums = new int[4];
        for (int num: placement[i])
            sums[0] += num;
        for (int x = 0;x<3;x++ )
            sums[1] += placement[x][j];
 
        if (i==j)
            for (int x = 0; x<3;x++)
                sums[2] += placement[x][x];
        if (i==(2-j))
            for (int x = 0; x<3;x++)
                sums[3] += placement[x][2-x];
        int weight = 0;
        for (int num: sums){
            if (num==2 * who) weight += 1000;
            else if (num==-2* who) weight += 10000;
            else if (num==1* who) weight += 10;
            else if (num==-1* who) weight += 6;
            else if (num==0) weight += 4;
        }
        if (i==1 && j== 1)
            weight += 3;
        else if (i==j || i==(2-j))
            weight += 2;
        else
            weight += 1;
        return weight;
    }
    public int[] place(int[][] placement, int who){
        int[][] emptyCells = new int[3][3];
        for (int i = 0;i<3;i++){
            for (int j = 0;j<3;j++){
                if (placement[i][j] == 0){
                    emptyCells[i][j] = evaluate(placement, i, j, who);
                }
            }
        }
        int[] result = new int[2];
        int max = 0;
        for (int i = 0;i<3;i++)
            for (int j = 0;j<3;j++){
                if (emptyCells[i][j]>0 && emptyCells[i][j]>max){
                    max = emptyCells[i][j];
                    result = new int[]{i,j};
                }
            }
        return result;
    }
}
 


Class TTT_AI —— Using Minimax Algorithm (Imperfect and beatable, probably due to the imperfections in my recursive structure)


package Tic_Tac_Toe;
 
public class TTT_AI {
    private final int depth = 9;
    public int statistic(int[][] placement, int num){
        int sumRow = 0;
        int sumColumn = 0;
        int sumMjDiagonal = 0;
        int sumMnDiagonal = 0;
        int stat = 0;
        for (int i = 0;i<3;i++){
            for (int j = 0;j<3;j++){
                sumRow += placement[i][j];
                sumColumn += placement[j][i];
                if (i==j) sumMjDiagonal += placement[i][j];
                if (i==Math.abs(j-2)) sumMnDiagonal += placement[i][j];
            }
            if (sumRow == num)
                stat+=1;
            if(sumColumn== num)
                stat+=1;
            sumRow = 0;
            sumColumn = 0;
        }
        if (sumMjDiagonal == num) stat += 1;
        if(sumMnDiagonal == num) stat += 1;
        return stat;
 
    }
    public int[][] clone(int[][] placement){
        int[][] newPlacement = new int[placement.length][];
        for (int row = 0; row<placement.length;row++)
            newPlacement[row] = placement[row].clone();
        return newPlacement;
    }
    public int[][] fillWith(int[][] placement ,int num){
        int[][] newPlacement = clone(placement);
        for (int i = 0;i<3;i++)
            for (int j = 0;j<3;j++)
                if (placement[i][j]==0) newPlacement[i][j] = num;
        return newPlacement;
    }
    public int evaluate(int[][] placement){
        int[][] placementO = fillWith(placement, 1);
        int[][] placementX = fillWith(placement, -1);
        return (statistic(placementX, -3) - statistic(placementO, 3));
    }
    public boolean isFull(int[][] placement){
        int count = 0;
        for (int i = 0;i<3;i++)
            for (int j = 0;j<3;j++)
                if (placement[i][j]==0) count += 1;
        return !(count>0);
    }
    public int calculate(int[][] placement, int recursionDepth, int who){
        int[][] newPlacement;
        ArrayList<Integer> numC = new ArrayList<Integer>();
        for (int i = 0;i<3;i++){
            for (int j = 0;j<3;j++){
                if (placement[i][j] == 0){
                    newPlacement = clone(placement);
                    newPlacement[i][j] = who;
                    if (recursionDepth == 0 || isFull(newPlacement)){
                        return evaluate(newPlacement);
                    }
                    else
                        numC.add(calculate(newPlacement, recursionDepth-1, -who));
                }
            }
        }
 
        if (who == 1){
            int min = 100;
            for (Integer num: numC){
                if (num.intValue()<min) min = num.intValue();
            }
            return min;
        }
        else{
            int max = -100;
            for (Integer num: numC){
                if (num.intValue()>max) max = num.intValue();
            }
            return max;
        }
    }
    public int[] place(int[][] placement, int who){
 
        ArrayList<int[]> ways = new ArrayList<int[]>();
        for (int i = 0;i<3;i++){
            for (int j = 0;j<3;j++){
                if (placement[i][j] == 0){
                    int[][] newPlacement = clone(placement);
                    newPlacement[i][j] = who;
                    ways.add(new int[]{i,j,calculate(newPlacement, depth-1, -who)});
                }
            }
        }
        int[] maxPos = new int[]{-1,-1,-100};
            for (int[] a : ways){
                if (a[2]>maxPos[2]){
                    maxPos[2] = a[2];
                    maxPos[0] = a[0];
                    maxPos[1] = a[1];
            }
        }
        return maxPos;
    }
