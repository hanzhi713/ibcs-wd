June 4th Tom's work on drawing functions


The following method of drawing functions is ONLY suitable for continuous functions. To get started, you need to define the scale of cartesian coordinate, enter the first derivative, enter the coordinate of any point and define the domain. Please notice that the expression you would like to enter must be acceptable to the Python's internal module "eval".


To see another way of drawing functions (using the expression and plot all points directly), please click me.

import turtle
import math
wn = turtle.Screen()
Tom = turtle.Turtle()
wn.bgcolor("black")
Tom.color("white")
Tom.pensize(1)
Tom.speed(1000)
Tom.hideturtle()
scale = float(input ("Please define the length of one unit (in pixels)"))
xLength = int(input ("Please enter the length of x-axis, in pixels"))
yLength = int(input ("Please enter the length of y-axis, in pixels"))
function = input ("Please type the first derivative")
xcoor = float(input ("Please input the x-coordinate of any point on the function"))
ycoor = float(input ("Please input the y-coordinate of any point on the function"))
lowerlimit = float(input ("Please type the lower limit of domain (must be smaller than your given x-coordinate)"))
upperlimit = float(input ("Please type the upper limit of domain (must be greater than your given x-coordinate)"))
i = 0
Tom.penup()
Tom.right(90)
Tom.forward(16)
Tom.write(0, False, align="right", font=("Calibri", 9, "bold"))
Tom.left(90)
Tom.goto(0,0)
Tom.pendown()
def DrawAxes(Scale,Direction,XorYaxis,length):
    howmany = length // Scale
    i = 0
    while 1:
        Tom.forward(Direction * Scale)
        Tom.left(Direction * 90)
        Tom.forward(Direction * 5)
        Tom.forward(Direction * -5)
        Tom.right(Direction * 90)
        i = i + 1
        if i % 5 == 0:
            if XorYaxis=="X":
                Tom.penup()
                Tom.right(90)
                Tom.forward(16)
                Tom.write(i, False, align="center", font=("Calibri", 9, "bold"))
                Tom.forward(-16)
                Tom.left(90)
                Tom.pendown()
            else:
                Tom.penup()
                Tom.right(-90)
                Tom.forward(5)
                Tom.left(90)
                Tom.forward(7)
                Tom.write(i, False, align="right", font=("Calibri", 9, "bold"))
                Tom.forward(-7)
                Tom.left(-90)
                Tom.forward(-5)
                Tom.left(-90)
                Tom.pendown()
        if i > howmany:
            break
DrawAxes(scale,1,"X",xLength / 2)
Tom.forward(20)
Tom.stamp()
Tom.penup()
Tom.right(90)
Tom.forward(20)
Tom.write("x",False,align="right",font=("Calibri",15,"bold"))
Tom.goto(0,0)
Tom.left(90)
Tom.pendown()
DrawAxes(scale,-1,"X",xLength / 2)
Tom.goto(0,0)
Tom.left (90)
DrawAxes(scale,1,"Y",yLength / 2)
Tom.forward(20)
Tom.stamp()
Tom.penup()
Tom.right(-180)
Tom.forward(20)
Tom.left(-90)
Tom.forward(5)
Tom.write("y",False,align="right",font=("Calibri",15,"bold"))
Tom.goto(0,0)
Tom.left(-90)
Tom.pendown()
DrawAxes(scale,-1,"Y",yLength / 2)
Tom.goto(0,0)
Tom.penup()
Tom.goto(xcoor*scale,ycoor*scale)
Tom.pendown()
Tom.right(90)
def DrawingFunc(Lowerlimit,Upperlimit,Direction):
    angle = 0
    for a in range(Lowerlimit, Upperlimit,Direction):
        try:
            x = a / scale
            gradient = eval(function)
            oldangle = angle
            angle = (math.atan(gradient) / math.pi) * 180
            Tom.left(angle - oldangle)
            Tom.forward(1 / math.cos(math.atan(gradient)))
        except:
            None
    return angle
angle = DrawingFunc(round(xcoor * scale),round(upperlimit * scale+1),1)
Tom.stamp()
Tom.penup()
Tom.goto (xcoor * scale, ycoor * scale)
Tom.pendown()
Tom.left(180 - angle)
DrawingFunc(round(xcoor * scale), round(lowerlimit * scale - 1), -1)
Tom.stamp()
wn.mainloop()
