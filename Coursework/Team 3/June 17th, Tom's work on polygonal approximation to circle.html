June 17th, Tom's work on polygonal approximation to circle #TomsProjects

Approximation to circle by drawing polygons with increasing sides.

import turtle
Window = turtle.Screen()
Tom = turtle.Turtle()
Tom.hideturtle()
Tom.speed(0)
xcorlist = []
ycorlist = []
index = -1
for i in range(3,100):
    Tom.penup()
    Tom.forward(300)
    xcorlist.append(Tom.xcor())
    ycorlist.append(Tom.ycor())
    index +=1
    Tom.backward(300)
    for sides in range(0,i):
        Tom.left(360 / i)
        Tom.forward(300)
        xcorlist.append(Tom.xcor())
        ycorlist.append(Tom.ycor())
        Tom.pendown()
        Tom.goto(xcorlist[index],ycorlist[index])
        index +=1
        Tom.goto(xcorlist[index], ycorlist[index])
        Tom.penup()
        Tom.backward(300)
Window.mainloop()
