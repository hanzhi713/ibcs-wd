def DrawAxes(Scale,Direction,XorYaxis,length):
    howmany = length // Scale
    i = 0
    while 1:
        Tom.forward(Direction * Scale)
        Tom.left(Direction * 90)
        Tom.forward(Direction * 5)
        Tom.forward(Direction * -5)
        Tom.right(Direction * 90)
        i = i + 1
        if i % 5 == 0:
            if XorYaxis=="X":
                Tom.penup()
                Tom.right(90)
                Tom.forward(16)
                Tom.write(i, False, align="center", font=("Calibri", 9, "bold"))
                Tom.forward(-16)
                Tom.left(90)
                Tom.pendown()
            else:
                Tom.penup()
                Tom.right(-90)
                Tom.forward(5)
                Tom.left(90)
                Tom.forward(7)
                Tom.write(i, False, align="right", font=("Calibri", 9, "bold"))
                Tom.forward(-7)
                Tom.left(-90)
                Tom.forward(-5)
                Tom.left(-90)
                Tom.pendown()
        if i > howmany:
            break
