Two ways of drawing Pythagora's tree


Recursion:

import turtle
import math
sqrt2 = math.sqrt(2)
Tom = turtle.Turtle()
Tom.speed(0)
wn = turtle.Screen()
Tom.hideturtle()
def DrawSquare(sidelength,x,y,rotation,oddore):
    Tom.setheading(rotation+90)
    Tom.penup()
    Tom.goto(x,y)
    Tom.pendown()
    xcorlist = []
    ycorlist = []
    oddorelist = []
    flag = True
    if sidelength <= 15:
        return
    for i in range(4):
        Tom.forward(sidelength)
        Tom.right(90)
        if oddore % 2 == 0:
            flag = True
            if i == 0:
                xcorlist.append(Tom.xcor())
                ycorlist.append(Tom.ycor())
                oddorelist.append(oddore+2)
            elif i == 1:
                xcorlist.append(Tom.xcor())
                ycorlist.append(Tom.ycor())
                oddorelist.append(oddore + 1)
        else:
            flag = False
            if i == 1:
                xcorlist.append(Tom.xcor())
                ycorlist.append(Tom.ycor())
                oddorelist.append(oddore + 1)
            elif i == 2:
                xcorlist.append(Tom.xcor())
                ycorlist.append(Tom.ycor())
                oddorelist.append(oddore + 2)
    if flag == True:
        DrawSquare(sidelength / sqrt2 , xcorlist[0],ycorlist[0],rotation + 45,oddorelist[0])
        DrawSquare(sidelength / sqrt2, xcorlist[1], ycorlist[1], rotation + 45, oddorelist[1])
    else:
        DrawSquare(sidelength / sqrt2, xcorlist[0], ycorlist[0], rotation - 45, oddorelist[0])
        DrawSquare(sidelength / sqrt2, xcorlist[1], ycorlist[1], rotation - 45, oddorelist[1])
DrawSquare(200,-100,-380,0,0)
wn.mainloop()
