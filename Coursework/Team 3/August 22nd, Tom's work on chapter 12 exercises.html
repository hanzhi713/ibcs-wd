import string
from unit_tester import test
 
 
def cleanword(text, ifaddspace = False):
    cleaned = ""
    for let in text:
        if let not in string.punctuation:
            cleaned += let
        else:
            if ifaddspace == True:
                cleaned += " "
    return cleaned
 
 
def has_dashdash(text):
    if text.find("--") > -1:
        return True
    else:
        return False
 
 
def extract_words(text):
    cleaned = cleanword(text,True)
    cleaned = cleaned.lower()
    return cleaned.split()
 
 
def wordcount(word,text):
    count = 0
    for wd in text:
        if wd == word:
            count += 1
    return count
 
 
def wordset(text):
    result = []
    for i in text:
        if not i in result:
            result.append(i)
    result.sort()
    return result
 
 
def longestword(words):
    tempnum = 0
    for word in words:
        if len(word) > tempnum:
            tempnum = len(word)
    return tempnum
