def add_vectors(u, v):
    sum = []
    for index in range(len(u)):
        sum.append(u[index]+v[index])
    return sum
print(add_vectors([1,2],[1,4]))

def scalar_mult(s, v):
    resul = []
    for item in v:
        resul.append(item*s)
    return resul
print(scalar_mult(5,[1,2]))

def dot_product(u, v):
    resul = 0
    for i in range(len(u)):
        resul += u[i]*v[i]
    return resul
print(dot_product([1,2],[1,4]))

def cross_product(u,v):
    resul = []
    resul.append(u[1] * v[2] - u[2] * v[1])
    resul.append(u[2] * v[0] - u[0] * v[2])
    resul.append(u[0] * v[1] - u[1] * v[0])
    return resul
print(cross_product([1,3,6],[2,4,9]))
