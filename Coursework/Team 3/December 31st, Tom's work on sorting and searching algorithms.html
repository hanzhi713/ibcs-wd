Sorting algorithms

package sorting;
 
import java.util.Arrays;
 
public class Sorting {
    public static void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
 
    public static void selectionSort(int[] arr) {
        int maxPos, max;
        for (int i = 0; i < arr.length - 1; i++) {
            max = arr[i];
            maxPos = i;
            for (int j = i + 1; j < arr.length; j++)
                if (max > arr[j]) {
                    max = arr[j];
                    maxPos = j;
                }
            swap(arr, i, maxPos);
        }
    }
 
    public static void insertionSort(int[] arr) {
        int current;
        for (int i = 0; i < arr.length; i++) {
            int j = i;
            current = arr[j];
            while (j > 0 && arr[j - 1] > current) {
                arr[j] = arr[j - 1];
                j--;
            }
            arr[j] = current;
        }
    }
 
    public static void mergeSort(int[] arr) {
        if (arr.length > 1) {
            int mid = arr.length / 2;
            int[] lefthalf = new int[mid];
            for (int i = 0; i < mid; i++)
                lefthalf[i] = arr[i];
            int[] righthalf = new int[arr.length - lefthalf.length];
            for (int i = mid; i < arr.length; i++)
                righthalf[i - mid] = arr[i];
            mergeSort(lefthalf);
            mergeSort(righthalf);
            int[] sortResult = merge(lefthalf, righthalf);
            for (int i = 0; i < sortResult.length; i++) {
                arr[i] = sortResult[i];
            }
        }
    }
 
    public static int[] merge(int[] a, int[] b) {
        int[] result = new int[a.length + b.length];
        int ai = 0, bi = 0, ri = 0;
        while (true) {
            if (ai >= a.length) {
                for (int i = ri; i < result.length; i++) {
                    result[i] = b[bi];
                    bi++;
                }
 
                return result;
            }
            if (bi >= b.length) {
                for (int i = ri; i < result.length; i++) {
                    result[i] = a[ai];
                    ai++;
                }
                return result;
            }
            if (a[ai] <= b[bi]) {
                result[ri] = a[ai];
                ri++;
                ai++;
            } else {
                result[ri] = b[bi];
                ri++;
                bi++;
            }
        }
    }
 
    public static void quickSort(int[] arr) {
        quickSort(arr, 0, arr.length - 1);
    }
 
    public static void quickSort(int[] arr, int start, int end) {
 
        int i = start, j = end;
        int pivot = arr[i];
        int pi = i;
 
        while (i < j) {
            while (i < j && pivot <= arr[j])
                j--;
            if (i < j) {
                arr[pi] = arr[j];
                pi = j;
            }
 
            while (i < j && arr[i] <= pivot)
                i++;
            if (i < j) {
                arr[pi] = arr[i];
                pi = i;
            }
        }
        arr[pi] = pivot;
        if (i - start > 1) {
            quickSort(arr, 0, i - 1);
        }
        if (end - j > 1) {
            quickSort(arr, j + 1, end);
        }
    }
 
    public static void main(String[] args) {
        int[] c = new int[] { 1, 6, 21, 35, 2, 4, 8, 4 };
        selectionSort(c);
        System.out.println(Arrays.toString(c));
 
        c = new int[] { 1, 6, 21, 35, 2, 4, 8, 4 };
        insertionSort(c);
        System.out.println(Arrays.toString(c));
 
        c = new int[] { 1, 6, 21, 35, 2, 4, 8, 4 };
        mergeSort(c);
        System.out.println(Arrays.toString(c));
 
        c = new int[] { 1, 6, 21, 35, 2, 4, 8, 4 };
        quickSort(c);
        System.out.println(Arrays.toString(c));
 
    }
}
 

Searching algorithms

package sorting;
 
public class Searching {
 
    public static int linearSearch(int[] arr, int target){
        for (int i = 0; i< arr.length;i++){
            if (target==arr[i]) return i;
        }
        return -1;
    }
    public static int binarySearch(int[] arr, int target){
        int lower = 0, upper = arr.length, mid;
        while (true){
            if (lower==upper) return -1;
            mid = (lower+upper)/2;
            if (arr[mid] == target) return mid;
            else if (arr[mid]<target){
                lower = mid+1;
            }
            else{
                upper = mid-1;
            }
        }
 
    }
    public static void main(String[] args) {
        // TODO Auto-generated method stub
        int[] a = {1,3,7,9,12,45,67,77,78,90};
        System.out.println(linearSearch(a,78));
        System.out.println(binarySearch(a,78));
    }
 
}
