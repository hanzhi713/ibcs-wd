def is_sorted(s):
    for index in range(len(s)):
        if index+1>len(s)-1:
            return True
        if s[index+1] < s[index]:
            return False
 
print(is_sorted([1,3,5,6,1]))

def is_anagrams(word1,word2):
    word1 = word1.lower()
    word2 = word2.lower()
    word1list = []
    word2list = []
    for letter in word1:
        word1list.append(letter)
    for letter in word2:
        word2list.append(letter)
    word1list.sort()
    word2list.sort()
    if word1list == word2list:
        return True
    else:
        return False
 
print(is_anagrams("Apple","Paple"))

import random
 
def has_duplicates(list1):
    listcopy = []
    for item in list1:
        if item not in listcopy:
            listcopy.append(item)
        else:
            return True
    return False
print(has_duplicates(["1","ab",4,6,"a"]))
 
def random_birthday_generator(number):
    birthday = []
    for i in range(number):
        birthday.append(random.randrange(1,366))
    return birthday
 
def same_birthday_statistics():
    count = 0
    for i in range(100000):
        if has_duplicates(random_birthday_generator(23)) == True:
            count +=1
    print(count/100000)
 
same_birthday_statistics()
