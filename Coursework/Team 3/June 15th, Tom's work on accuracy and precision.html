import turtle
import random
wn = turtle.Screen()
Tom = turtle.Turtle()
Tom.speed(0)
def DrawingSamples50(FD,FA,Minangle, Maxangle, Mindistance, Maxdistance):
    i = 0
    while i < 50:
        distance = FD+random.uniform(Maxdistance,Mindistance)
        angle = FA+random.uniform(Minangle,Maxangle)
        Tom.penup()
        Tom.left(angle)
        Tom.forward(distance)
        Tom.dot(5, "red")
        Tom.backward(distance)
        Tom.right(angle)
        Tom.pendown
        i = i + 1
for radius in range(20,220,20):
    Tom.circle(radius)
    Tom.write(11 - radius // 20)
    Tom.penup()
    Tom.goto(0,-radius)
    Tom.pendown()
Tom.penup()
Tom.goto(0,20)
Tom.pendown()
while 1:
    try:
        a = input ("What kind of distribution would you like to plot? Enter 1 for accurate sample, 2 for precise sample, 3 for both and 4 for neither.")
        if 1<= int(a) <= 4:break
    except: None
if a == "1":
    DrawingSamples50(0,0,-360,360,-200,200)
elif a == "2":
   DrawingSamples50(random.uniform(-200, 200),random.uniform(0, 360),-15,15,-20,20)
elif a == "3":
    DrawingSamples50(0,0,-360,360,-20,20)
elif a =="4":
    DrawingSamples50(0, 0, random.uniform(-360, 0), random.uniform(0, 360), random.uniform(-200, 0), random.uniform(-0, 200))
wn.mainloop()
