Class OrderedTree

package tree;
 
// universal implementation of a tree
// though it's only partially implemented
// not specifically designed for stations
 
public class OrderedTree {
    private TreeNode root;
 
    public OrderedTree(TreeNode t){
        root = t;
    }
 
    public OrderedTree(String key, Object cargo){
        root = new TreeNode(key, cargo, null, null);
    }
 
    public TreeNode getRoot(){
        return root;
    }
 
    public void add(String key, Object cargo){
        TreeNode current = root;
        while (true){
            if (current.getKey().compareTo(key)>0){
                if (current.Right() != null)
                    current = current.Right();
                else{
                    current.setRight(new TreeNode(key, cargo, null, null));
                    return;
                }
            }
            else if (current.getKey().compareTo(key)<0){
                if (current.Left() != null)
                    current = current.Left();
                else{
                    current.setLeft(new TreeNode(key, cargo, null, null));
                    return;
                }
            }
            else{
                throw new IllegalArgumentException("Duplicated key!!!");
            }
        }
 
    }
    public TreeNode Search(String key){
        TreeNode current = root;
        while (current!=null){
            if (current.getKey().compareTo(key)>0)
                current = current.Right();
            else if (current.getKey().compareTo(key)<0)
                current = current.Left();
            else
                return current;
        }
        return null;
    }
}
 

Class TreeNode

package tree;
 
//universal implementation of a tree node
//not specifically designed for stations
 
public class TreeNode {
    private Object cargo;
    private TreeNode left, right;
    private String key;
    public TreeNode(String key, Object cargo, TreeNode left, TreeNode right){
        this.key = key;
        this.cargo = cargo;
        this.left = left;
        this.right = right;
    }
    public Object getCargo(){
        return cargo;
    }
    public TreeNode Left(){
        return left;
    }
    public TreeNode Right(){
        return right;
    }
    public String getKey(){
        return key;
    }
    public void setRight(TreeNode t){
        right = t;
    }
    public void setLeft(TreeNode t){
        left = t;
    }
}
 

Class Stations

package tree;
 
public class Station {
    private String name;
    private int line;
    private int zone;
    public static final double FARE_PER_ZONE = 1.00;
 
    public Station(String n, int l, int z){
        name = n;
        line = l;
        zone = z;
    }
    public String getName(){
        return name;
    }
 
    public int getLine(){
        return line;
    }
    public int getZone(){
        return zone;
    }
    public TreeNode toTreeNode(){
        return new TreeNode(name, new int[]{line, zone}, null, null);
    }
    public void addToOrderedTree(OrderedTree t){
        t.add(name, new int[]{line, zone});
    }
 
    //static methods
 
    /* These three overloaded methods can extract
     * attributes of a station from a TreeNode,
     * given that the tree node is initialized
     * by toTreeNode() method or equivalent approaches.
     * */
    public static String getName(TreeNode t){
        return t.getKey();
    }
    public static int getZone(TreeNode t){
        return ((int[])(t.getCargo()))[1];
    }
    public static int getLine(TreeNode t){
        return ((int[])(t.getCargo()))[0];
    }
 
    // calculate the fares from station a to station b
    // using either two stations or two tree nodes
    public static double calculateFare(Station a, Station b){
        if (a.getLine()==b.getLine())
            return (Math.abs(a.zone - b.zone)+1) * FARE_PER_ZONE;
        else
            return (a.zone + b.zone + 1) * FARE_PER_ZONE;
    }
    public static double calculateFare(TreeNode a, TreeNode b){
        if (Station.getLine(a)==Station.getLine(b))
            return (Math.abs(Station.getZone(a) - Station.getZone(b))+1) * FARE_PER_ZONE;
        else
            return (Station.getZone(a) + Station.getZone(b) + 1) * FARE_PER_ZONE;
    }
 
}
 

Class Fares

package tree;
 
public class Fares {
    private double fares[][];
    public String stations[];
    public Fares(String[] stations, OrderedTree t){
        recalculateFares(stations, t);
    }
    public void recalculateFares(String[] stations, OrderedTree t){
        fares = new double[stations.length][stations.length];
        this.stations = stations;
        for (int i = 0; i<stations.length - 1; i++){
            for (int j = 0; j<i; j++){
                fares[i][j] = Station.calculateFare(t.Search(stations[i]), t.Search(stations[j]));
                fares[j][i] = fares[i][j];
            }
            fares[i][i] = 0.00;
        }
    }
 
    //format output
    //more like a table
    public String toString(){
        String str =  "              |";
        for (String n : stations)
            str += String.format("%-14s|", n);
        str += "\n";
        for (int i = 0; i<=stations.length;i++)
            str += "--------------|";
        str += "\n";
        for (int i = 0; i<fares.length;i++){
            str += String.format("%-14s|", stations[i]);
            for (int j = 0; j<fares[i].length;j++)
                str += String.format("%-14.2f|", fares[i][j]);
            str += "\n";
            for (int x = 0; x<=stations.length;x++)
                str += "--------------|";
            str += "\n";
        }
        return str;
    }
}

    }
 
    //construct a ordered tree using the stations array initialized above
    public static OrderedTree addStationsToOrderedTree(Station[] sts){
        OrderedTree ot = new OrderedTree(sts[0].toTreeNode());
        for (int i = 1; i<sts.length; i++){
            sts[i].addToOrderedTree(ot);
        }
        return ot;
    }
    public static void main(String[] args) {
        Station[] sts = initializeStations();
        String[] stations = new String[sts.length];
        for (int i = 0; i<sts.length;i++)
            stations[i] = sts[i].getName();
        OrderedTree ot = addStationsToOrderedTree(sts);
        Fares f = new Fares(stations, ot);
        System.out.println(f.toString());
    }
