import random
def count_the_num(list,a):
    count=0
    for i in list:
        if i==a:
            count+=1
    return count

    for i in range (time):

    return people

    bias=0
    unbias=0
    for i in range(1,6):

public class Puzzle {
    public static int[][] Puzzle(int n){
        int[][] MagicSquare = new int[n][n];
        int x = 0;
        int y = n/2;
        MagicSquare[x][y] = 1;
        for(int i = 2;i<=n*n;i++){
            int a = x;
            int b = y;
            if (x == 0){
                x = n-1;
            }
            else{
                x --;
            }
            if (y == 0){
                y = n-1;
            }
            else{
                y --;
            }
            if (MagicSquare[x][y]!=0){
                x = (a+1)/n;
                y = b;
            }
            MagicSquare[x][y]=i;
        }
        return MagicSquare;
    }
}
