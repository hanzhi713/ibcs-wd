import turtle
import random
import math
wn = turtle.Screen()
matt = turtle.Turtle()
god = turtle.Turtle()
matt.hideturtle()
for i in range(0,11):
    if i % 2 == 0:
        a = -300+25*i
        matt.penup()
        matt.goto(a, 0)
        matt.right(180)
        matt.forward(10)
        matt.write(11-i)
        matt.left(180)
        matt.forward(10)
        matt.right(90)
        matt.pendown()
        matt.begin_fill()
        matt.fillcolor("red")
        matt.circle(-a)
        matt.end_fill()
        matt.left(90)
    else:
        a = -300+25*i
        matt.penup()
        matt.goto(a, 0)
        matt.right(180)
        matt.forward(10)
        matt.write(11-i)
        matt.left(180)
        matt.forward(10)
        matt.right(90)
        matt.pendown()
        matt.begin_fill()
        matt.fillcolor("white")
        matt.circle(-a)
        matt.end_fill()
        matt.left(90)
for z in range(0,6):
    x=random.uniform(-300,300)
    y=random.uniform(-300,300)
    god.penup()
    god.goto(x+z,y)
    god.pendown()
    god.stamp()
wn.mainloop()
