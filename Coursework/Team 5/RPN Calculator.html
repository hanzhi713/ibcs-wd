import math
def postfix_evaluation(s):
    s=s.split()
    n=len(s)
    stack=[]
    for i in range (n):
        if s[i].isdigit():
            stack.append(int(s[i]))
        elif s[i]=="+":
            a=stack.pop()
            b=stack.pop()
            stack.append(int(a)+int(b))
        elif s[i]=="*":
            a=stack.pop()
            b=stack.pop()
            stack.append(int(a)*int(b))
        elif s[i]=="/":
            a=stack.pop()
            b=stack.pop()
            stack.append(int(b)/int(a))
        elif s[i]=="-":
            a=stack.pop()
            b=stack.pop()
            stack.append(int(b)-int(a))
        elif s[i]=="**":
            a=stack.pop()
            b=stack.pop()
            stack.append(int(b)**int(a))
        elif s[i]=="sin":
            a=stack.pop()
            stack.append(math.sin(a))
        elif s[i]=="cos":
            a=stack.pop()
            stack.append(math.cos(a))
        elif s[i]=="tan":
            a=stack.pop()
            stack.append(math.tan(a))
    return stack.pop()
