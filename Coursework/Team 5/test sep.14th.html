def is_sorted(a):
    i=0
    result=[]
    while i+1<=len(a):
        if a[i]<a[i+1]:
            result.append(a[i])
            i+=1
        elif a[i]<=a[i+1]:
            result.append(a[i+1])
            i+=1
    is_sorted(result)
    return result
print(is_sorted([1,4,3,2,8,9]))

def is_anagram(a,b):
    if len(a)!=len(b):
        return False
    first=[]
    for alpha in a:
        first.append(alpha)
    second=[]
    for alpha in b:
        second.append(alpha)
    for check in first:
        if second.count(check)!=first.count(check):
            return False
    return True
print(is_anagram("apple","pplea"))

def has_duplicates(a):
    for item in a:
        if a.count(item)>1:
            return True
    return False
print(has_duplicates("aple"))

def birthday_paradox(a):
    i=0
    b=[]
    while i<=a:
        num=(365-i)/365
        b.append(num)
        i+=1
    huhu=1
    for i in b:
        huhu*=i
    return 1-huhu
print(birthday_paradox(22))

def remove_duplicates(a):
    new=[]
    for i in a:
        if a.count(i)>1:
            continue
        else:
            new.append(i)
    return new
print(remove_duplicates([1,1,1,2,3,4]))
