import cv2
import numpy as np
from matplotlib import pyplot as plt
 
def noise(image):
    for i in range(10000):
        a = int(np.random.random() * image.shape[1])
        b = int(np.random.random() * image.shape[0])
        image[a,b] = [255,255,255]
    return image
img = cv2.imread('C:/Users/Matt An/Desktop/testimg.jpg')
temp=noise(img)
b,g,r=cv2.split(temp)
result=cv2.merge([r,g,b])
plt.imshow(result)
plt.show()
