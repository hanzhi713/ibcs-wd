package com.company;
 
/**
 * Created by Matt An on 2016/11/27.
 */
public class Author { // The private instance variables
    private String name;
    private String email;
    private char gender;   // 'm' or 'f'
 
    // The constructor
    public Author(String name, String email, char gender) {
        this.name = name;
        this.email = email;
        this.gender = gender;
    }
 
    // The public getters and setters for the private instance variables.
    // No setter for name and gender as they are not designed to be changed.
    public String getName() {
        return name;
    }
    public char getGender() {
        return gender;
    }
    public String getEmail() {
        return email;
    }
    public void setEmail(String email) {
        this.email = email;
    }
 
    // The toString() describes itself
    public String toString() {
        return name + " (" + gender + ") at " + email;
    }
}

package com.company;
 
/**
 * Created by Matt An on 2016/11/27.
 */
public class Book {// The private instance variables
    private String name;
    private Author author;
    private double price;
    private int qty;
 
    // Constructor
    public Book(String name, Author author, double price, int qty) {
        this.name = name;
        this.author = author;
        this.price = price;
        this.qty = qty;
    }
 
    // Getters and Setters
    public String getName() {
        return name;
    }
    public Author getAuthor() {
        return author;  // return member author, which is an instance of the class Author
    }
    public double getPrice() {
        return price;
    }
    public void setPrice(double price) {
        this.price = price;
    }
    public int getQty() {
        return qty;
    }
    public void setQty(int qty) {
        this.qty = qty;
    }
 
    // The toString() describes itself
    public String toString() {
        return "'" + name + "' by " + author;  // author.toString()
    }
}

package com.company;

public class MyPoint {
    private int x;
    private int y;
    public MyPoint(){
        x=0;
        y=0;
    }
    public MyPoint(int x, int y){
        x=x;
        y=y;
    }
    public int getX(){
        return x;
    }
    public int getY(){
        return y;
    }
    public void setX(int input){
        this.x= input;
    }
    public void setY(int input){
        this.y=input;
    }
    public int[] getXY(){
        int[] a=new int[]{x,y};
        return a;
    }
    public void setXY(int a,int b){
        this.x=a;
        this.y=b;
    }
    public String toString(){
        return "the point is"+x+y;
    }
    public double distance(int x,int y){
        double xdiff=Math.pow(x-this.x,2);
        double ydiff=Math.pow(y-this.y,2);
        double result=Math.pow(xdiff+ydiff,0.5);
        return result;
    }
    public double distance(MyPoint Matt){
