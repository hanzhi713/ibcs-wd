class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    def setX(self, x):
        self.x = x
    def setY(self,y):
        self.y = y
    def getX(self):
        return self.x
    def getY(self):
        return self.y
    def midPoint(self, target):
        t = Point(0.5 * (self.x + target.x), 0.5 * (self.x + target.x))
        return t
    def reflect_x(self):
        copy = Point(self.x, self.y)
        return copy
    def slope(self):
        i = round(self.y / self.x, 2)
        return i
    def get_line_to(self, point):
        slope = (self.y - point.y) / (self.x - point.x)
        y_inter = self.y - slope * self.x
        return (slope, y_inter)
    def reflect_y(self):
        return Point(-self.x,self.y)
    def reflect_point(self):
        return Point(-self.x,-self.y)
    def __str__(self):
        return "({0},{1})".format(self.x, self.y)
