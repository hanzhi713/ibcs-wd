    print("    conversion")
    print("ounces      grams")
    for i in range(0,ounce+1):
        gram=28.35*i
        gram=round(gram)
        b=str(i)
        if len(b)==1:
            print("  ",i,"        ",gram)

def is_sort(list):
    for x in range(0,len(list)):
        thisvalue=list[x]
        y=x-1
        while list[y]>thisvalue and y>0:
            list[y+1]=list[y]
            y=y-1
        list[y+1]=thisvalue
    return list
print(is_sort([1,5,3,6,2,23,4]))

def choose_small(a,b):
    if a>b:
        return b
    else:
        return a
def word_sort(word1,word2):
    limit=choose_small(len(word1),len(word2))
    for index in range(0,limit):
        if word1[index]>word2[index]:
            return (word2,word1)
        elif word1[index]==word2[index]:
            continue
        else:
            return (word1,word2)
print(word_sort("apple","append"))

def book_search(searchbook):
    book=open("booktitle.txt")
    booklist=book.readlines()
    for i in booklist:
        if searchbook==i:
            return True
    return False
    book.close()
print(book_search("Tortoise Care\n"))

def check_perfect(n):
    sum=0
    for count in range(1,(n//2)+1):
        if n%2==0:
            sum+=count
    if sum==n:
        print("perfect")
    else:
        print("not perfect")
check_perfect(6)
