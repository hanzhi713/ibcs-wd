1. random pick from 1 to 50
def randompicknumber(n,a,b):
    import random
    l=[]
    for i in range(n):
        x=random.randint(a,b)
        if x in l:
            continue
        else:
            l.append(x)
    print(l)
randompicknumber(6,1,50)

def encrypt(a):
    import string
    alphabet="ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz "
    substitute="PLFNOCQUDZVGIXMWJBKEAHSYRTplfnocqudzvgixmwjbkeahsyrt "
    l=[]
    for i in a:
        if i not in string.punctuation:
            l+=substitute[alphabet.index(i)]
        else:
            l+=i
    print("".join(l))
encrypt("hello world!")
