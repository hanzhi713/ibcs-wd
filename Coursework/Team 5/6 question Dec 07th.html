1.countHi

public int countHi(String str) {
int result = 0;
for(int i = 0; i < str.length() - 1; i++)
{
if(str.substring(i, i+2).equals("hi"))
{
result++;
}
}
return result;
}

public String repeatSeparator(String word, String seperator, int count) {
  String result = "";
for(int i = 0; i < count; i++) {
if(count==1) {
result += word;
return result;
} else {
result += (word + seperator);
}
}
String finalresult=result.substring(0,result.length()-seperator.length());
return finalresult;
}
 

3.matchup

public int matchUp(String[] a, String[] b) {
int count = 0;
for(int i = 0; i < a.length; i++){
  if( (a[i]!="")
        && (b[i]!="")
        && (a[i].charAt(0) == b[i].charAt(0)) ){
    count++;
    }
  }
  return count;
}

public boolean linearIn(int[] outer, int[] inner) {
int num = 0;
if(inner.length == 0) {
  return true;
}
int a = 0;
for(int b = 0; b < outer.length; b++) {
    if(outer[b] == inner[a]) {
      num++;
      a++;
    }
    else if(outer[b] > inner[a]) {
      return false;
    }
 
    if(num == inner.length)
      return true;
}
return false;
}
