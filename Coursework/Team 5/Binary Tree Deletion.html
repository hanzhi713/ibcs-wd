public class TreeNode {
    private TreeNode lefTreeNode;
    // 右节点
    private TreeNode rightTreeNode;
    // 数据
    private int value;
 
    public TreeNode getLefTreeNode() {
        return lefTreeNode;
    }
 
    public void setLefTreeNode(TreeNode lefTreeNode) {
        this.lefTreeNode = lefTreeNode;
    }
 
    public TreeNode getRightTreeNode() {
        return rightTreeNode;
    }
 
    public void setRightTreeNode(TreeNode rightTreeNode) {
        this.rightTreeNode = rightTreeNode;
    }
 
    public int getValue() {
        return value;
    }
 
    public void setValue(int value) {
        this.value = value;
    }
 
    public TreeNode(int value) {
        this(null, null, value);
    }
 
    public TreeNode(TreeNode lefTreeNode, TreeNode rightNode, int value){
        this.lefTreeNode = lefTreeNode;
        this.rightTreeNode = rightNode;
        this.value = value;
    }
}

package tree;
 
/**

public class Binarytree {
    private TreeNode root;
    public TreeNode getRoot(){
        return root;
    }
    public void insert(int value) {
        TreeNode newNode = new TreeNode(value);
        //set root
        if (root == null) {
            root = newNode;
            root.setLefTreeNode(null);
            root.setRightTreeNode(null);
        } else {
            TreeNode currentNode = root;
            TreeNode parentNode;
            while (true) {
                parentNode = currentNode;
                // set right
                if (newNode.getValue() > currentNode.getValue()) {
                    currentNode = currentNode.getRightTreeNode();
                    if (currentNode == null) {
                        parentNode.setRightTreeNode(newNode);
                        return;
                    }
                } else {
                    // set left
                    currentNode = currentNode.getLefTreeNode();
                    if (currentNode == null) {
                        parentNode.setLefTreeNode(newNode);
                        return;
                    }
                }
            }
        }
    }
    public TreeNode find(int key) {
 
        TreeNode currentNode = root;
 
        if (currentNode != null) {
 
            while (currentNode.getValue() != key) {
 
                if (currentNode.getValue() > key) {
                    currentNode = currentNode.getLefTreeNode();
                } else {
                    currentNode = currentNode.getRightTreeNode();
                }
 
                if (currentNode == null) {
                    return null;
                }
                if(currentNode.getValue()==key){
                    break;
                }
            }
 
        }
    return currentNode;
    }
    public TreeNode parent(TreeNode find){
        TreeNode temp=parent(root,find);
        return temp;
    }
    public TreeNode parent(TreeNode node, TreeNode findnode){
        if(node==null){
            return null;
        }
        if((node.getLefTreeNode()!=null&&node.getLefTreeNode().getValue()==findnode.getValue())||(node.getRightTreeNode()!=null&&node.getRightTreeNode().getValue()==findnode.getValue())){
            return node;
        }
        TreeNode N;
        if((N=parent(node.getLefTreeNode(),findnode))!=null){
            return N;
        }else{
            return parent(node.getRightTreeNode(),findnode);
        }
    }
 
    public TreeNode Minright(TreeNode node){
        if(node.getLefTreeNode()==null){
            return node;
        }
        if(node.getLefTreeNode().getLefTreeNode()==null){
            if(node.getLefTreeNode().getRightTreeNode()!=null){
                TreeNode temp=node.getLefTreeNode();
                node.setLefTreeNode(node.getLefTreeNode().getRightTreeNode());
                temp.setRightTreeNode(null);
                return temp;
            }
            TreeNode temp=node.getLefTreeNode();
            node.setLefTreeNode(null);
            return temp;
        }
        return Minright(node.getLefTreeNode());
    }
 
    public TreeNode Maxleft(TreeNode node){
        if(node.getRightTreeNode()==null){
            return node;
        }
        if(node.getRightTreeNode().getRightTreeNode()==null){
            if(node.getRightTreeNode().getLefTreeNode()!=null){
                TreeNode temp=node.getRightTreeNode();
                node.setRightTreeNode(node.getRightTreeNode().getLefTreeNode());
                temp.setLefTreeNode(null);
                return temp;
            }
            TreeNode temp=node.getRightTreeNode();
            node.setRightTreeNode(null);
            return temp;
        }
        return Maxleft(node.getRightTreeNode());
    }
 
    public void delete(TreeNode temp){
        TreeNode parent=parent(temp);
        if(temp==root){
            TreeNode node1;
            node1=root.getLefTreeNode();
            root.setLefTreeNode(root.getLefTreeNode().getLefTreeNode());
            root.setRightTreeNode(root.getRightTreeNode());
            root=node1;
        }
        else{
            if (temp.getLefTreeNode()==null&&temp.getRightTreeNode()==null){
                if(temp.getValue()>parent.getValue()){
                    temp.setRightTreeNode(null);
                }
                else if(temp.getValue()<parent.getValue()){
                    temp.setLefTreeNode(null);
                }
            }
            else if(temp.getLefTreeNode()!=null&&temp.getRightTreeNode()!=null){
                TreeNode Node=Maxleft(temp.getLefTreeNode());
                if(temp.getValue()>parent.getValue()) {
                    if (temp.getLefTreeNode().getValue() == Node.getValue()) {
                        parent.setRightTreeNode(temp.getLefTreeNode());
                        temp.getLefTreeNode().setRightTreeNode(temp.getRightTreeNode());
                    } else {
                        parent.setRightTreeNode(Node);
                        Node.setLefTreeNode(temp.getLefTreeNode());
                        Node.setRightTreeNode(temp.getRightTreeNode());
                    }
                }
                    if(temp.getValue()<parent.getValue()){
                        if(temp.getLefTreeNode().getValue()==parent.getValue()){
                            parent.setLefTreeNode(temp.getLefTreeNode());
                            temp.getLefTreeNode().setRightTreeNode(temp.getRightTreeNode());
                        }
                        else{
                            parent.setLefTreeNode(Node);
                            Node.setLefTreeNode(temp.getLefTreeNode());
                            Node.setRightTreeNode(temp.getRightTreeNode());
                        }
                    }
                }
                else if(temp.getLefTreeNode()!=null&&temp.getRightTreeNode()==null){
                if(temp.getValue()>parent.getValue()){
                    parent.setRightTreeNode(temp.getLefTreeNode());
                }
                else if(temp.getValue()<parent.getValue()){
                    parent.setLefTreeNode(temp.getLefTreeNode());
                }
            }
            else if(temp.getLefTreeNode()==null&&temp.getRightTreeNode()!=null){
                if(temp.getValue()>parent.getValue()){
                    parent.setRightTreeNode(temp.getRightTreeNode());
                }
                else if(temp.getValue()<parent.getValue()){
                    parent.setLefTreeNode(temp.getRightTreeNode());
                }
            }
        }
    }
}
