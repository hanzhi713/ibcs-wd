f = open("textfile_alice_on_wonderland.txt", "r")
xs = f.readlines()
 
f.close()
 
 
n = open("textfile_alice_on_wonderland_YOURNAME.txt", "w")
for e in xs:
    t=e.split()
    for b in t:
 
        if b=="Alice" or b=="alice":
            n.write("Charles ")
        else:
            n.write(b)
            n.write(" ")
    n.write("\n")
n.close()
