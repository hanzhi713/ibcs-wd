def overlapping(a,b):
    Flag = False
    for o in b:
        for i in a :
            if o == i:
                Flag = True
    return(Flag)

def draw_pie(t,length,degree):
    t.begin_fill()
    t.forward(length)
    t.left(90)
    t.circle(length,degree)
    t.left(90)
    t.forward(length)
    t.right(180)
    t.end_fill()
