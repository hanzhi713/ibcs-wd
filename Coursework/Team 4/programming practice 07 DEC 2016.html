public int countHi(String str) {
  int count = 0;
  for(int i = 0;i<str.length()-1;i++){
    if(str.substring(i,i+1).compareTo("h")==0){
      if(str.substring(i+1,i+2).compareTo("i")==0){
        count = count +1;
      }
 
    }
 
  }
  return count;
}

public String repeatSeparator(String word, String sep, int count) {
  String str = "";
  if(count == 0){
    return str;
  }
  else{
    for(int i = 0;i<count-1;i++ ){
      str = str+word;
      str = str + sep;
    }
  }
  str = str + word;
  return str;
}

  boolean c = true;
  for (int i=0;i<inner.length;i++ ){
    int a = 0;

      if( outer[o]==inner[i] ){
        a = 1;
      }
    }
    if(a == 0){

    }
  }

}

public int[] seriesUp(int n) {
  int num = 0;
  int[] a= new int[n*(n + 1)/2];
  for(int i = 1; i<n+1;i++){
    for(int o =1;o<i+1;o++){
      a[num] = o;
      num = num +1;
    }
  }
  return a;
}

public int matchUp(String[] a, String[] b) {
  int num = 0;
  for(int i=0; i<a.length;i++){
    if(a[i]!="" && b[i]!=""){
      if (a[i].substring(0,1).compareTo(b[i].substring(0,1))==0){
        num = num+1;
      }
 
    }
  }
  return num;
}
