    num = 0
    for i in range(len(v)):
        su = u[i] + v[i]
        num = num + su
    return num
u = [1,2]

def dot_product(u,v):
    new = []
    a = u[1]*v[2] - u[2]*v[1]
    b = u[2]*v[0] - u[0]*v[2]
    c = u[0]*v[1] - u[1]*v[0]
    new.append(a)
    new.append(b)
    new.append(c)
    return new
u = [1,2,3]
v = [2,3,4]
res = dot_product(u,v)
print(res)

def replace(s,old,new):
 
    wds = s.split(old)
    wdss = new.join(wds)
    return wdss

old = "old"
new = "ald"
