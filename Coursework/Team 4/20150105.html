public class Station {
    String Line;
    int Zone;
    String Name;
    public Station(String line, int zone, String name){
        Line = line;
        Zone = zone;
        Name = name;
    }
    public String getLine(){
        return Line;
    }
    public int getZone(){
        return Zone;
    }
    public String getName(){
        return Name;
    }
}
 


public class fares {

                if(al==bl ||station1.getName()=="Centro" ||station2.getName()=="Centro"){
