import pygame

def get_position():
    a = int(random.uniform(0,770))
    b = int(random.uniform(0,770))
    pos=[a,b]
    return pos
 
 
def in_rec(a,b):
 
    mx, my = pygame.mouse.get_pos()
 
    if mx>a:
        if mx<(a+30):
            if my > b:
                if my < (b + 30):
                    return True
    else:
        return False
