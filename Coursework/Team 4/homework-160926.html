一、 Point类添加两个method: 1) 求当前点p与另一个点q的连线斜率, 2) 该线与x轴正方向夹角。二、构建一个三维坐标点类 Point3D, 能够求解该点与原点的三维距离，与另一点q的三维距离，与另一点q的中间某点坐标，比例系数为一个参数k，当k=0.5时，就是求两点的中点。

import math
class Point:
    """ Create a new Point, at coordinates x, y """
 
    def __init__(self, x=0, y=0):
        """ Create a new point at x, y """
        self.x = x
        self.y = y
 
    def setX(self, num):
        self.x = num
 
    def setY(self, num):
        self.y = num
 
    def gradient(self, target):
        m = (target.y - self.y) / (target.x - self.x)
        return m
    def arch(self,target):
 
        m = (target.y - self.y) / (target.x - self.x)
        a = math.atan(m)
        a = a*180.0/math.pi
        return a
 
 
 
a = Point(2,2)
b = Point(3,3)
print(a.arch(b))

import math
class Point3D:
    """ Create a new Point, at coordinates x, y """
 
    def __init__(self, x=0, y=0,z=0):
        """ Create a new point at x, y """
        self.x = x
        self.y = y
        self.z = z
    def distance_origin(self):
 
        s = math.sqrt(self.x**2+self.y**2+self.z**2)
        return s
    def distance_other(self,target):
        s = math.sqrt((target.x-self.x)**2+(target.y-self.y)**2+(target.z-self.z)**2)
        return s
    def between(self,target,a):
        mx = (self.x + target.x)/a
        my = (self.y + target.y)/a
        mz = (self.z + target.z)/a
        return Point3D(mx,my,mz)
 
 
 
a=Point3D(1,1,1)
b = Point3D(3,3,3)
 
print(a.between(b,2))
