def transfer(a):
    new = ""
    message_character = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    substitude_character = "PLFNOCQUDZVGIXMWJBKEAHSYRT"
    num = len(a)
    for i in range(num):
        sit = message_character.index(a[i])
        new = new + (substitude_character[sit])
    return(new)
print(transfer("ATSEVEN"))
