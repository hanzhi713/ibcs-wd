import cv2
import numpy as np
from matplotlib import pyplot as plt
import random
 
 
 
def main():
    img = cv2.imread('testimg.jpg')
    salt(img,100000)
    plt.imshow(cv2.cvtColor(img, cv2.COLOR_BGR2RGB))
    plt.show()
 
def salt(img,n):
    h, w, c = img.shape
    for i in range(n):
        x= int(random.uniform(0,h))
        y= int(random.uniform(0,w))
        img[x,y]=255
    return img
 
 
 
main()
