def guess_number():
    higher = 1000
    lower = 0
    guesses = 0
    num = int(input("please give the number beyween 1 and 1000:"))
 
    while True:
        g_num = int((higher+lower)/2)
        guesses += 1
        print("I guess it is {0}".format(g_num))
        if g_num> num :
            higher = g_num
            lower = lower
        elif g_num < num :
            higher =higher
            lower = g_num
        else:
            input("\n\nGreat, I got it in {0} guesses!\n\n".format(guesses))
            break
 
guess_number()
