import turtle
 
wn=turtle.Screen()
l=turtle.Turtle()
l.speed(999999)
l.hideturtle()
l.penup()
l.goto(-500,0)
l.pendown()
 
length = int(input("length"))
times = int(input("times"))
 
def draw_needle(length,times):
 
    if times == 1:
        l.forward(length)
        l.left(60)
        l.forward(length)
        l.right(120)
        l.forward(length)
        l.left(60)
        l.forward(length)
    else:
        draw_needle(length/2,(times - 1))
        l.left(60)
        draw_needle(length/2,(times - 1))
        l.right(120)
        draw_needle(length/2,(times - 1))
        l.left(60)
        draw_needle(length/2,(times - 1))
 
draw_needle(length,times)
