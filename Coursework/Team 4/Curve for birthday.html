import matplotlib.pyplot as plt
import random
 
def has_duplicates(a):
    b = False
    new = []
    for i in a:
        if i not in new:
            new.append(i)
        else:
            b = True
    return b
def generator(n):
    birthday = []
    for i in range(n):
        birthday.append(random.randrange(1,366))
    return birthday
 
def check_birthday(b):
    count = 0
    for i in range(10000):
        if has_duplicates(generator(b)) == True:
            count +=1
    return(count/10000)
 
l1 = []
l2 = []
b = int(input("number"))
 
for i in range(1,b):
    l1.append(check_birthday(i))
    l2.append(i)
 
plt.plot(l2,l1)
plt.show()
