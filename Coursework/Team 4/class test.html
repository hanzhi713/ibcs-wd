public String[] mergeTwo(String[] a, String[] b, int n) {
  int x = 0;
  int y = 0;
  int z = 0;
 
  String[] vec = new String[n];
  while(z<n){
    if(a[x].compareTo(b[y])<0){
      vec[z] = a[x];
      x = x+1;
      z = z+1;
    }
    else if(a[x].compareTo(b[y])>0){
      vec[z]=b[y];
      y = y+1;
      z = z+1;
    }
    else{
      vec[z] = a[x];
      x = x+1;
      y = y+1;
      z = z+1;
    }
  }
  return vec;
}
 


public int fibonacci(int n) {
  int a;
  int b;
  int c;
  ArrayList<Integer> list = new ArrayList<Integer>();
  list.add(0);
  list.add(1);
  for(a = 2;a<=n;a++){
    b= list.get(a-1)+list.get(a-2);
    list.add(b);
  }
  c = list.get(n);
  return c;
 
 
 
 
}
