import java.util.ArrayList;
import java.util.List;
 
public class list {
    public static void main(String[] args) {
        List<String> list = new ArrayList<String>();
 
        list.add("a"); // list is 0 7 1 5
        list.add("b");
        list.add("c"); // list is 0 7 1 5
        list.add("d");
        // insert(list,5);
        // swap(list,2,3);
        changeEvenToEmpty(list);
    }
 
     public static void insert(List<Integer> list, Integer value)
     {
     Integer index = 0;
     //find insertion point
     while (index < list.size() &&
     value.compareTo(list.get(index)) > 0)
     index++;
     //insert value
     list.add(index, value);
 
     System.out.println(list);
     }
     public static void swap(List<Integer> list, int i, int j)
     {
     Integer temp = list.get(i);
     list.set(i, list.get(j));
     list.set(j, temp);
     System.out.println(list);
     }
    public static void changeEvenToEmpty(List<String> strList) {
        boolean even = true;
        int index = 0;
        while (index < strList.size()) {
            if (even)
                strList.set(index, "");
            index++;
            even = !even;
        }
        System.out.println(strList);
    }
}
