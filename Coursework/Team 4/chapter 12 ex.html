import string
 
def cleanword(s):
    s_without_punct = ""
    for letter in s:
        if letter not in string.punctuation:
            s_without_punct += letter
    return s_without_punct
 
def has_dashdash(s):
    a = s.find("--")
    if a == -1:
        return False
    else:
        return True
 
def extract_words(s):
    b = s.split("--")
    e = []
    c = []
    for i in b:
        c = c + i.split()
    for i in c:
        i = cleanword(i)
        i = i.lower()
        e.append(i)
    return e
 
 
def wordcount(w,s):
    c = 0
    for i in s:
        if i== w:
            c += 1
    return c
 
def wordset(w):
    w = list(set(w))
    w.sort()
    return w
 
def longestword(w):
    a = 0
    for i in w:
        if len(i)> a:
            a = len(i)
    return a
