        public void newClient(Client client, LinkedList<Client> Bookings){
            boolean found=false;
            int index=0;
            if(Bookings.isEmpty()){
                Bookings.addFirst(client);
            }
            else{
                Client nextClient=(Client) Bookings.getFirst();
                while((nextClient!=null)&&(!found)){
                    if(Dates.equalDate(Dates.compareDate(nextClient.getArrive(),client.getArrive()),client.getArrive())||
                            (Dates.equalDate(nextClient.getArrive(),client.getArrive()))){
                        Bookings.add(index,client);
                        found=true;
                    }else{
                        index = index + 1;
                        nextClient=(Client) Bookings.get(index);
                    }
                    if(!found){
                        Bookings.addLast(client);
                    }
                }
            }
        }
    }
    public static class Dates{
        private int day;
        private int month;
        private int year;
        public Dates(int day,int month,int year){
            this.day=day;
            this.month=month;
            this.year=year;
        }
        public int getDay(){return day;}
        public int getMonth(){return month;}
        public int getYear(){return year;}
        public static boolean equalDate(Dates x, Dates y){
            int dayA=x.getDay();
            int monthA=x.getMonth();
            int yearA=x.getYear();
            int dayB=y.getDay();
            int monthB=y.getMonth();
            int yearB=y.getYear();
            if(dayA==dayB){
                if(monthA==monthB){
                    if(yearA==yearB){
                        return true;
                    }else{return false;}
                }else{return false;}
            }else{return false;}
        }
        public static Dates compareDate(Dates x, Dates y){
            int dayA=x.getDay();
            int monthA=x.getMonth();
            int yearA=x.getYear();
            int dayB=y.getDay();
            int monthB=y.getMonth();
            int yearB=y.getYear();
            int dayD=dayB-dayA;
            int monthD=monthB-monthA;
            int yearD=yearB-yearA;
            if(yearD>0){
                return x;
            }
            else if(yearD==0){
                if(monthD>0){
                    return x;
                }
                else if(monthD==0){
                    if(dayD>0){
                        return x;
                    }else if(dayD==0){
                        return x;
                    }else{return y;}
                }else{return y;}
            }else{return y;}
 
        }
 
    }

    public  int StayDays(Dates x,Dates y){
        int dayA=x.getDay();
        int monthA=x.getMonth();
        int yearA=x.getYear();
        int dayB=y.getDay();
        int monthB=y.getMonth();
        int yearB=y.getYear();
        int dayD=dayB-dayA;
        int monthD=monthB-monthA;
        if (monthD<0){
            monthD=monthD+12;
        }
        int yearD=yearB-yearA;
        int totalDay=yearD*365+monthD*30+dayD;
        return totalDay;
    }

                    Client temp=arrivalsToday[k];
                    arrivalsToday[k]=arrivalsToday[k+1];
                    arrivalsToday[k+1]=temp;
                }
            }
        }
        }
