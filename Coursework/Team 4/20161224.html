public class insertionsort {
    private Integer[] a;
 
    public void insertionSort() {
        for (int i = 1; i < a.length; i++) {
            Integer temp = a[i];
            int j = i - 1;
            while (j >= 0 && temp.compareTo(a[j]) > 0) {
                a[j + 1] = a[j];
                j--;
            }
            a[j + 1] = temp;
        }
    }
 
}
 

public class SelectionSort {
    private int[] a;
 
    public SelectionSort(int[] arr) {
        a = arr;
    }
 
    private void swap(int i, int j) {
        int temp = a[i];
        a[i] = a[j];
        a[j] = temp;
    }
 
    public void selectionSort() {
        int maxPos, max;
        for (int i = 0; i < a.length - 1; i++) {
            // find max element in a[i+1] to a[a.length-1]
            max = a[i];
            maxPos = i;
            for (int j = i + 1; j < a.length; j++)
                if (max < a[j]) {
                    max = a[j];
                    maxPos = j;
                }
            swap(i, maxPos); // swap a[i] and a[maxPos]
        }
    }
}

public class MergeSort {
 
    public static void main(String[] args) {
        int[] array = { 9, 8, 7, 6, 5, 4, 3, 2, 1, 0, -1, -2, -3 };
        sortArray(array,0,array.length-1);
        for(int i = 0;i<array.length;i++){
            System.out.print(array[i]);
        }
    }
 
    private static void sortArray(int[] array, int start, int end) {
        if (start == end) {
            return;
        }
 
        int sortSize = end - start + 1;
        int mid;
        if (sortSize % 2 == 0) {
            mid = start + sortSize / 2 - 1;
        }
        else {
            mid = start + sortSize / 2;
        }
 
        sortArray(array, start, mid);
        sortArray(array, mid + 1, end);
 
        mergeArray(array, start, mid, end);
    }
 
    private static void mergeArray(int[] array, int start, int mid, int end) {
        int totalSize = end - start + 1;
        int size1 = mid - start + 1;
        int size2 = end - mid;
 
        int[] array1 = new int[size1];
        int[] array2 = new int[size2];
 
        for (int i = 0; i < size1; i++) {
            array1[i] = array[start + i];
        }
 
        for (int i = 0; i < size2; i++) {
            array2[i] = array[mid + 1 + i];
        }
 
        int mergeCount = 0;
        int index1 = 0;
        int index2 = 0;
 
        while (mergeCount < totalSize) {
            if (index1 == size1) {
                for (int i = index2; i < size2; i++) {
                    array[start + mergeCount] = array2[i];
                    mergeCount++;
                    index2++;
                }
            } else if (index2 == size2) {
                for (int i = index1; i < size1; i++) {
                    array[start + mergeCount] = array1[i];
                    mergeCount++;
                    index1++;
                }
            } else {
                int value1 = array1[index1];
                int value2 = array2[index2];
 
                if (value1 == value2) {
                    array[start + mergeCount] = value1;
                    array[start + mergeCount + 1] = value1;
                    mergeCount += 2;
                    index1++;
                    index2++;
                } else if (value1 < value2) {
                    array[start + mergeCount] = value1;
                    mergeCount++;
                    index1++;
                } else if (value1 > value2) {
                    array[start + mergeCount] = value2;
                    mergeCount++;
                    index2++;
                }
            }
        }
    }
}


public class quickSort {
    public static void main(String[] args) {
        int[] array = new int[]{1,4,7,3,2,5,7};
        sort(array,0,6);
        for(int i = 0; i< 7;i++){
            System.out.print(array[i]);
        }
    }
    public static int partition(int []array,int low,int high){
 
        int key=array[low];
        while(low<high){
            while(array[high]>=key&&high>low){
                high--;
            }
            array[low]=array[high];
            while(array[low]<=key&&high>low){
                low++;
            }
            array[high]=array[low];
        }
        array[high]=key;
        return high;
    }
 
    public static void sort(int[] array,int low ,int high){
        if(low>=high){
            return ;
        }
        int index=partition(array,low,high);
        sort(array,low,index-1);
        sort(array,index+1,high);
    }
