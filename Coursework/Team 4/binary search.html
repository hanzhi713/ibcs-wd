def search_binary_iter(xs, target):
 
    """ Find and return the index of key in sequence xs """
    higher = len(xs) - 1
    lower = 0
 
    while True:
        mid = (higher+lower)//2
        if higher == lower:
            return -1
        if target > xs[mid]:
            lower =mid +1
        elif target < xs[mid]:
            higher = mid
        else:
            return 1
