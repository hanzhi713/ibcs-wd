def binarysearch(t,f,l,h):
    if h<l:
        return -1
    else:
        middle = int((h+l)/2)
        if t[int(1/2*(len(t)))]> f:
            binarysearch(t,f,l,middle-1)
        else:
            if t[middle]<f:
                binarysearch(t,f,middle+1,h)
            else:
                return middle

def fragment(N):
    sum = 0
    for i in range(1,N//2 + 1):
        sum = sum + i
    if sum == N:
        print("perfect")
    else:
        print("not perfect")
