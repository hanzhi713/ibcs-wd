class Point:
    """ Point class represents and manipulates x,y coords. """
 
    def __init__(self, x=0, y=0):
        """ Create a new point at x, y """
        self.x = x
        self.y = y
    def long(p,q):
        p = Point(p[0],p[1])
        q = Point(q[0],q[1])
        dx = p.x - q.x
        dy = p.y - q.y
        dsquared = dx**2 + dy**2
        result = dsquared**0.5
        return result
