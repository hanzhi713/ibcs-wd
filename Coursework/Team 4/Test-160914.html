def is_sorted(n):
    a = len(n)
    b = True
    for i in range(a-1):
        if n[i]>=n[i+1]:
            b= False
    return b
print(is_sorted([1,2,3]))

    new = ""
    d = True
    c = list(a)
 
    for i in range(len(c)):
        new = new+ c[len(c)-i-1]
    if new != b:
        d = False

def is_anagrams(a,b):
    a = list(a)
    b = list(b)
    a1 = a[:]
    b1 = b[:]
    a1.sort()
    b1.sort()
    if a1 == b1:
        return True
    else:
        return False
