import turtle
 
wn = turtle.Screen()
wn.title("3.11")
 
wn = turtle.Screen()
wn.bgcolor("lightgreen")
wn.title("3.12")
 
alex = turtle.Turtle()
alex.speed(0)
alex.pencolor('Blue')
alex.pensize(5)
alex.shape('turtle')
alex.stamp()
 
for i in range(12):
    alex.penup()
    alex.fd(80)
    alex.pendown()
    alex.fd(10)
    alex.penup()
    alex.fd(10)
    alex.stamp()
    alex.setpos(0,0)
    alex.rt(30)
 
alex.hideturtle()
 
wn.mainloop()
