import turtle
 
length=243
vertex=[]
layer=int(input('How many layers?'))
 
scr=turtle.Screen()
scr.bgcolor('black')
 
alex=turtle.Turtle()
alex.speed(0)
alex.pensize(1)
alex.hideturtle()
alex.penup()
alex.pencolor('white')
alex.setpos(-380,-210)
alex.pendown()
 
for i in range(3):
    vertex+=[(alex.xcor(),alex.ycor(),alex.heading())]
    alex.fd(length*3)
    alex.left(120)
 
def move():
    global length,alex
    addvertex=[]
    addvertex+=[(alex.xcor(),alex.ycor(),alex.heading())]
    alex.fd(length)
    alex.pencolor('black')
    alex.fd(length)
    alex.back(length)
    alex.pencolor('white')
    alex.rt(60)
    addvertex+=[(alex.xcor(),alex.ycor(),alex.heading())]
    alex.fd(length)
    alex.left(120)
    addvertex+=[(alex.xcor(),alex.ycor(),alex.heading())]
    alex.fd(length)
    alex.rt(60)
    addvertex+=[(alex.xcor(),alex.ycor(),alex.heading())]
    alex.fd(length)
    return addvertex
 
def draw():
    global length,alex,vertex
    newvertex=[]
    for (x,y,z) in vertex:
        alex.setpos(x,y)
        alex.seth(z)
        addvertex=move()
        newvertex+=addvertex
    return newvertex
 
for k in range(layer+1):
    vertex=draw()
    length/=3
 
scr.mainloop()

import turtle
 
wn=turtle.Screen()
wn.bgcolor('black')
alex=turtle.Turtle()
alex.speed(0)
alex.color('white')
alex.pensize(1)
alex.penup()
alex.setpos(-380,-210)
alex.pendown()
alex.hideturtle()
 
length=243
layer=int(input('How many layers? '))
 
def side(length,layer):
    if layer == 1:
        alex.fd(length)
        alex.rt(60)
        alex.fd(length)
        alex.left(120)
        alex.fd(length)
        alex.rt(60)
        alex.fd(length)
    else:
        side(length/3,layer-1)
        alex.rt(60)
        side(length/3,layer-1)
        alex.left(120)
        side(length/3,layer-1)
        alex.rt(60)
        side(length/3,layer-1)
 
for i in range(3):
    side(length,layer)
    alex.left(120)
 
wn.mainloop()
