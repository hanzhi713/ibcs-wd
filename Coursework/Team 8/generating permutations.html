import copy
def unitpermutations(n):
    if n==1:
        return [[0]]
    else:
        result=unitpermutations(n-1)
        newresult=[]
        k=len(result[0])
        for j in range(k+1):
            for i in result:
                add=copy.deepcopy(i)
                add[j:j]=[n-1]
                newresult.append(add)
        newresult.sort()
        return newresult
 
def remove_adjacent_dups(xs):
    result = []
    most_recent_elem = None
    for e in xs:
        if e != most_recent_elem:
            result.append(e)
            most_recent_elem = e
    return result
 
def permutations(lst):
    mod=unitpermutations(len(lst))
    for i in mod:
        for j in range(len(i)):
            i[j]=lst[i[j]]
    mod.sort()
    result=remove_adjacent_dups(mod)
    return result
