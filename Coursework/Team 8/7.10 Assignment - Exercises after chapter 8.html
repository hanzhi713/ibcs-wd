def count_letters(string,char,start=0,count=0):
    xi=string.find(char,start)
    if xi!=-1:
        count+=1
        start=xi+1
        count=count_letters(string,char,start,count)
    else:
        return count
    return count

    count=0
    xi=0

def remove(str1,str2):
    xi=str2.find(str1)
    if xi==-1:
        return str2
    else:
        new=str2[:xi]+str2[xi+len(str1):]
        return new
