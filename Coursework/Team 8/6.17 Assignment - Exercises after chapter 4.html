import turtle
 
def draw_square(t, sz):
    """Make turtle t draw a square of sz. And move forward"""
    for i in range(4):
        t.forward(sz)
        t.left(90)
    t.penup()
    t.fd(2*sz)
    t.pendown()
 
wn = turtle.Screen()
wn.bgcolor("lightgreen")
wn.title("4.1")
 
alex = turtle.Turtle()
alex.speed(0)
alex.pencolor('hotpink')
alex.pensize(5)
for i in range(5):
    draw_square(alex, 20)
wn.mainloop()

import turtle
 
def draw_square(t):
    """Make turtle t draw a square of sz with cross."""
    for i in range(4):
        for j in range(4):
            t.fd(80)
            t.rt(90)
        t.rt(90)
 
wn = turtle.Screen()

    alex.rt(18)
