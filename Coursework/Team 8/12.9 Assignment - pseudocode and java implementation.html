import java.util.Scanner;
 
public class MagicSquare {
    static Scanner in = new Scanner(System.in);
    public static int[][] Grid;
 
    public static void main(String[] args) {
        while (true) {
            System.out.println("Please input an odd number:");
            int scale = in.nextInt();
            if (scale % 2 == 1) {
            } else {
                System.out.println("Odd number only!");
                continue;
            }
            Grid = new int[scale][scale];
            Construct(scale);
            show(scale);
            if (isMS(scale) == true) {
                System.out.println("This IS a Magic Square!");
            } else {
                System.out.println("This is NOT a Magic Square!");
            }
        }
    }
 
    public static int getDiagnalSum(int scale) {
        int sum = 0;
        for (int i = 0; i < scale; i++) {
            sum += Grid[i][i];
        }
        return sum;
    }
 
    public static int getMinorDiagnalSum(int scale) {
        int sum = 0;
        for (int i = 0; i < scale; i++) {
            sum += Grid[scale - 1 - i][i];
        }
        return sum;
    }
 
    public static int getRowSum(int scale, int n) {
        int sum = 0;
        for (int i = 0; i < scale; i++) {
            sum += Grid[n][i];
        }
        return sum;
    }
 
    public static int getColumnSum(int scale, int n) {
        int sum = 0;
        for (int i = 0; i < scale; i++) {
            sum += Grid[i][n];
        }
        return sum;
    }
 
    public static boolean isMS(int scale) {
        int Sum = (scale * scale + 1) * scale / 2;
        for (int i = 0; i < scale; i++) {
            if (getColumnSum(scale, i) != Sum || getRowSum(scale, i) != Sum) {
                return false;
            }
        }
        if (getDiagnalSum(scale) != Sum) {
            return false;
        } else if (getMinorDiagnalSum(scale) != Sum) {
            return false;
        } else {
            return true;
        }
    }
    public static void Construct(int scale){
        int row=0;
        int column=(scale-1)/2;
        int newrow=0;
        int newcolumn=0;
//        for(int i=0;i<scale;i++){
//            for (int j=0;j<scale;j++){
//                Grid[i][j]=0;
//            }
//        }
        Grid[row][column]=1;
        for (int i=2;i<=scale*scale;i++){
            newrow=row-1;
            newcolumn=column+1;
            if (newrow==-1 && newcolumn==scale){
                newrow=row+1;
                newcolumn=column;
            }
            else if(newrow==-1){
                newrow=scale-1;
            }
            else if(newcolumn==scale){
                newcolumn=0;
            }
            else if(Grid[newrow][newcolumn]!=0){
                newrow=row+1;
                newcolumn=column;
            }
            Grid[newrow][newcolumn]=i;
            row=newrow;
            column=newcolumn;
        }
    }
    public static void show(int scale){
        for (int i=0;i<scale;i++){
            for (int j=0;j<scale;j++){
                System.out.print(Grid[i][j]+"\t");
            }
            System.out.println("");
        }
    }
}
