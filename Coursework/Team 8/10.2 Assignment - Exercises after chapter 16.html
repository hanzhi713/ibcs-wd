import point
from unittesting import test
class Point:
    """ Point class represents and manipulates x,y,z coords. """
 
    def __init__(self,x=0,y=0,z=0):
        self.x = x
        self.y = y
        self.z = z
    def __str__(self):
        return "({0},{1},{2})".format(self.x,self.y,self.z)
 
    def set_x(self,x):
        self.x=x
    def set_y(self,y):
        self.y=y
    def set_z(self,z):
        self.z=z
    def get_x(self):
        return self.x
    def get_y(self):
        return self.y
    def get_z(self):
        return self.z
    def coord(self):
        print(self.x,self.y,self.z)
 
    def distance_from_point(self,other=None):
        if other == None:
           other=Point(0,0,0)
        return ((self.x-other.x)**2+(self.y-other.y)**2+(self.z-other.z)**2)**0.5
 
    def slope_with_point(self,other=None):
        if other == None:
            other=Point(0,0,0)
        elif self.x == other.x:
            return "It's vertical."
        return ((self.y-other.y)/(self.x-other.x))
 
    def angle_with_xaxis(self,other=None):
        from math import atan,degrees
        if other == None:
            other=Point(0,0,0)
        if self.x == other.x:
            return "90"
        else:
            return degrees(atan(self.slope_with_point(other)))
 
    def mid_point(self,other=None,k=0.5):
        if other == None:
            other=Point(0,0,0)
        result=Point(self.x-k*(self.x-other.x),self.y-k*(self.y-other.y),self.z-k*(self.z-other.z))
        return result.__str__()
 
 
class Rectangle:
 
    def __init__(self,corner=Point(0,0),width=0,height=0):
        self.corner=corner
        self.width=width
        self.height=height
        self.center=Point(self.corner.x+self.width/2,self.corner.y+self.height/2)
 
    def __str__(self):
        return "({0},{1},{2})".format(self.corner,self.width,self.height)
 
    def set_corner(self,corner):
        self.corner=corner
    def set_width(self,width):
        self.width=width
    def set_height(self,height):
        self.height=height
    def get_corner(self):
        return self.corner
    def get_width(self):
        return self.width
    def get_height(self):
        return self.height
 
    def grow(self,delta_width,delta_height):
        self.width+=delta_width
        self.height+=delta_height
 
    def move(self,dx,dy):
        self.corner.x+=dx
        self.corner.y+=dy
 
    def area(self):
        return self.width*self.height
 
    def perimeter(self):
        return 2*(self.width+self.height)
 
    def flip(self):
        a=self.width
        self.width=self.height
        self.height=a
 
    def contains(self,pt):
        if pt.x>=self.corner.x and pt.x<self.corner.x+self.width:
            if pt.y>=self.corner.y and pt.y<self.corner.y+self.height:
                return True
        return False
 
    def is_collide(self,other):
        if abs(self.center.x-other.center.x)>=(self.width+other.width)/2:
            if abs(self.center.y-other.center.y)>=(self.height+other.height)/2:
                return True
        return False
 
r = Rectangle(Point(0, 0), 10, 5)
test(r.area() == 50)
test(r.perimeter() == 30)
r = Rectangle(Point(100, 50), 10, 5)
test(r.width == 10 and r.height == 5)
r.flip()
test(r.width == 5 and r.height == 10)
r = Rectangle(Point(0, 0), 10, 5)
test(r.contains(Point(0, 0)))
test(r.contains(Point(3, 3)))
test(not r.contains(Point(3, 7)))
test(not r.contains(Point(3, 5)))
test(r.contains(Point(3, 4.99999)))
test(not r.contains(Point(-3, -3)))
