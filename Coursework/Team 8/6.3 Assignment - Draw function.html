import turtle
import math
import time
wn = turtle.Screen()
wn.bgcolor("black")
wn.title("graph")
 
scale=turtle.Turtle()
scale.pensize(2)
scale.speed(0)
scale.pencolor('white')
scale.penup()
scale.setpos(-240,0)
scale.pendown()
scale.fd(40)
for i in range(-6,5):
    scale.left(90)
    scale.fd(5)
    scale.back(5)
    scale.right(90)
    scale.write(i+1,align="right")
    scale.fd(40)
scale.stamp()
