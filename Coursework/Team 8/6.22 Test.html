def overlapping(a,b):
    check=0
    for i in range(0,len(a)):
        for j in range(0,len(b)):
            if a[i]==b[j]:
                check=1
    if check == 1:
        return 'Yes'
    else:
        return 'No'

import turtle
import random
 
def plot(a):
    extent=[]
    Sum=sum(a)
    for i in range(0,len(a)):
        extent=extent+[a[i]/Sum]
    #set up
    wn=turtle.Screen()
    wn.bgcolor('light green')
    wn.colormode(255)
    ar=turtle.Turtle()
    ar.pencolor('black')
    ar.speed(0)
    ar.pensize(2)
    for j in range(0,len(a)):
        r=int(random.uniform(0,256))
        g=int(random.uniform(0,256))
        b=int(random.uniform(0,256))
        ar.fillcolor((r,g,b))
        ar.begin_fill()
        ar.rt(180)
        ar.fd(200)
        ar.left(90)
        ar.circle(200,extent[j]*180)
        ar.penup()
        ar.left(90)
        ar.fd(40)
        lable=str(chr(65+j))+'='+str(a[j])+'/'+str(Sum)
        ar.write(lable,align="center",font=("Arial",18, "normal"))
        ar.back(40)
        ar.rt(90)
        ar.pendown()
        ar.circle(200,extent[j]*180)
        ar.left(90)
        ar.fd(200)
        ar.end_fill()
    ar.hideturtle()
    wn.mainloop()
 
print("Input your data, write 'stop' to stop.")
newy=[]
while 1:
    y=input('raw data are:  ')
    if y == 'stop':
        break
    newy.append(int(y))
plot(newy)
