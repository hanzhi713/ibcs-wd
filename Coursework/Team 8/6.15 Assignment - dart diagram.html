import turtle
import random
#set up
wn=turtle.Screen()
wn.bgcolor('white')
ch=turtle.Turtle()
ch.penup()
ch.hideturtle()
ch.speed(0)
ch.setpos(0,-100)
 
#draw target
x=18
while x>=0:
    if x % 4 == 2:
        ch.fillcolor('green')
        ch.begin_fill()
        ch.circle(x*10+2.5)
        ch.end_fill()
    else:
        ch.fillcolor('red')
        ch.begin_fill()
        ch.circle(x*10+2.5)
        ch.end_fill()
    x=x-2
    ch.write(int(10-(x/2+1)),align="center",font=("Arial", 8, "normal"))
    ch.left(90)
    ch.forward(20)
    ch.right(90)
 
#input
a=input('Are you precise?')
b=input('Are you accurate?')
 
#judge
if a=="yes" and b=="yes":
    for j in range(0,20):
        ang=random.uniform(0,360)
        dist=random.normalvariate(0,10)
        ch.right(ang)
        ch.fd(dist)
        ch.write('x',align="center",font=("Arial", 18, "bold"))
        ch.right(180-ang)
        ch.back(dist)
elif a=="yes" and b=="no":
    print('precise but not accurate')
    ang=random.uniform(0,360)
    dist=random.uniform(100,180)
    ch.right(ang)
    ch.fd(dist)
    ch.write('x',align="center",font=("Arial", 18, "bold"))
    for j in range(0,19):
        ang=random.uniform(0,360)
        dist=random.normalvariate(0,10)
        ch.right(ang)
        ch.fd(dist)
        ch.write('x',align="center",font=("Arial", 18, "bold"))
        ch.right(180-ang)
        ch.back(dist)
 
elif a=="no" and b=="no":
    ang=random.uniform(0,360)
    dist=random.uniform(100,180)
    ch.right(ang)
    ch.fd(dist)
    ch.write('x',align="center",font=("Arial", 18, "bold"))
    for j in range(0,19):
        ang=random.uniform(0,360)
        dist=random.normalvariate(0,100)
        ch.right(ang)
        ch.fd(dist)
        ch.write('x',align="center",font=("Arial", 18, "bold"))
        ch.right(180-ang)
        ch.back(dist)
 
elif a=="no" and b=="yes":
    for j in range(0,20):
        ang=random.uniform(0,360)
        dist=random.normalvariate(0,100)
        ch.right(ang)
        ch.fd(dist)
        ch.write('x',align="center",font=("Arial", 18, "bold"))
        ch.setpos(0,100)
else:
    print('Just yes or no!')
wn.mainloop()
