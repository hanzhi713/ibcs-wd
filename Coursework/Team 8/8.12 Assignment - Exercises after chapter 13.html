def filereverse(From,To):
    handle1 = open(From)
    handle2 = open(To,'w')
    line = handle1.readlines()
    line[len(line)-1]+='\n'
    line[0]=line[0][:len(line[0])-1]
    line.reverse()
    for i in line:
        handle2.write(i)
    handle1.close()
    handle2.close()

def nosnake(From,To):
    handle1 = open(From)
    handle2 = open(To,'w')
    line = handle1.readlines()
    for i in line:
        if not 'snake' in i:
            handle2.write(i)
    handle1.close()
    handle2.close()

def number(From,To):
    handle1 = open(From)
    handle2 = open(To,'w')
    line = handle1.readlines()
    num='0123456789'
    seq=0
    for i in line:
        check=1
        seq+=1
        handle2.write(str(seq))
        handle2.write(' ')
        if seq<10:
            handle2.write(' ')
        try:
            if i[0] in num and i[1] in num and i[2] in num and i[3] in num and i[4]==' ':
                check=0
        except:
            check=1
        if check==0:
            handle2.write(i[5:])
        else:
            handle2.write(i)
    handle1.close()
    handle2.close()
