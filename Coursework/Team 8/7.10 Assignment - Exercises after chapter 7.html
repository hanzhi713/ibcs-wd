def sqrt(n):
    approx = n/2.0
    while True:
        better = (approx + n/approx)/2.0
        if abs(approx - better) < 0.00001:
            return better
        approx = better
        print(better)

def num_digits(n):
    if n>0:
        None
    elif n==0:
        return 1
    else:
        n=-n
    count = 0
    while n != 0:
        count = count + 1
        n = n // 10
    return count
 
##>>> num_digits(0) == 1
##True
##>>> num_digits(-12345) == 5
##True

def num_even_digits(n):
    if n>0:
        None
    elif n==0:
        return 1
    else:
        n=-n
    count = 0
    n=list(str(n))
    for i in n:
        if int(i)%2==0:

def play_once(human_plays_first,you_win,drawn,you_lose):
    """
        Must play one round of the game. If the parameter
        is True, the human gets to play first, else the
        computer gets to play first. When the round ends,
        the return value of the function is one of
        -1 (human wins), 0 (game drawn), 1 (computer wins).
    """
    import random
    rng=random.Random()
    result=rng.randrange(-1,2)
    print("Human plays first={0}, winner={1} "
.format(human_plays_first, result))
    return result
 
def play_first(times):
    if times%2==0:
        play_first=True
        print('You play first')
    else:
        play_first=False
        print('I play first')
    return play_first
 
def game(times=0,you_win=0,drawn=0,you_lose=0):
    times+=1
    result=play_once(play_first(times),you_win,drawn,you_lose)
    if result==-1:
        print('You win!')
        you_win+=1
