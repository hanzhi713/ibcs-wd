    for i in range(len(lst)-1):
        if lst[i+1]<lst[i]:
            return False
    return True

def is_anagrams(a,b):
    a=list(a)
    b=list(b)
    a.sort()
    b.sort()
    if len(a)!=len(b):
        return False
    for i in range(len(a)):
        if a[i]!=b[i]:
            return False
    return True
print(is_anagrams('123','231'))

def remove_duplicates(lst):
    lst2=[]
    for i in lst:
        if lst.count(i)==1:
            lst2.append(i)
    return lst2
print(remove_duplicates([2,3,[1,2],[1,2],'g','g',2]))
