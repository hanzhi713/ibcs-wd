ColorGrid:
package program;
 
public class ColorGrid {
    private String[][] pixels;
    private int rows;
    private int cols;
 
    public ColorGrid(String s, int numRows, int numCols) {
        pixels = new String[numRows][numCols];
        for (int i = 0; i < numRows; i++) {
            for (int j = 0; j < numRows; j++) {
                pixels[i][j] = s.substring(i * numRows + j, i * numRows + j + 1);
            }
        }
    }
 
    public void paintRegion(int row, int col, String newColor, String oldColor) {
        if (row >= 0 && col >= 0 && row < rows && col < cols) {
            if (!pixels[row][col].equals(newColor)) {
                if (pixels[row][col].equals(oldColor)) {
                    pixels[row][col] = newColor;
                    paintRegion(row + 1, col, newColor, oldColor);
                    paintRegion(row - 1, col, newColor, oldColor);
                    paintRegion(row, col + 1, newColor, oldColor);
                    paintRegion(row, col - 1, newColor, oldColor);
                }
            }
        }
    }
}

package program;
 
public class Sentence {
    private String sentence;
    private int numWords;
 
    public Sentence(String str) {
        sentence = new String();
        sentence = str;
        numWords = 0;
        for (int i = 0; i < str.length(); i++) {
            if (str.charAt(i) == ' ') {
                numWords += 1;
            }
        }
        numWords += 1;
    }
 
    public int getNumWords() {
        return numWords;
    }
 
    public String getSentence() {
        return sentence;
    }
 
    private static String removeBlanks(String s) {
        String result = new String();
        result = "";
        for (int i = 0; i < s.length(); i++) {
            if (!(s.charAt(i) == ' ')) {
                result += s.charAt(i);
            }
        }
        return result;
    }
 
    private static String lowerCase(String s) {
        return s.toLowerCase();
    }
 
    private static String removePunctuation(String s) {
        String result = new String();
        char[] punc = { ' ', ',', '.', '!', '?', '(', ')', '/', '"', '\'', '\\', ':', ';', '<', '>', '*', '&', '^', '%',
                '$', '#', '@', '~', '`', '+', '-', '_' };
        boolean ispunctuation = false;
        char curchar;
        result = "";
        for (int i = 0; i < s.length(); i++) {
            curchar = s.charAt(i);
            for (char x : punc) {
                if (x == curchar) {
                    ispunctuation = true;
                } else {
                    ispunctuation = false;
                }
            }
            if (!ispunctuation) {
                result += s.charAt(i);
            }
        }
        return result;
    }
 
    private boolean isPalindrome(String s, int lb, int ub) {
        if (ub - lb == -1 || ub - lb == 0) {
            return true;
        } else {
            if (s.charAt(lb) == s.charAt(ub)) {
                return isPalindrome(s, lb + 1, ub - 1);
            } else {
                return false;
            }
        }
    }
 
    public boolean isPalindrome() {
        String temp = removeBlanks(sentence);
        temp = removePunctuation(temp);
        temp = lowerCase(temp);
        return isPalindrome(temp, 0, temp.length() - 1);
    }
}
