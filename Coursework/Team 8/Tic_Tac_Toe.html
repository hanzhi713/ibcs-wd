package program;
 
import java.util.Scanner;
import java.lang.Math;
 
public class Tic_Tac_Toe {
    private static int[][] table = new int[][] { { 0, 0, 0 }, { 0, 0, 0 }, { 0, 0, 0 } };
    private static Scanner in = new Scanner(System.in);
    private static String[] signs = new String[] { " ", "O", "X" };
 
    private static void show() {
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                System.out.print("| " + signs[table[i][j]] + " |");
            }
            System.out.println("");
        }
    }
 
    private static void input(int who) {
        while (true) {
            System.out.println("which row? (0 for the top; 1 for the middle; 2 for the bottom)");
            int row = in.nextInt();
            System.out.println("which column? (0 for the left; 1 for the middle; 2 for the right)");
            int column = in.nextInt();
            if (table[row][column] == 0) {
                table[row][column] = who;
                break;
            } else {
                System.out.println("This block has been taken! Choose another one.");
            }
        }
        show();
    }
 
    public static boolean check() {
        for (int i = 0; i < 3; i++) {
            if (table[i][0] == table[i][1] && table[i][0] == table[i][2] && table[i][0] != 0) {
                return true;
            } else if (table[0][i] == table[1][i] && table[0][i] == table[2][i] && table[0][i] != 0) {
                return true;
            }
        }
        if (table[0][0] == table[1][1] && table[0][0] == table[2][2] && table[0][0] != 0) {
            return true;
        } else if (table[0][2] == table[1][1] && table[2][0] == table[1][1] && table[1][1] != 0) {
            return true;
        }
        return false;
    }
 
    public static void main(String[] args) {
        System.out.println("Who first? (Input '1' for player1 or '2' for player2.)");
        int who = in.nextInt();
        boolean over = false;
        show();
        while (!over) {
            System.out.println("Player" + who + " now!");
            input(who);
            over = check();
            who = 3-who;
        }
        if (who == 1) {
            System.out.println("player2 wins!");
        } else {
            System.out.println("player1 wins!");
        }
    }
 
}
