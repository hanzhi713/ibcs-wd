import cv2
import numpy as np
import random
from matplotlib import pyplot as plt
img_bgr = cv2.imread('testimg.jpg')
img_rgb = np.zeros(img_bgr.shape, img_bgr.dtype)
img_rgb[:, :, 0] = img_bgr[:, :, 2]
img_rgb[:, :, 1] = img_bgr[:, :, 1]
img_rgb[:, :, 2] = img_bgr[:, :, 0]
rng=random.Random()
for i in range(int(img_rgb.shape[0]*img_rgb.shape[1]*0.2)):
    xcor=rng.randint(0,img_rgb.shape[0]-1)
    ycor=rng.randint(0,img_rgb.shape[1]-1)
    color=rng.randint(0,1)
    img_rgb[xcor,ycor]=[255*color,255*color,255*color]
plt.imshow(img_rgb)
plt.show()
