import turtle
 
pensize=10
scr=turtle.Screen()
tess=turtle.Turtle()
tess.speed(0)
tess.pensize(pensize)
def cr():
    tess.pencolor('red')
def cb():
    tess.pencolor('blue')
def cg():
    tess.pencolor('green')
def pp():
    global pensize
    pensize+=2
    if pensize>=20:
        pensize=20
    tess.pensize(pensize)
def pm():
    global pensize
    pensize-=2
    if pensize<=1:
        pensize=1
    tess.pensize(pensize)
def left():
    tess.left(10)
def right():
    tess.rt(10)
def forward():
    tess.fd(10)
def back():
    tess.back(10)
 
scr.onkey(cr,'r')
scr.onkey(cb,'b')
scr.onkey(cg,'g')
scr.onkey(pp,'=')
scr.onkey(pm,'-')
scr.onkey(left,'Left')
scr.onkey(right,'Right')
scr.onkey(forward,'Up')
scr.onkey(back,'Down')
scr.listen()
 
scr.mainloop()

import turtle
 
wn=turtle.Screen()
wn.title("Tess becomes a traffic light!")
wn.bgcolor("lightgreen")
tess = turtle.Turtle()
tess.speed(0)
def draw_housing():
    """ Draw a nice housing to hold the traffic lights """
    tess.pensize(3)
    tess.color("black", "darkgrey")
    tess.begin_fill()
    tess.forward(80)
    tess.left(90)
    tess.forward(200)
    tess.circle(40, 180)
    tess.forward(200)
    tess.left(90)
    tess.end_fill()
 
draw_housing()
tess.penup()
tess.fd(40)
tess.left(90)
tess.fd(40)
tess.shape('circle')
tess.shapesize(3)
tess.fillcolor('green')
state_num=0
 
def automatic_state_machine():
    global state_num
    if state_num == 0:
        tess.forward(70)
        tess.fillcolor("orange")
        state_num = 1
        wn.ontimer(automatic_state_machine,600)
    elif state_num == 1:
        tess.forward(70)
        tess.fillcolor("red")
        state_num = 2
        wn.ontimer(automatic_state_machine,6000)
    elif state_num == 2:
        tess.back(70)
        tess.fillcolor("orange")
        state_num = 3
        wn.ontimer(automatic_state_machine,600)
    else:
        tess.back(70)
        tess.fillcolor("green")
        state_num = 0
        wn.ontimer(automatic_state_machine,6000)
 
automatic_state_machine()
 
wn.mainloop()
