The birthdat problem:
import matplotlib.pyplot as plt
import random
import time
 
t1=time.clock()
result=[0]
rng=random.Random()
birth=[]
for i in range(365):
    birth.append(0)
reset=birth[:]
 
def trail(num):
    global birth,reset
    birth=reset[:]
    for i in range(num):
        a=rng.randint(0,364)
        if birth[a]==1:
            return 1
        birth[a]=1
    return 0
 
for num in range(1,101):
    count=0
    for i in range(100000):
        count+=trail(num)
    result.append(count/100000)
    if num%10==0:
        print("*",end="  ")
plt.plot(result)
plt.ylabel('probability')
plt.xlabel('number of students')
plt.title('The probability of existing same birthdays')
t2=time.clock()
t=t2-t1
print()
print('The time used is {0} minutes and {1:.2f} seconds.'.format(t//60,t%60))
plt.show()
