class Point:
    """ Point class represents and manipulates x,y,z coords. """
 
    def __init__(self,x=0,y=0,z=0):
        self.x = x
        self.y = y
        self.z = z
    def __str__(self):
        return "({0},{1},{2})".format(self.x,self.y,self.z)
 
#member function
    def set_x(self,x):
        self.x=x
    def set_y(self,y):
        self.y=y
    def set_z(self,z):
        self.z=z
    def get_x(self):
        return self.x
    def get_y(self):
        return self.y
    def get_z(self):
        return self.z
    def coord(self):
        print(self.x,self.y,self.z)
 
    def distance_from_point(self,other=None):
        if other == None:
           other=Point(0,0,0)
        return ((self.x-other.x)**2+(self.y-other.y)**2+(self.z-other.z)**2)**0.5
 
    def slope_with_point(self,other=None):
        if other == None:
            other=Point(0,0,0)
        elif self.x == other.x:
            return "It's vertical."
        return ((self.y-other.y)/(self.x-other.x))
 
    def angle_with_xaxis(self,other=None):
        from math import atan,degrees
        if other == None:
            other=Point(0,0,0)
        if self.x == other.x:
            return "90"
        else:
            return degrees(atan(self.slope_with_point(other)))
 
    def mid_point(self,other=None,k=0.5):
        if other == None:
            other=Point(0,0,0)
        result=Point(self.x-k*(self.x-other.x),self.y-k*(self.y-other.y),self.z-k*(self.z-other.z))
        return result.__str__()
