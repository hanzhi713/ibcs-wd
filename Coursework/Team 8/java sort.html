package program;
 
public class Sort {
    public static void selectionSort(int[] lst) {
        for (int ind = 0; ind < lst.length-1; ind++) {
            int min = ind;
            for (int i = ind + 1; i < lst.length; i++) {
                if (lst[i] < lst[min]) {
                    min = i;
                }
            }
            int temp = lst[ind];
            lst[ind] = lst[min];
            lst[min] = temp;
        }
    }
 
    public static String showArray(int[] lst) {
        String result = new String();
        result = "[ ";
        for (int i : lst) {
            result = result + i + " , ";
        }
        result = result.substring(0,result.length()-2)+"]";
        return result;
    }
 
    public static void insertionSort(int[] lst){
        for(int i=1;i<lst.length;i++){
            int curval = lst[i];
            for (int j=i-1;j>=-1;j--){
                if (j==-1){
                    lst[0]=curval;
                }
                else if (curval<lst[j]){
                    lst[j+1] = lst[j];
                }
                else{
                    lst[j]=curval;
                    break;
                }
            }
        }
    }
 
    public static void main(String[] args) {
        int[] lst1 = { 4, 3, 2, 1 };
        System.out.println("Before:");
        System.out.println(showArray(lst1));
        selectionSort(lst1);
        System.out.println("After:");
        System.out.println(showArray(lst1));
        int[] lst2 = { 4, 3, 2, 1 };
        System.out.println("Before:");
        System.out.println(showArray(lst2));
        insertionSort(lst2);
        System.out.println("After:");
        System.out.println(showArray(lst2));
    }
 
}
