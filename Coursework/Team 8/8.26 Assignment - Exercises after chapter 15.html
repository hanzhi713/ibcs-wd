class Point:
    """ Create a new Point, at coordinates x, y """
 
    def __init__(self, x=0, y=0):
        """ Create a new point at x, y """
        self.x = x
        self.y = y
 
    def distance_from_origin(self):
        """ Compute my distance from the origin """
        return ((self.x ** 2) + (self.y ** 2)) ** 0.5
 
    def __str__(self):
        return "({0}, {1})".format(self.x, self.y)
 
    def halfway(self, target):
        """ Return the halfway point between myself and the target """
        mx = (self.x + target.x)/2
        my = (self.y + target.y)/2
        return Point(mx, my)
 
    ##(2)
    def reflect_x(self):
        return Point(self.x,-self.y)
 
    ##(3)
    def slope_from_origin(self):
        return self.y/self.x
 
    ##(4)
    def get_line_to(self,target):
        m=(target.y-self.y)/(target.x-self.x)
        c=target.y-m*target.x
        return (m,c)

def distance(self,target):
    dx=self.x-target.x
    dy=self.y-target.y
    return (dx**2+dy**2)**0.5

    p=a.halfway(b)

    return (m,c)
 
def find_intersection(a,b):
    try:
        x=(a[1]-b[1])/(b[0]-a[0])
        y=(b[0]*a[1]-a[0]*b[1])/(b[0]-a[0])
        return Point(x,y)
    except:
        return None

from unittest import test
 
class SMS_store:
    '''The class will instantiate SMS_store objects,
        similar to an inbox or outbox on a cellphone'''
    def __init__(self,lst=[]):
        self.message=lst
 
    def add_new_arrival(self,from_number, time_arrived, text_of_SMS):
        self.message.append((False,from_number, time_arrived, text_of_SMS))
 
    def message_count(self):
        return len(self.message)
 
    def get_unread_indexes(self):
        result=[]
        for (ind,msg) in enumerate(self.message):
            if msg[0] == False:
                result.append(ind)
        return result
 
    def get_message(self,i):
        try:
            self.message[i]=(True,self.message[i][1],self.message[i][2],self.message[i][3])
            return self.message[i][1:]
        except:
            return None
 
    def delete(self,i):
        del self.message[i]
 
    def clear(self):
        self.message=[]
 
my_inbox = SMS_store()
my_inbox.add_new_arrival(110,'00:00', 'WANTED')
my_inbox.add_new_arrival(120,'01:00', 'DEAD')
my_inbox.add_new_arrival(119,'03:00', 'FIRED')
my_inbox.add_new_arrival(10086,'12:00', 'BILL')
test(my_inbox.message_count()==4)
test(my_inbox.get_message(2)==(119,'03:00', 'FIRED'))
test(my_inbox.get_unread_indexes()==[0,1,3])
my_inbox.delete(2)
test(my_inbox.message_count()==3)
my_inbox.clear()
test(my_inbox.message_count()==0)
