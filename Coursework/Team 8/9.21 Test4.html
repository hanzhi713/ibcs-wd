From=['Alice','ALICE','She ','she ','Herself','Her ',' herself',' hers',' her ',' her',' she\n','girl']
To=['Arthur','ARTHUR','He ','he ','Himself','His ',' himself',' his',' his ',' him',' he\n','boy']
def clear(a):
    x=len(a)
    if x<5:
        return a
    else:
        for i in range(len(From)):
            while True:
                ind=a.find(From[i])
                if ind==-1:
                    break
                else:
                    a=a[:ind]+To[i]+a[ind+len(From[i]):]
        return a
handle3 = open('textfile_alice_on_wonderland.txt','r')
handle4 = open("textfile_alice_on_wonderland_Arthur.txt",'w')
while True:
    line=handle3.readline()
    if len(line)==0:
        break
    line=clear(line)
    handle4.write(line)
handle3.close()
handle4.close()
 
print('3 pass')
