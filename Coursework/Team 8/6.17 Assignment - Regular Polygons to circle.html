import turtle
import math
 
def poly(t,x):
    t.penup()
    t.right(90)
    t.fd(math.sin(math.radians(90-180/x))*200)
    t.pendown()
    t.right(90)
    t.fd(math.cos(math.radians(90-180/x))*200)
    t.right(180)
    for i in range(0,x):
        t.fd(math.cos(math.radians(90-180/x))*400)
        t.left(360/x)
    t.penup()
    t.setpos(0,0)
    t.pendown()
 
wn = turtle.Screen()
wn.bgcolor("lightgreen")
wn.title("standard polygens")
 
alex = turtle.Turtle()
alex.hideturtle()
alex.setpos(0,0)
alex.pencolor('black')
alex.speed(0)
alex.pensize(1)
 
x=3
while x<30:
    poly(alex, x)
    x+=1
alex.pencolor('red')
alex.pensize(5)
poly(alex,x)
wn.mainloop()
