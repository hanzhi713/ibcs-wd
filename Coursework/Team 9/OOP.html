class Point3D:
    """ Create a new Point, at coordinates x, y """
 
    def __init__(self, x=0, y=0,z=0):
        self.x = x
        self.y = y
        self.z = z
    def distance_from_origin(self):
 
        s = (self.x**2+self.y**2+self.z**2)**0.5
        return s
    def distance_to_other(self,target):
        s = ((target.x-self.x)**2+(target.y-self.y)**2+(target.z-self.z)**2)**0.5
        return s
    def between(self,target):
        coordx = (self.x + target.x)/2
        coordy = (self.y + target.y)/2
        coordz = (self.z + target.z)/2
        return Point3D(coordx,coordy,coordz)
 
 
 
a=Point3D(1,1,1)
b=Point3D(3,3,3)
 
print(a.mid(b))
