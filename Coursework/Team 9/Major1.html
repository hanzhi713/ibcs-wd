def choose_small(a,b):
    if a > b:
        return(b)
    else:
        return(a)
 
def word_sorted(word1,word2):
    limit = choose_small(len(word1),len(word2))
    for index in range(0,limit):
        if word1[index]>word2[index]:
            return(word2,word1)
        elif word1[index] == word2[index]:
            continue
        else:
            return(word1,word2)
print(word_sorted("death","daddy"))

def is_perfect(n):
    sum = 0
    for count in range(1, n//2+1):
        if n % count == 0:
            sum = sum + count
    if sum == n:
        return("perfect")
    else:
        return("not_perfect")
 
print(is_perfect(28))
