import turtle
 
def Draw_Square(t, l):
    """Make turtle t Draw_Square"""
    for i in range(4):
        t.forward(l)
        t.left(90)
    t.penup()
    t.fd(2*l)
    t.pendown()
    t.hideturtle()

A.pencolor('black')

    Draw_Square(A, 20)

import turtle
 
def Draw_Square(t, l):
    """Make turtle t draw a square"""
    for i in range(4):
        t.forward(l)
        t.left(90)
    t.penup()
    t.right(180)
    t.forward(10)
    t.left(90)
    t.fd(10)
    t.left(90)
    t.pendown()

B.speed(0)
B.pencolor("black")

    Draw_Square(B, l+20*i)
