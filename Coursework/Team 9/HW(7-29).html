def dot_product(u,v):
    s = []
    a = u[1]*v[2] - u[2]*v[1]
    b = u[2]*v[0] - u[0]*v[2]
    c = u[0]*v[1] - u[1]*v[0]
    s.append(a)
    s.append(b)
    s.append(c)
    return s
