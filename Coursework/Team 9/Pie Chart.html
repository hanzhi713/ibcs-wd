import turtle
import random
 
T = turtle.Turtle()
T.pensize(2)
T.hideturtle()
 
wn = turtle.Screen()
wn.bgcolor("yellow")
wn.title("The Piechart")
wn.colormode(255)
 
n = [1,2,4,6]
a = len(n)
for l in n:
    T.begin_fill()
    T.fillcolor(int(random.uniform(0,255)),int(random.uniform(0,255)),int(random.uniform(0,255)))
    T.left(90)
    T.fd(200)
    T.left(90)
    T.circle(200,360/sum(n)*l)
    T.left(90)
    T.fd(200)
    T.end_fill()
    T.left(90)
 
wn.mainloop()
