import turtle
 
turtle.setup(400,500)                # Determine the window size
wn = turtle.Screen()                 # Get a reference to the window
wn.title("Handling keypresses!")     # Change the window title
wn.bgcolor("lightgreen")             # Set the background color
tess = turtle.Turtle()# Create our favorite turtle
t = 1
# The next four functions are our "event handlers".
def h1():
   tess.forward(30)
 
def h2():
   tess.left(45)
 
def h3():
   tess.right(45)
 
def h4():
    wn.bye()# Close down the turtle window
def add():
    global t
    print(t)
    if t <20:
        t = t + 1
        tess.pensize(t)
def minus():
    global t
    print(t)
    if t > 1:
        t = t - 1
        tess.pensize(t)
# These lines "wire up" keypresses to the handlers we've defined.
wn.onkey(h1, "Up")
wn.onkey(h2, "Left")
wn.onkey(h3, "Right")
wn.onkey(h4, "q")
wn.onkey(add, "+")
wn.onkey(minus, "-")
 
# Now we need to tell the window to start listening for events,
# If any of the keys that we're monitoring is pressed, its
# handler will be called.
wn.listen()
wn.mainloop()

def turtles(t, r, c):
 
    t.speed(0)
    t.forward(40)
    t.left(90)
    t.forward(50)
    t.shape("circle")
    t.shapesize(3)
    t.forward(r)
    t.fillcolor(c)
 
 
turtles(green, 0, "blue")
turtles(orange, 70, "blue")
turtles(red, 140, "blue")
state_num = 0
wn.listen()
 
def advance_state_machine():
    global state_num
    if state_num == 0:
        orange.fillcolor('orange')
        green.fillcolor('blue')
        wn.ontimer(advance_state_machine, '1000')
        state_num = 1
    elif state_num == 1:
        red.fillcolor('red')
        orange.fillcolor('blue')
        wn.ontimer(advance_state_machine, '2000')
        state_num = 2
    elif state_num == 2:
        green.fillcolor('green')
        red.fillcolor('blue')
        wn.ontimer(advance_state_machine, '3000')
        state_num = 3
    else:
        orange.fillcolor('orange')
        wn.ontimer(advance_state_machine, '1000')
        state_num = 0
wn.ontimer(advance_state_machine, 0)
green.showturtle()
wn.mainloop()
