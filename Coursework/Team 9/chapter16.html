class Point:
 
    def __init__(self, x, y):
        self.x = x
        self.y = y
 
    def setX(self, x):
        self.x = x
 
    def setY(self,y):
        self.y = y
 
    def getX(self):
        return self.x
 
    def getY(self):
        return self.y
 
    def midPoint(self, target):
        t = Point(0.5 * (self.x + target.x), 0.5 * (self.x + target.x))
        return t
 
    def reflect_x(self):
        reflection = Point(self.x, self.y)
        return reflection
 
    def slope(self):
        i = round(self.y / self.x, 3)
        return i
 
    def draw_line(self, target):
        slope = (self.y - target.y) / (self.x - target.x)
        y_intercept = self.y - slope * self.x
        return (slope, y_intercept)
 
    def __str__(self):
        return "({0},{1})".format(self.x, self.y)
