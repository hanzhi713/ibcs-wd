def salt(image,x):
    a=x
    h, w = image.size
    pixel=image.load()
    for i in range(10000):
        x= int(random.uniform(0,h))
        y= int(random.uniform(0,w))
        pixel[x,y]= (255,255,255)
    if a == 0:
        image.show()
    elif a == 1:
        name = s.get()
        image.save(name)
 
 
def sketch(image,x):
    threshold = 10
    if threshold < 0:
        threshold = 0
    if threshold > 60:
        threshold  = 60
 
    width,height = image.size
    img = image.convert('L')
    pixel = img.load()
 
    for w in range(width):
        for h in range(height):
            if w == width -1 or h == height - 1:
                continue
 
            screen = pixel[w,h]
            distance = pixel[w+1,h+1]
 
            difference = abs(screen - distance)
 
            if difference >= threshold:
                pixel[w,h] = 0
            else:
                pixel[w,h] = 255
 
    if x == 0:
        img.show()
    if x == 1:
        name = s.get()
        image.save(name)
