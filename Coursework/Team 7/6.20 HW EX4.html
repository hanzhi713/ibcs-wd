def draw_sq(u,sz):
       for i in range(4):
              u.forward(sz)
              u.right(90)
              u.forward(sz)

f = turtle.Turtle()
f.pensize(3)
f.speed(0)
f.left(90)
l = (20)
for k in range(l):
       f.forward(100)
       f.right(90)
       draw_sq(f, 100)
       f.right(90)
       f.forward(100)
       f.left(180)
       f.right(360/l)
