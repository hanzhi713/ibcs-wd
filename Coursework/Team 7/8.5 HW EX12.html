def extract_words(f):
      f = f.split(" ")
      d = []
      for i in f:
            ff = " "
            for h in i:
                  if h not in string.punctuation:
                        ff += h
            d.append(ff)
      return d
 
f = extract_words("she tried to curtsey as she spoke--fancy")

def wordset(f):
      ff = []
      for i in f:
            if i not in ff:
                  ff.append(i)
      ff.sort()
      return ff
 
f = wordset(["now", "is", "time", "is", "now", "is", "is"])
print(f)
