import turtle
wn = turtle.Screen()
wn.bgcolor("alice blue")
wn.title('TREE')
 
g = turtle.Turtle()
g.penup()
g.setx(-25)
g.sety(-200)
g.pendown()
g.pencolor('rosy brown')
g.pensize(3)
g.fillcolor('saddle brown')
g.begin_fill()
g.right(90)
g.forward(100)
g.left(90)
g.forward(50)
g.left(90)
g.forward(100)
g.hideturtle()
g.end_fill()
