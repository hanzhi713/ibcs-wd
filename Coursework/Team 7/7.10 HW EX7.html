def print_triangular_numbers(n):
      f = 0
      for i in range(1,n+1):
            f = f + i
            print(i ,'     ', f)
 
print_triangular_numbers(5)

def num_digits(n):
      f = str(n)
      if n >= 0:
            l = len(f)
      else:
            l = len(f) - 1
      print(l)

def num_even_digits(n):
      f = list(str(n))
      d = 0
      for i in f:
            if int(i) % 2 == 0:
                  d +=1
            else:
                  d +=0
      print(d)
