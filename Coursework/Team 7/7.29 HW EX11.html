      res = []
      for i in v:
            f = s * i
            res.append(f)
      return res

def dot_product(u, v):
      res = 0
      for (i,b) in enumerate(u):
            l = b * v[i]
            res = res + l
      return res
 
f = dot_product([1,2,1],[1,4,3])
print(f)

def cross_product(u, v):
      res = []
      a = u[1] * v[2] - u[2] * v[1]
      b = u[2] * v[0] - u[0] * v[2]
      c = u[0] * v[1] - u[1] * v[0]
      res.append(a)
      res.append(b)
      res.append(c)
      return res
 
f = cross_product([1,2,1],[1,4,3])
print(f)
