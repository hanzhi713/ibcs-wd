def CT(ounces):
      print('       Conversion')
      print('Ounces          Grams')
      for o in range(1,ounces+1):
            Grams = o * 28.35
            Grams = round(Grams)
            if len(str(o)) == 2:
                  print('   ',
                        o,'             ',Grams)
            else:
                  print('    ',o,'              ',Grams)
CT(16)

def CNO(num):
      NOB = num // 6
      if num % 6 != 0:
            NOB += 1
      return NOB
 
num = int(input('What is the number of eggs?'))
print(CNO(num))

def prpr(N):
      suM = 0
      for count in range(1, N//2 + 1):
            if N % count == 0:
                  suM = suM + count
      if suM == N:
            print('perfect')
