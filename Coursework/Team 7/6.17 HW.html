import turtle
import random
 
wn = turtle.Screen()
f = turtle.Turtle()
n= turtle.Turtle()
m=turtle.Turtle()
f.hideturtle()
n.hideturtle()
f.speed(0)
n.penup()
n.right(90)
for k in range(1,480,80)[::-1]:
               f.fillcolor('dark sea green3')
               f.penup()
               f.goto(0,-k)
               f.pendown()
               f.begin_fill()
               f.circle(k)
               f.end_fill()
               f.fillcolor('white')
               f.penup()
               f.left(90)
               f.forward(40)
               f.right(90)
               f.begin_fill()
               f.circle(k-40)
               f.end_fill()
for p in range(0,10):
       n.forward(40)
       n.write(10-p)
for r in range(7):
       x = random.uniform(-480,480)
       y = random.uniform(480,-480)
       m.penup()
       m.goto(x,y)
       m.pendown()
       m.stamp()
 
wn.mainloop()
