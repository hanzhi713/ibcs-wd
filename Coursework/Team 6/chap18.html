import turtle
import math
def draw_triangle(x,y,mal):
    t.penup()
    t.goto(x,y)
    t.pendown()
    t.setheading(300)
    for i in range(0,3):
        t.forward(mal)
        t.right(120)
 
def recur_tri(x,y,mal,order):
    if order == 0:
        draw_triangle(x,y,mal)
    else:
        recur_tri(x,y,mal/2,order-1)
        recur_tri(x-mal/4,y - math.sqrt(3)*mal/4,mal/2,order-1)
        recur_tri(x+mal/4,y - math.sqrt(3)*mal/4,mal/2,order-1)
 
wn = turtle.Screen()
t = turtle.Turtle()
t.speed(0)
t.penup()
t.goto(-150,0)
t.pendown()
recur_tri(0,0,300,5)
wn.mainloop()

def r_min(nxs):
    smallest = None
    first_time = True
    for e in nxs:
        if type(e) == type([]):
            val = r_min(e)
        else:
            val = e
 
        if first_time or val < smallest:
            smallest = val
            first_time = False
 
    return smallest

def count(target,nxs):
    counter = 0
    for e in nxs:
        if type(e) == type([]):
            counter += count(target,e)
        else:
            val = e
 
        if  val == target:
            counter += 1
    return counter

def flatten(nxs):
    new_list = []
    for e in nxs:
        if type(e) == list:
           new_list = new_list + flatten(e)
        else:
            new_list.append(e)
 
    return new_list

def it_fib(order):
    if order == 1 or order == 2:
        return 1
    first_num = 1
    second_num = 1
    for i in range(0, int((order-1)/2)):
        first_num = first_num + second_num
        second_num = first_num + second_num
    if order%2 == 1:
        return first_num
    else:
        return second_num
