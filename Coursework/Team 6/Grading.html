import java.util.Scanner;
import java.util.ArrayList;
import java.io.File;
import java.io.IOException;
 
public class Histo {
 
    public static void main(String[] args) {
        Scanner data = null;
        ArrayList<Integer> count;
        int id;
        int idx;
 
        try {
                data = new Scanner(new File("/Users/Terminator/Desktop/test2.txt"));
        }
        catch ( IOException e) {
            System.out.println("Sorry but I was unable to open your data file");
            e.printStackTrace();
            System.exit(0);
        }
        count = new ArrayList<Integer>(5);
        for (int i =0; i<5;i++) {
            count.add(i,0);
        }
 
 
        while(data.hasNextInt()) {
            idx = data.nextInt();
            id = 0;
            if(idx > 90 && idx <= 100){
                System.out.println("A");
                count.set(0,count.get(0)+1);
            }
            else if(idx > 80 && idx <= 90 ){
                System.out.println("B");
                count.set(1,count.get(1)+1);
            }
            else if(idx > 70 && idx <= 80 ){
                System.out.println("C");
                count.set(2,count.get(2)+1);
            }
            else if(idx > 60 && idx <= 70 ){
                System.out.println("D");
                count.set(3,count.get(3)+1);
            }
            else if(idx > 0 && idx <= 60 ){
                System.out.println("F");
                count.set(4,count.get(4)+1);
            }
        }
        for (int i = 0; i < count.size() ;i++) {
            if (i == 0){
                System.out.println("students get A:"+count.get(i));
            }
            else if(i == 1){
                System.out.println("students get B:"+count.get(i));
            }
            else if(i == 2){
                System.out.println("students get C:"+count.get(i));
            }
            else if(i == 3){
                System.out.println("students get D:"+count.get(i));
            }
            else if(i == 4){
                System.out.println("students get F:"+count.get(i));
            }
        }
 
 
    }
 
 
}
