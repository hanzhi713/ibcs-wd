import static java.lang.Math.*;
public class dieset{
    int[][] RESULTS = new int[100][6];
    int time1 = 0;
    int time2 = 0;
    int time3 = 0;
    int time4 = 0;
    int time5 = 0;
    int time6 = 0;
    double upl = 35.0;
 
    public dieset(){
        for(int i = 0; i < 100; i++){
            for(int k = 0; k < 6; k++){
                RESULTS[i][k] = (int) (random()*6+1);
            }
        }
    }
 
    public int[][] getall(){
        return RESULTS;
    }
 
    public int[] getperson(int x){
        return RESULTS[x];
    }
 
    public int getspecfic(int x,int y){
        return RESULTS[x][y];
    }
 
    public void changespecific(int x,int y, int val){
        RESULTS[x][y] = val;
    }
 
    public void changeoneperson(int x, int[]val){
        RESULTS[x] = val;
    }
 
    public void distribution(){
        for(int i = 0; i < 100; i++){
            for(int k = 0; k < 6; k++){
                if(RESULTS[i][k] == 1){
                    time1++;
                }
                else if(RESULTS[i][k] == 2){
                    time2++;
                }
                else if(RESULTS[i][k] == 3){
                    time3++;
                }
                else if(RESULTS[i][k] == 4){
                    time4++;
                }
                else if(RESULTS[i][k] == 5){
                    time5++;
                }
                else{
                    time6++;
                }
            }
        }
        System.out.println("times of die throwing 1"+"  "+time1);
        System.out.println("times of die throwing 2"+"  "+time2);
        System.out.println("times of die throwing 3"+"  "+time3);
        System.out.println("times of die throwing 4"+"  "+time4);
        System.out.println("times of die throwing 5"+"  "+time5);
        System.out.println("times of die throwing 6"+"  "+time6);
    }
 
    public boolean isbiased(){
        double mean = (time1 + time2 + time3 + time4 + time5 + time6)/6;
        double calsd = Math.sqrt(Math.pow(mean-time1,2)+Math.pow(mean-time2,2)+Math.pow(mean-time3,2)+Math.pow(mean-time4,2)+Math.pow(mean-time5,2)+Math.pow(mean-time6,2));
        if(calsd>upl){
            return true;
        }
        else{
            return false;
        }
    }
 
 
 
 
}

import static java.lang.Math.*;
public class puzzle{
 
    int[][] all;
    int[] num;
    int x;
    int n;
 
    public puzzle(){}
    public puzzle(int n){
        this.n = n;
 
        x = ((1+n*n)*n)/2;
        all = new int[n][n];
        num = new int[n*n];
        for(int i = 1; i < n*n+1; i++){
            num[i-1] = i;
        }
        boolean w;
        while(w = true){
        //create random board
        for(int i = 0; i < n; i++){
            for(int k = 0; k < n; k++){
                int ins = (int) (random()*(n*n));
                if(num[ins] == -1){
                    continue;
                }
                all[i][k] = num[ins];
                num[ins] = -1;
            }
        }
 
 
 
 
        // check
        for(int i = 1; i < n+1; i++){
            int tempvar = 0;
            for(int l = 1; l < n+1; l++){
                //for horizontal
                tempvar = tempvar + all[i][l];
                if(tempvar != x){
                    break;
                }
            }
            if(tempvar != x){
                break;
            }
 
            for(int l = 1; l < n+1; l++){
                //for vertical
                tempvar = tempvar + all[l][i];
                if(tempvar != x){
                    break;
                }
            }
            if(tempvar != x){
                break;
            }
 
            for(int l = 1; l < n+1; l++){
                //for diagonal
                tempvar = tempvar + all[l][l];
                if(tempvar != x){
                    break;
                }
            }
            if(tempvar != x){
                break;
            }
 
            for(int l = 1; l < n+1; l++){
                //for reverse horizontal
                tempvar = tempvar + all[l][n+1-l];
                if(tempvar != x){
                    break;
                }
            }
            if(tempvar != x){
                break;
            }
 
        }
 
        w = false;
        }
 
    }
 
    public void getpuzzle(){
        for(int i = 0; i < n; i++){
            for(int k = 0; k < n; k++){
                System.out.print(all[i][k]+"     ");
            }
            System.out.println();
        }
    }
 
 
}
