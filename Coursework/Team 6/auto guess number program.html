import random
rng = random.Random()
lowerrange = 1
upperrange = 1000
number = rng.randrange(1,1000)
guesses = 0
msg = ""
while True:
    guess = int((lowerrange + upperrange)/2) #int(input(msg + "\nGuess my number between 1 and 1000"))
    print(guess)
    guesses += 1
    if guess > number:
        msg = "your "+str(guesses)+" guess is too big"
        upperrange = guess
        print(msg)
    elif guess < number:
        msg = "your "+str(guesses)+" guess is too small"
        lowerrange = guess
        print(msg)
    else:
        break
print("you guessed it in", guesses)
