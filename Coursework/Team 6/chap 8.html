def organizer(strng):
    counter = 0
    ecounter = 0
    strng = strng.split()
    for i in strng:
        counter += 1
        if "e" in i:
            ecounter += 1
    s = "Your text contains {0} words, of which {1}({2:.1f}%) contain an 'e'".format(counter, ecounter, ecounter/counter*100)
    print(s)
organizer('''adasdawd aw eadaw da e a d aew w da wewea edwe''')
