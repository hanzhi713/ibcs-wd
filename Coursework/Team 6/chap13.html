def reverse(a,b):
    fileinput = open(a,'r')
    fileoutput = open(b,'w')
    line = fileinput.readlines()
    line.reverse()
    for i in line:
        fileoutput.write(i)
    fileinput.close()
    fileoutput.close()
 
reverse("test.txt","b.txt")

def clearsnake(a):
    fileinput = open(a,'r')
    line = fileinput.readlines()
    for i in line:
        if "snake" in i:
            continue
        else:
            print(i)
    fileinput.close()
 
clearsnake("test.txt")

def sequenceline(a,b):
    handle1 = open(a,'r')
    handle2 = open(b,'w')
    line = handle1.readlines()
    k = "1"
    for i in line:
        l = k.zfill(4) + " " + i
        handle2.write(l)
        k = int(k)
        k = k + 1
        k = str(k)
 
    handle1.close()
    handle2.close()
 
sequenceline("test.txt","b.txt")

def undosequence(a,b):
    handle1 = open(a,'r')
    handle2 = open(b,'w')
    line = handle1.readlines()
    for i in line:
        handle2.write(i[5:])
    handle1.close()
    handle2.close()
