infile = open("newfile1.txt", "r")
outfile = open("newfileNo2.txt", "w")
file1 = infile.read()
k = file1.split()
def reverse(a):
    b = a[::-1]
    return b
 
newlist = []
for i in k:
    p = reverse(i)
    newlist.append(p)
 
final = " ".join(newlist)
outfile.write(final)
