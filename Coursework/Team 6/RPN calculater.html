def postfix_eval(s):
    s = s.split(" ")
    print(s)
    n = len(s)
    stack = []
    for i in range(n):
        print(stack)
        if s[i].isdigit():
            stack.append(int(s[i]))
        elif s[i] == "+":
            a = stack.pop()
            b = stack.pop()
            stack.append(int(b)+int(a))
        elif s[i] == "-":
            a = stack.pop()
            b = stack.pop()
            stack.append(int(b)-int(a))
        elif s[i] == "*":
            a = stack.pop()
            b = stack.pop()
            stack.append(int(b)*int(a))
        elif s[i] == "/":
            a = stack.pop()
            b = stack.pop()
            stack.append(int(b)/int(a))
    return stack.pop()
 
print(postfix_eval("10 2 8 * + 3 -"))
def postfix_trans(s):
    s = s.split(" ")
    print(s)
    n = len(s)
    stack = []
 
    for i in range(n+1):
        print(stack)
        if ord(s[i]) >=97 and ord(s[i])<=122:
            stack.append(s[i])
        elif s[i] == "+":
            a = stack.pop()
            b = stack.pop()
            temp = "("+a+"+"+b+")"
            stack.append(temp)
 
        elif s[i] == "-":
            a = stack.pop()
            b = stack.pop()
            temp = "("+a+"-"+b+")"
            stack.append(temp)
 
 
        elif s[i] == "*":
            a = stack.pop()
            b = stack.pop()
            tempstr = a +"*"+ b
            stack.append(tempstr)
 
        elif s[i] == "/":
            a = stack.pop()
            b = stack.pop()
            tempstr = a +"/"+ b
            stack.append(tempstr)
 
 
 
print(postfix_trans("a b c * + d -"))
