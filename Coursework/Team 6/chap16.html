class Point:
    """ Point class represents and manipulates x,y coords. """
 
    def __init__(self, x, y):
        """ Create a new point at the origin """
        self.x = x
        self.y = y
 
    def setX(self, x):
        self.x = x
 
    def setY(self):
        self.y = y
 
    def getX(self):
        return self.x
 
    def getY(self):
        return self.y
 
    def midPoint(self, target):
        t = Point(0.5 * (self.x + target.x), 0.5 * (self.x + target.x))
        return t
 
    def reflect_x(self):  # NO.2
        copy = Point(self.x, self.y)
        return copy
 
    def slope_from_origin(self):  # NO.3
        i = round(self.y / self.x, 2)
        return i
 
    def get_line_to(self, point):  # NO.4
        slope = (self.y - point.y) / (self.x - point.x)
        y_inter = self.y - slope * self.x
        return (slope, y_inter)
 
    def __str__(self):
        return "({0},{1})".format(self.x, self.y)
 
 
class Rectangle:
    """ A class to manufacture rectangle objects """
 
    def __init__(self, posn, w, h):
        """ Initialize rectangle at posn, with width w, height h """
        self.corner = posn
        self.width = w
        self.height = h
 
    def __str__(self):
        return "({0}, {1}, {2})".format(self.corner, self.width, self.height)
 
    def grow(self, delta_width, delta_height):
        """ Grow (or shrink) this object by the deltas """
        self.width += delta_width
        self.height += delta_height
 
    def move(self, dx, dy):
        """ Move this object by the deltas """
        self.corner.x += dx
        self.corner.y += dy
 
    def same_coordinates(p1, p2):
        return (p1.x == p2.x) and (p1.y == p2.y)
 
    def area(self):
        i = self.width * self.height
        return i
 
    def perimeter(self):
        i = 2 * (self.width + self.height)
        return i
 
    def filp(self):
        w = self.width
        h = self.height
        self.width = h
        self.height = w
        return self
 
    def checkin(self, target):
        if self.corner.x - self.width / 2 < target.x < self.corner.x + self.width / 2 and self.corner.y - self.height / 2 < target.y < self.corner.y + self.height / 2:
            return True
        else:
            return False
    def check_rects(self, target):
        if self.corner.x + self.width / 2 < target.corner.x - target.width/2 or self.corner.x - self.width / 2 > target.corner.x + target.width/2:
            if self.corner.y + self.height/2 < target.corner.y - target.height/2 or self.corner.y - self.height/2 > target.corner.y + target.height/2:
                return True
            else:
                return False
        else:
            return False
