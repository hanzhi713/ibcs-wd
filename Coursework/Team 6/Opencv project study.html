import cv2 as cv
import random
from matplotlib import pyplot as plt
 
def salt(number_pixel,img):
    (height,width,k) = img.shape
    for k in range(number_pixel):
        i = random.randint(0,height-1)
        j = random.randint(0,width-1)
        img[i,j] = (255,255,255)
 
def pepper(number_pixel,img):
    (height,width,k) = img.shape
    for k in range(number_pixel):
        i = random.randint(0,height-1)
        j = random.randint(0,width-1)
        img[i,j] = (0,0,0)
 
im = cv.imread("testimg.jpg")
salt(5000,im)
pepper(5000,im)
cv.imwrite("testok.jpg",im)
plt.imshow(cv.cvtColor(im, cv.COLOR_BGR2RGB))
plt.show()
