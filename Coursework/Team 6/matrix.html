public class TempConvGUI {
 
    public static void main(String[] args) {
        int counter = 0;
        int state = 0;
        for(int i=0;i<=10;i++){
            if(i == 0){
                continue;
            }
            for(int k=0;k<=10;k++){
                if(i==k){
                    continue;
                }
                if(state == 0){
                    counter += 1;
                    if(counter==i){
                        System.out.printf("%3s",counter);
                        state = 1;
                        continue;
                    }
                    System.out.printf("%3s",0);
                }
                if(state == 1){
                    System.out.printf("%3s",-1);
                }
            }
            counter = 0;
            state =0;
            System.out.printf("\n");
        }
    }
 
}
