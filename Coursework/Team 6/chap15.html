import math
class Point:
    """ Point class represents and manipulates x,y coords. """
 
    def __init__(self,x,y):
        """ Create a new point at the origin """
        self.x = x
        self.y = y
 
    def setX(self,x):
        self.x = x
 
    def setY(self):
        self.y = y
 
    def getX(self):
        return self.x
    def getY(self):
        return self.y
    def midPoint(self,target):
        t = Point(0.5*(self.x + target.x),0.5*(self.x + target.x))
        return t
    def reflect_x(self):# NO.2
        copy = Point(self.x,self.y)
        return copy
    def slope_from_origin(self):#NO.3
        i = round(self.y/self.x,2)
        return i
    def get_line_to(self,point):#NO.4
        slope = (self.y - point.y)/(self.x - point.x)
        y_inter = self.y - slope*self.x
        return (slope,y_inter)
 
def intercept(a,b):
    x = (b[1] - a[1])/a[0] - b[0])
    y = x*a + b
    i = Point(x,y)
    return i
 
def find_center(a,b,c,d):
    slope1 = -1/(a.get_line_to(b)[0])
    t1 = a.midPoint(b)
    y_inter1 = t1.y - slope1*t1.x
    slope2 = -1/(c.get_line_to(d)[0])
    t2 = c.midPoint(d)
    y_inter2 = t2.y - slope2*t2.x
    print(intercept((slope1,y_inter1),(slope2,y_inter2)))
 
def distance(x,y):      #NO.1
    result = math.sqrt((x.x - y.x)**2+(x.y - y.y)**2)
    return round(result,3)

class SMS_store:
    """ Point class represents and manipulates x,y coords. """
 
    def __init__(self):
        self.new = []
        self.old = []
 
    def add_new_arrival(self, from_number, time_arrived, text_of_SMS):
        has_been_viewed = False
        i = (has_been_viewed, from_number, time_arrived, text_of_SMS)
        self.new.append(i)
 
    def message_count(self):
        return len(self.old) + len(self.new)
 
    def get_unread_indexes(self):
        return self.new
 
    def get_message(self, i):
        if self.new[i] == None:
            print("no element found")
        print(self.new[i][1:4])
        has_been_viewed = True
        new_element = (has_been_viewed, self.new[i][1],self.new[i][2],self.new[i][3])
        self.old.append(new_element)
        return
 
    def delete(self, i):
        del self.old[i]
        return self.old
