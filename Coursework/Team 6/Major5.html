package bintree;
 
public class Node {
    private int key;
    private int cargo;
    private int leftpointer;
    private int rightpointer;
    Node(int k,int a, int l, int r){
        key = k;
        cargo = a;
        leftpointer = l;
        rightpointer = r;
    }
    public int getcargo(){
        return cargo;
    }
    public int getkey(){
        return key;
    }
    public int getleft(){
        return leftpointer;
    }
    public int getright(){
        return leftpointer;
    }
 
    public void setkey(int k){
        key = k;
    }
    public void setleftpointer(int k){
        leftpointer = k;
    }
    public void setrightpointer(int k){
        rightpointer = k;
    }
}
