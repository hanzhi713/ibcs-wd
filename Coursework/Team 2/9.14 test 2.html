def is_sorted(s):
    c = True
    for i in range(len(s) - 1):
        j = s[i]
        k = s[i + 1]
        if k < j:
            c = False
    return c
print(is_sorted(["b", "c"]))

def is_anagram(a, b):
    x = []
    y = []
    for i in a:
        x.append(i)
    for i in b:
        y.append(i)
    x.sort()
    y.sort()
    if x == y:
        return True
    else:
        return False

    for j in a:
        c = 0
        for i in a:
            if j == i:

        if c >= 2:

    c = []
    for i in a:
        if i not in c:
            c.append(i)
    return c
