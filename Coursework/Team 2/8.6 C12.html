import string
def cleanword(s):
    l = []
    for i in s:
        if i not in string.punctuation:
            l.append(i)
    k = ''.join(l)
    return k
 
def has_dashdash(s):
    c = False
    for i in range(len(s)):
        if s[i] == '-':
            if i + 1 == len(s):
                break
            if s[i+1] == '-':
                c = True
                break
    return c
 
def extract_words(s):
    for c in string.punctuation:
        s = s.replace(c,' ')
    l = s.lower()
    k = l.split()
    return k
 
def wordcount(c, s):
    counter = 0
    for i in s:
        if c == i:
            counter = counter + 1
    return counter
 
def wordset(s):
    l = []
    for i in s:
        if i not in l:
            l.append(i)
    return l
 
def longestword(s):
    c = len(s)
    if c == 0:
        return 0
    for i in range(0, c):
        if i <= c - 2:
            if len(s[i]) > len(s[i + 1]):
                a = s[i]
                s[i] = s[i + 1]
                s[i + 1] = a
        else:
            break
    l = len(s[c - 1])
    return l
