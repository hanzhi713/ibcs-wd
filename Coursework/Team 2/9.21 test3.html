vocablist = open("vocab.txt", "r")
xs = vocablist.readlines()
vocablist.close()
 
xs2 = []
for w in xs:
    new = []
    c = len(w)
    for j in range(c):
        new.append(w[c - 1 - j])
    a = ''.join(new)
    xs2.append(a)
 
g = open("textfile_vocab_Alex.txt", "w")
for v in xs2:
    g.write(v)
g.close()
