import turtle
wn = turtle.Screen()
alex = turtle.Turtle()
side_l = int(input("length:"))
side_n = int(input("numbers:"))
 
def draw_polygon(x, y):
    for i in range(1, x+1):
        alex.forward(y)
        alex.left(180-180*(x-2)/x)
 
draw_polygon(side_n, side_l)
wn.mainloop()
