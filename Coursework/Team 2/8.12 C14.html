    result = []
    xi = 0
    yi = 0
 
    while True:
        if xi >= len(vocab):
            return result
 
        if yi >= len(wds):
            return result
 
        if vocab[xi] == wds[yi]:
            yi += 1
            result.append(vocab[xi])
 
        elif vocab[xi] < wds[yi]:
            xi += 1
 
        else:
            yi += 1

def share_diagonal(x0, y0, x1, y1):
    """ Is (x0, y0) on a shared diagonal with (x1, y1)? """
    dy = abs(y1 - y0)        # Calc the absolute y distance
    dx = abs(x1 - x0)        # CXalc the absolute x distance
    return dx == dy          # They clash if dx == dy
 
def col_clashes(bs, c):
    """ Return True if the queen at column c clashes
         with any queen to its left.
    """
    for i in range(c):     # Look at all columns to the left of c
          if share_diagonal(i, bs[i], c, bs[c]):
              return True
 
    return False           # No clashes - col c has a safe placement.
 
def has_clashes(the_board):
    """ Determine whether we have any queens clashing on the diagonals.
        We're assuming here that the_board is a permutation of column
        numbers, so we're not explicitly checking row or column clashes.
    """
    for col in range(1,len(the_board)):
        if col_clashes(the_board, col):
            return True
    return False

def mirrory(s, a, b):
    lens = len(s)
    c = []
    for i in range(0, lens// b):
        c.append(s[:lens // a])
        del s[:lens // a]
    for i in range(0, len(c)):
        c[i].reverse()
    return c
 
def mirrorx(s, a, b):
    lens = len(s)
    c = []
    for i in range(0, lens // b):
        c.append(s[:lens // a])
        del s[:lens // a]
    for i in range(0, len(c)):
        for j in range(0, lens // a):
            c[i][j] = 10 - c[i][j]
    return c
 
def rotate(s, degrees, a, b):
    lens = len(s)
    c = []
    for i in range(0, lens // 10):
        c.append(s[:lens // 3])
        del s[:lens // 3]
 
    for i in range(0, len(c)):
        for j in range(0, lens // a):
            if degrees == 90:
                c[i][j] = (-c[i][j], j)
            elif degrees == 180:
                c[i][j] = (-j, -c[i][j])
            elif degrees == 270:
                c[i][j] = (c[i][j], -j)
    return c

def prime(s):
    c = True
    if s == 2:
        c = True
    elif s == 1:
        c = False
    else:
        for i in range(2, s):
            if s % i == 0:
                c = False
    return c
def lotto_draw():
    draw = []
    rng = random.Random()
    while len(draw) < 6:
        guess = rng.randrange(1, 49)
        if prime(guess) == False:
            continue
        elif guess not in draw:
            draw.append(guess)
    return draw

def lotto_match(a, b):
    count = 0
    lena = len(a)
    lenb = len(b)
    for i in range(lenb):
        for j in range(lena):
            if a[j] == b[i]:
                count += 1
    return count

def prime_misses(tickets):
    primes = []
    primes.append(2)
    misses = []
    for i in range(1, 49):
        c = False
        if i == 1 or i == 2:
            continue
        else:
            if prime(i):
                primes.append(i)
    lent = len(tickets)
    for j in primes:
        c = False
        for k in range(lent):
            if j in tickets[k]:
                c = True
        if c == False:
            misses.append(j)
    return misses
