import string
old = open('newZen.txt', 'r')
new = open('newerZen.txt', 'w')
while 1:
    text = old.readline()
    if len(text) == 0:
        break
    l = text[:]
    k = l.split()
    i = 0
    while 1:
        if k[i] in string.digits:
            k[i] = ''
        elif k[i] == ' ':
            k[i] = ''
        else:
            break
        i = i + 1
    txt = ' '.join(k)
    new.write(txt)
    new.write('\n')
old.close()
new.close()
The file that will be needed when running the code of question 4 is the same as the one generated by the code of question 3
