import java.util.*;
public class ThePuzzleGame {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        System.out.print("Insert an Integer: ");
        int n = sc.nextInt();
        int[][] x = magicSquare(n);
        for(int i = 0; i < n; i ++){
            for(int j = 0; j < n; j ++){
                System.out.print(x[i][j]);
                System.out.print("   ");
            }
            System.out.println();
        }
    }
 
    public static int[][] magicSquare(int n){
        int[][] magicsquare = new int[n][];
        for(int i = 0; i < n; i ++){
            int[] temp = new int[n];
            magicsquare[i] = temp;
        }
        int x = n / 2;
        int y = 0;
        magicsquare[y][x] = 1;
        for(int i = 2; i < n * n + 1; i ++){
            int tmpx = x;
            int tmpy = y;
            x = x + 1;
            y = y - 1;
            if(x > n - 1){
                x = 0;
            }
            if(y < 0){
                y = n - 1;
            }
            if(magicsquare[y][x] != 0){
                if(tmpx == n - 1){
                    x = tmpx;
                }
                if(tmpy == 0){
                    y = tmpy;
                }
                x = tmpx;
                y = tmpy + 1;
                magicsquare[y][x] = i;
            }
            else{
                magicsquare[y][x] = i;
            }
        }
        return magicsquare;
    }
}
