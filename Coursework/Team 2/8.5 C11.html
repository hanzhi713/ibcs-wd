def dot_product(u, v):
    sum = 0
    for (index, value) in enumerate(u):
        u[index] = value * v[index]
    for i in u:
        sum = sum +  i
    return sum
 
print(dot_product([1, 2, 1], [1, 4, 3]))

def cross_product(u,v):
    product = []
    product.append(u[1] * v[2] - u[2] * v[1])
    product.append(u[2] * v[0] - u[0] * v[2])
    product.append(u[0] * v[1] - u[1] * v[0])
    return product
print(cross_product([1, 3, 6],[2, 4, 9]))
