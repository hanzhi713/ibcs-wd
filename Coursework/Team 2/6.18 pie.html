import turtle
import math
wn = turtle.Screen()
alex = turtle.Turtle()
alex.speed(10)
def draw (length, number):
    for i in range(3, number + 1):
        a = 360 / i
        c = math.cos(math.radians(90 - a / 2))
        b = length * c * 2
        alex.penup()
        alex.goto(0, 0)
        alex.right(90 + a / 2)
        alex.forward(length)
        alex.left(90 + a / 2)
        alex.pendown()
        alex.forward(b)
        for j in range(1, i):
            alex.left(a)
            alex.forward(b)
        alex.left(a)
        print(b * i * 0.5 / length)
 
radius = int(input("How long is the radius:"))
much = int(input("How many:"))
draw(radius, much)
 
wn.mainloop()
