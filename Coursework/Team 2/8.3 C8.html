def count(sub, word):
    sublen = len(sub)
    wordlen = len(word)
    sec = ''
    count = 0
    for i in range(0, wordlen - sublen + 1):
        for j in range(i, i + sublen):
            sec = sec + word[j]
        if sub == sec:
            count = count + 1
        sec = ''
    return count
