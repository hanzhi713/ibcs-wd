1.二分法
import time
import random
t0 = time.clock()
ran = random.Random()
guess = ran.randrange(0, 10000)
print(guess, '\n')
a = 5000
up = 10000
down = 0
print(5000)
while 1:
    if a < guess:
        down = a
        a = int((a + up) / 2)
        print(a)
    elif a > guess:
        up = a
        a = int((a + down) / 2)
        print(a)
    else:
        break
print('done')
t1 = time.clock()
print(t1 - t0, 's')
2.乘法表

def print_oneline(i):
    for j in range(1, i + 1):
        print(" %d * %d = %d" %(j, i, i * j), end = '\t')
 
def print_multiplicationTable(n):
    for i in range(1, n + 1):
        print_oneline(i)
        print()
print_multiplicationTable(10)
