import turtle
wn = turtle.Screen()
alex = turtle.Turtle()
alex.speed(10)
def koch(t, order, size):
   if order == 0:
         t.forward(size)
   else:
        for angle in [90, -90, -90, 90, 0]:
            koch(t, order-1, size/3)
            t.left(angle)
koch(alex, 4, 400)
wn.mainloop()
