public class BankAccountA {

        BankAccount b = new BankAccount("BrianL", 10000);
        BankAccount c = new BankAccount("KevinC", 800.00);
        BankAccount better;
        better = chooseBetterone(b, c);

        System.out.println(better.getBalance());
        c.deposit("KevinC", 300000);
        better = chooseBetterone(b,c);
        System.out.println(better.getBalance());
    }
    public static BankAccount chooseBetterone(BankAccount b1, BankAccount b2){
        if (b1.getBalance() >= b2.getBalance()){
            return b1;
        }
        else{
            return b2;
        }
    }
}
class BankAccount{
    private String password;
    private double balance;
    public static final double OVERDRAWN_PENALTY = 20.0;
    public BankAccount(){
        password = "";
        balance = 0.0;
 
    }
    public BankAccount(String acctPassword, double acctBalance){
        password = acctPassword;
        balance = acctBalance;
    }
    public double getBalance(){
        return balance;
    }
    public void deposit(String acctPassword, double amount){
        if (!acctPassword.equals(password)){
            System.out.println("Access Denied");
        }
        else{
            balance += amount;
        }
    }
    public void withdraw(String acctPassword, double amount){
        if(!acctPassword.equals(password)){
            System.out.println("Access Denied");
        }
        else{
            balance -=amount;
            if (balance < 0){
                balance -= OVERDRAWN_PENALTY;
            }
        }
    }
 
}

public class StudentA {
    public static void main(String []args){
        int []atest = {23, 27, 90};
        Students a = new Students("A", atest, "pass");
        a.computeGrade();
        System.out.println(a.getGrade());
        UnderGrad b = new UnderGrad("B", atest, "pass");
        b.computeGrade();
        System.out.println(b.getGrade());
 
    }
}
class Students{
    private String name;
    public final int NUM_TESTS = 3;
    private int[] tests;
    private String grade;
    public Students(){
        name = "";
        tests = new int[NUM_TESTS];
        grade = "";
 
    }
    public Students(String studName, int[] studTests, String studGrade){
        name = studName;
        tests = studTests;
        grade = studGrade;
    }
    public String getName(){
        return name;
    }
    public String getGrade(){
        return grade;
    }
    public void setGrade(String newGrade){
        grade = newGrade;
    }
    public void computeGrade(){
        if (name.equals("")){
            grade = "No grade";
        }
        else if (getTestAverage() >= 65){
            grade = "Pass";
        }
        else{
            grade = "Fail";
        }
    }
    public double getTestAverage(){
        double total = 0;
        for (int score : tests){
            total += score;
        }
        return total/NUM_TESTS;
        }
    }
class UnderGrad extends Students{
    public UnderGrad(){
        super();
    }
    public UnderGrad(String studName, int[] studTests, String studGrade){
        super(studName, studTests, studGrade);
    }
    public void computeGrade(){
        if (getTestAverage() >= 70){
            setGrade("Pass");
        }
        else{
            setGrade("Fail");
        }
    }
}

public class TemperatureT {
    public static void main(String []args){
        Temperature c = new Temperature(0, "C");
        c.toFahrenheit();
        c.raise(20);
        System.out.println(c.scale + c.degrees);
    }
}
class Temperature{
    public String scale;
    public double degrees;
    public Temperature(double temp, String scale2){
        scale = scale2;
        degrees = temp;
    }
    public void toFahrenheit(){
        degrees = degrees * 1.8 + 32;
        scale = "F";
    }
    public void raise(double amt){
        degrees += amt;
    }
 
}
