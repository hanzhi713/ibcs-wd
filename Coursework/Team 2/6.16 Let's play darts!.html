import turtle
import random
 
wn = turtle.Screen()
alex = turtle.Turtle()
 
alex.penup()
alex.goto(0,45)
alex.pendown()
alex.pencolor("white")
for i in range(1, 7):
    if(i%2==1):
        alex.fillcolor("lightgreen")
    else: alex.fillcolor("lightblue")
    alex.begin_fill()
    alex.circle(150-20*i)
    alex.end_fill()
    alex.penup()
    alex.goto(0,45+20*i)
    alex.pendown()
 
emile = turtle.Turtle()
emile.color("pink")
l = int(input("Left x:"))
r = int(input("right x:"))
u = int(input("up y:"))
d = int(input("dowb y:"))
 
 
for n in range(1, 10):
    ax = random.uniform(l, r)
    ay = random.uniform(u, r)
    emile.penup()
    emile.goto(ax, ay)
    emile.pendown()
    emile.stamp()
 
wn.mainloop()
