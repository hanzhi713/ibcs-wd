import random
import matplotlib.pyplot as plt
import time
 
 
def insertionSort(alist):
    N = len(alist)
    for i in range(1, N):
        CurValue = alist[i]
        j = i
        while j > 0 and CurValue < alist[j - 1]:
            alist[j] = alist[j - 1]
            j = j - 1
        alist[j] = CurValue
    return None
 
 
def selectionSort(alist):
    N = len(alist)
    for i in range(0, N - 1):
        minSub = i
        for j in range(i + 1, N):
            if alist[j] < alist[minSub]:
                minSub = j
        temp = alist[i]
        alist[i] = alist[minSub]
        alist[minSub] = temp
 
 
def bubbleSort(alist):
    N = len(alist)
    for i in range(0, N - 2):
        for j in range(0, N - i - 1):
            if alist[j] > alist[j + 1]:
                temp = alist[j]
                alist[j] = alist[j + 1]
                alist[j + 1] = temp
    return None

def mergeSort(alist):
    N = len(alist)
    if N == 2:
        leftrow = alist[: 1]
        rightrow = alist[1:]
        return merge(leftrow, rightrow)
    elif N == 1:
        return alist
 
    else:
        mid = N//2
        leftrow = alist[:mid]
        newleft = mergeSort(leftrow)
        rightrow = alist[mid:]
        newright = mergeSort(rightrow)
        return merge(newleft, newright)

## We compare the running time of the 3 basic algorithms.
 
def CompRunningTimeWithPlot():
    rng = random.Random()
    itemNum = 10
    base = 2
    timeSeqInsertion = []
    timeSeqSelection = []
    timeSeqBubble = []
 
    for i in range(itemNum // 2, itemNum):
        alist = list(range(base ** i))
        rng.shuffle(alist)
        t0 = time.clock()
        insertionSort(alist)
        t1 = time.clock()
        print("The list has {} items, it takes {} seconds by insertion sort.".format(base ** i, t1 - t0))
        timeSeqInsertion.append(t1 - t0)
 
        rng.shuffle(alist)
        t0 = time.clock()
        selectionSort(alist)
        t1 = time.clock()
        print("The list has {} items, it takes {} seconds by Selection sort.".format(base ** i, t1 - t0))
        timeSeqSelection.append(t1 - t0)
 
        rng.shuffle(alist)
        t0 = time.clock()
        bubbleSort(alist)
        t1 = time.clock()
        print("The list has {} items, it takes {} seconds by Bubble sort.".format(base ** i, t1 - t0))
        timeSeqBubble.append(t1 - t0)
 
    plt.figure()
    plt.hold(True)
    plt.plot(list(range(itemNum // 2, itemNum)), timeSeqInsertion, 'r')
    plt.plot(list(range(itemNum // 2, itemNum)), timeSeqSelection, 'g')
    plt.plot(list(range(itemNum // 2, itemNum)), timeSeqBubble, 'b')
    plt.show()
