package battleship;
 
public abstract class Battleship implements Grids{
    private int shipSize;
    Battleship(int a){
        shipSize = a;
    }
    public void position(int x, int y){
        if(grid[y][x] == 0){
            grid[y][x] = 1;
    }
        else{
            System.out.println("This point is occupied");
        }
}
    public void position(int x, int y, String pos){
        boolean flag = true;
        if(pos.equals("R")){
            if(7 - x < shipSize - 1){
                flag = false;
                System.out.println("Please choose another direction");
            }
            else{
                for(int i = x + 1; i < shipSize + x; i ++){
                    if(grid[y][i] != 0){
                        flag = false;
                        System.out.println("not allowed");
                        break;
                    }
                }
                if(flag){
                    for(int i = x + 1; i < shipSize + x; i ++){
                        grid[y][i] = 1;
                    }
                }
            }
        }
        else if (pos.equals("L")){
            if(x < shipSize - 1){
                flag = false;
                System.out.println("Please choose another direction");
            }
            else{
                for(int i = x - 1; i > -1; i --){
                    if(grid[y][i] != 0){
                        flag = false;
                        System.out.println("One of the point is occupied");
                        break;
                    }
                }
                if(flag){
                    for(int i = x - 1; i > -1; i --){
                        grid[y][i] = 1;
                    }
                }
            }
 
        }
        else if(pos.equals("U")){
            if(y < shipSize - 1){
                flag = false;
                System.out.println("Please choose another direction");
            }
            else{
                for(int i = y - 1; i > -1; i --){
                    if(grid[i][x] != 0){
                        flag = false;
                        System.out.println("One of the point is occupied");
                        break;
                    }
                }
                if(flag){
                    for(int i = y - 1; i > -1; i --){
                        grid[i][x] = 1;
                    }
                }
            }
 
        }
        else if(pos.equals("D")){
            if(7 - y < shipSize - 1){
                flag = false;
                System.out.println("Please choose another direction");
            }
            else{
                for(int i = y + 1; i < shipSize + y; i ++){
                    if(grid[i][x] != 0){
                        flag = false;
                        System.out.println("One of the point is occupied");
                        break;
                    }
                }
                if(flag){
                    for(int i = y + 1; i < shipSize + y; i ++){
                        grid[i][x] = 1;
                    }
                }
            }
        }
        else if(pos.equals("LU")){
            if(x < shipSize - 1 || y < shipSize - 1){
                flag = false;
                System.out.println("Please choose another direction");
            }
            else{
                for(int i = y - 1, j = x - 1;  i > -1 && j > -1; i --, j--){
                    if(grid[i][j] != 0){
                        flag = false;
                        System.out.println("One of the point is occupied");
                        break;
                    }
                }
                if(flag){
                    for(int i = y - 1, j = x - 1;  i > -1 && j > -1; i --, j--){
                        grid[i][j] = 1;
                    }
                }
            }
 
        }
        if(flag == true){
            position(x, y);
        }
    }
}
