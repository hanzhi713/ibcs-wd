public class Tree {
    static Node c = new Node("c");
    static Node d = new Node("d");
    static Node b = new Node("b");
    static Node a = new Node("a");
    static Node f = new Node("f");
    static Node e = new Node("e");
    static Node q = new Node("q");
 
    public static void main(String[] args){
        c.setLeft(null);
        c.setRight(null);
        d.setLeft(null);
        d.setRight(null);
        b.setLeft(c);
        b.setRight(null);
        a.setLeft(d);
        a.setRight(b);
        f.setLeft(null);
        f.setRight(null);
        e.setLeft(a);
        e.setRight(f);
        q.setLeft(e);
        q.setRight(null);
        printTree(q);
        System.out.println();
        delete(c);
        printTree(q);
    }
    public static Node findParent(Node a, Node target){
        if(a.getLeft() == null && a.getRight() == null)
            return null;
        else if(a.getLeft() == target || a.getRight() == target)
            return a;
        else{
            Node gauche = null;
            Node droit = null;
            if(a.getLeft() != null)
                gauche = findParent(a.getLeft(), target);
            if(a.getRight() != null)
                droit = findParent(a.getRight(), target);
            if(gauche != null)
                return gauche;
            else if(droit != null)
                return droit;
            else
                return null;
        }
    }
    public static void delete(Node target){
        if(target.getLeft() == null && target.getRight() == null){
            Node parent = findParent(q, target);
            if(parent.getLeft() == target)
                parent.setLeft(null);
            else if(parent.getRight() == target)
                parent.setRight(null);
        }
        else{
            Node largest = target.largest();
            Node parent = findParent(q, largest);
            target.setCargo(largest.getCargo());
            if(largest.getLeft() == null || largest.getRight() == null)
                if(parent.getLeft() == largest)
                    parent.setLeft(largest.getLeft());
                else if(parent.getRight() == largest)
                    parent.setRight(largest.getLeft());
        }
    }
    public static void printTree(Node start){
        if(start.getLeft() == null && start.getRight() == null){
            System.out.print(start.getCargo() + "     ");
        }
        else{
            if(start.getLeft() != null)
                printTree(start.getLeft());
            System.out.print(start.getCargo() + "     ");
            if(start.getRight() != null)
                printTree(start.getRight());
    }
    }
}
class Node{
    private String cargo;
    private Node left;
    private Node right;
    public Node(String value){
        cargo = value;
    }
    public void setCargo(String chargement){
        cargo = chargement;
    }
    public void setLeft(Node gauche){
        left = gauche;
    }
    public void setRight(Node droit){
        right = droit;
    }
    public Node getLeft(){
        return (Node) left;
    }
    public Node getRight(){
        return (Node) right;
    }
    public String getCargo(){
        return cargo;
    }
    public void print(){
        System.out.println(cargo);
        System.out.println(left.getCargo());
        System.out.println("        " + ((Node) right).getCargo());
    }
    public Node largest(){
        if(left == null)
            return null;
        else
            return findLargest(left);
    }
    private Node findLargest(Node a){
        if(a == null)
            return null;
        if(a.getRight() == null)
            return a;
        else{
            return findLargest(a.getRight());
        }
    }
