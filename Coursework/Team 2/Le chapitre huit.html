public class Sort {
    public static void main(String[] args){
        int[] a = {4, 1, 2, 7, 5, -1, 8, 0, 6};
        int[] b = {4, 6, 2, 7, 9, 1, 5, 7};
        int[] c = {2, 6, 7, 8};
        int[] d = {3, 5, 8, 9, 12};
        c = merge(c, d);
        quick_sort(b, 0, 7);
        for(int i : b){
            System.out.print(i);
            System.out.print(" ");
        }

    public static void selection_sort(int[] a){
        int len = a.length;
        for(int i = 0; i < len - 1; i ++){
            int minPos = i;
            for(int j = i + 1; j < len; j ++){
                if(a[j] < a[minPos]){
                    minPos = j;
                }
            }
            int tmp = a[i];
            a[i] = a[minPos];
            a[minPos] = tmp;
 
        }
    }
    public static void insertion_sort(int[] a){
        int len = a.length;
        int pos;
        for(int i = 0; i < len - 1; i ++){
            int tmp = a[i + 1];
            for(int j = i; j >= 0; j --){
                if(tmp < a[j]){
                    a[j + 1] = a[j];
                    pos = j;
                }
                else {
                    break;
                }
                a[pos] = tmp;
            }
        }
    }
    public static int[] merge(int[] a, int[] b){
        int lena = a.length;
        int lenb = b.length;
        int[] merged = new int[lena + lenb];
        int lenmerged = merged.length;
        int xi = 0;
        int yi = 0;
        int pointer = -1;
        while (true){
            if (xi >= lena){
                int tmp = yi;
                for(int i = xi + yi; i < lenmerged; i ++){
                    merged[i] = b[tmp];
                    tmp ++;
                }
                return merged;
            }
            if(yi >= lenb){
                int tmp = xi;
                for(int i = xi + yi; i < lenmerged; i ++){
                    merged[i] = a[tmp];
                    tmp ++;
                }
                return merged;
            }
            if (a[xi] > b[yi]){
                pointer ++;
                merged[pointer] = b[yi];
                yi++;
            }
            else{
                pointer++;
                merged[pointer] = a[xi];
                xi ++;
            }
        }
    }
    public static int[] merge_sort(int[] a){
        int len = a.length;
        if(len == 1){
            return a;
        }
        else{
            int mid = len/2;
            int[] leftrow = new int[mid];
            int[] rightrow = new int[len - mid];
            int pointer = 0;
            for(int i = 0; i < leftrow.length; i ++){
                leftrow[i] = a[pointer];
                pointer ++;
            }
            for(int i = 0; i < rightrow.length; i ++){
                rightrow[i] = a[pointer];
                pointer ++;
            }
            leftrow = merge_sort(leftrow);
            rightrow = merge_sort(rightrow);
            return merge(leftrow, rightrow);
        }
    }
    public static void quick_sort(int[] a, int low, int high){
        int i = low;
        int j = high;
        if(i < j){
              int pivot = a[i];
              while( i < j){
                  while( i < j && (a[j] >= pivot)){
                      j -= 1;
                  }
                   while (i < j && a[j] < pivot){
                      a[i] = a[j];
                      a[j] = a[i + 1];
                      i += 1;
                        }
                      a[i] = pivot;
        quick_sort(a, low, i);
        quick_sort(a, i + 1, high);
        }
        }
 
 
    }

public class Search {
    public static void main(String [] args){
        int [] a = {3, 4, 5, 6, 9, 20};
        System.out.println(binarysearch(a, 15));
        System.out.println(binarysearch(a, 20));
    }
 
public static int binarysearch(int[] list, int target){
    int len = list.length;
    int up = len;
    int down = 0;
    int mid = 0;
    while(true){
        if(up == down){
            return -1;
        }
        mid = (up + down) / 2;
        if(list[mid] < target){
            down = mid + 1;
        }
        else if(list[mid] > target){
            up = mid;
        }
        else{
            return 1;
        }
 
    }
}
