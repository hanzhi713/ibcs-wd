import java.util.*;
public class TicTacToe {
    static Scanner sc = new Scanner(System.in);
    public static void main(String[] args){
        System.out.println("1 or 2?");
        int justification = sc.nextInt();
        String[][] grid = new String[3][];
        for(int i = 0; i < 3; i ++){
            String [] temp = new String[3];
            for(int j = 0; j < 3; j ++){
                temp[j] = "n";
        }
            grid[i] = temp;
        }
        System.out.println("Enter Coordinate");
        int[][] minorgrid = new int[3][3];
        int x = sc.nextInt();
        int y = sc.nextInt();
        grid[y][x] = "o";
        minorgrid[y][x] = 1;
        if(justification == 1){
        if(x < 2){
        grid = solo(x + 1, y, grid, minorgrid);
        }
 
        else{
            grid = solo(x - 1, y, grid, minorgrid);
        }
 
        for(int i = 0; i < 3; i++){
            for(int j = 0; j < 3; j ++){
                System.out.print(grid[i][j]);
                System.out.print(" ");
            }
            System.out.println();
        }
        }
 
        else{
            for(int i = 0; i < 3; i++){
                for(int j = 0; j < 3; j ++){
                    System.out.print(grid[i][j]);
                    System.out.print(" ");
                }
                System.out.println();
            }
            grid = duo(grid, minorgrid);
        }
        if(evaluate(minorgrid) == 1){
            System.out.println("o wins");
        }
        else if(evaluate(minorgrid) == -1){
            System.out.println("x wins");
        }
        else{
            System.out.println("tie");
        }
    }
 
 
    public static String[][] solo(int x, int y, String[][] grid, int[][] minorgrid){
        boolean flag = true;
        for(int i = 0; i < 3; i ++){
            for(int j = 0; j < 3; j ++){
                if(grid[i][j].compareTo("n") == 0){
                flag = false;
            }
            }
        }
        if(evaluate(minorgrid) == 1){
            return grid;
        }
        if(x >= 0 && x < 3 && y >= 0 && y < 3){
            if(!flag){
            if(grid[y][x].compareTo("n") == 0){
                grid[y][x] = "x";
                minorgrid[y][x] = -1;
                if(evaluate(minorgrid) == -1){
                    return grid;
                }
                for(int i = 0; i < 3; i++){
                    for(int j = 0; j < 3; j ++){
                        System.out.print(minorgrid[i][j]);
                        System.out.print(" ");
                    }
                    System.out.println();
                }
                for(int i = 0; i < 3; i++){
                    for(int j = 0; j < 3; j ++){
                        System.out.print(grid[i][j]);
                        System.out.print(" ");
                    }
                    System.out.println();
                }
                System.out.println("Enter Coordinate");
                x = sc.nextInt();
                y = sc.nextInt();
                grid[y][x] = "o";
                minorgrid[y][x] = 1;
                solo(x, y + 1, grid, minorgrid);
                solo(x - 1, y, grid, minorgrid);
                solo(x + 1, y, grid, minorgrid);
                solo(x, y - 1, grid, minorgrid);
            }
            }
        }
        return grid;
    }
    public static String[][] duo(String[][] grid, int[][] minorgrid){
        boolean flag = true;
        for(int i = 0; i < 3; i ++){
            for(int j = 0; j < 3; j ++){
                if(grid[i][j].compareTo("n") == 0){
                flag = false;
            }
            }
        }
        while(!flag && evaluate(minorgrid) == 0){
            System.out.println("x's turn");
            int x = sc.nextInt();
            int y = sc.nextInt();
            grid[y][x] = "x";
            minorgrid[y][x] = -1;
            for(int i = 0; i < 3; i++){
                for(int j = 0; j < 3; j ++){
                    System.out.print(grid[i][j]);
                    System.out.print(" ");
                }
                System.out.println();
            }
            if(!flag && evaluate(minorgrid) == 0){
            System.out.println("o's turn");
            x = sc.nextInt();
            y = sc.nextInt();
            grid[y][x] = "o";
            minorgrid[y][x] = 1;
            for(int i = 0; i < 3; i++){
                for(int j = 0; j < 3; j ++){
                    System.out.print(grid[i][j]);
                    System.out.print(" ");
                }
                System.out.println();
            }
            }
    }
        return grid;
    }
    public static int evaluate(int[][] grid2){
        int sum = 0;
        int sum2 = 0;
        for(int i = 0; i < 3; i ++){
            for(int j = 0; j < 3; j ++){
                sum = sum + grid2[i][j];
                sum2 = sum + grid2[j][i];
            }
            if(sum == 3 || sum2 == 3){
                return 1;
            }
            else if (sum == -3 || sum2 == -3)
                    return -1;
            sum = 0;
            sum2 = 0;
        }
        for(int i = 0; i < 3; i ++){
            sum = sum + grid2[i][i];
        }
        if(sum == 3){
            return 1;
        }
        else if(sum == -3){
            return -1;
        }
        sum = 0;
        for(int i = 0; i < 3; i ++){
            sum = sum + grid2[i][3 - i - 1];
        }
        if(sum == 3){
            return 1;
        }
        else if(sum == -3){
            return -1;
        }
        return 0;
    }
 
}
