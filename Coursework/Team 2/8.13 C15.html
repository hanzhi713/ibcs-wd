class Point:
    def __init__(self, x = 0, y = 0):
        self.x = x
        self.y = y
    def reflect_x(p1):
        y = -p1.y
        return Point(p1.x, y)
    def slope_from_origin(p2):
        slope = p2.y / p2.x
        return slope
    def get_line_to(p3, p4):
        m = (p4.y - p3.y) / (p4.x - p3.x)
        b = p3.y - m * p3.x
        return (m, b)
    def midpoint(q1, q2):
        midpointx = (q1.x + q2.x) / 2
        midpointy = (q1.y + q2.y) / 2
        return Point(midpointx, midpointy)
    def onacircleorwhat(q1, q2, q3, q4):
        (a1, a2) = Point.get_line_to(q1, q2)
        (a3, a4) = Point.get_line_to(q2, q3)
        b1 = -1 / a1
        b3 = -1 / a3
        c1 = Point.midpoint(q1, q2)
        b2 = c1.y - b1 * c1.x
        c2 = Point.midpoint(q2, q3)
        b4 = c2.y - b3 * c2.x
        circlex = (b4 - b2) / (b1 - b3)
        circley = b1 * circlex + b2
        if (q4.x - circlex) ** 2 + (q4.y - circley) ** 2 == (q1.x - circlex)**2 + (q1.y - circley) ** 2:
            return (circlex, circley)
        else:
            return "Those 4 points are not on the circle"

    dx = abs(p.x - q.x)
    return (dx ** 2 + dy ** 2) ** 0.5

class SMS_store:
    def __init__(self):
        self.messages = []
    def add_new_arrival(self, from_number, time_arrived, text_of_SMS):
        has_been_viewed = False
        self.messages.append((has_been_viewed, from_number, time_arrived, text_of_SMS))
    def message_count(self):
        return len(self.messages)
    def get_unread_indexes(self):
        unread = []
        for i in self.messages:
            if i[0] == False:
                unread.append(i)
        return unread
    def get_message(self, i):
        if i - 1 >= self.message_count():
            return "You don't have this piece of message"
        else:
            c = self.messages[i - 1]
            (a, b, s) = (c[1], c[2], c[3])
            a = True
            self.messages[i - 1] = (a, b, s)
            return (a, b, s)
    def delete(self, i):
        if i - 1 >= self.message_count():
            return "You don't have this piece of message"
        else:
            del self.messages[i - 1]
    def clear(self):
        del self.messages[:self.message_count()]
