import java.util.Scanner;
public class cs1152016{
    public static void main(String[] args){
        Scanner in;
        double n;
 
        System.out.println("Please select an integer: ");
 
        in = new Scanner(System.in);
        n = in.nextDouble();
 
        for (int j = 0; j < n+1; j++) {
             for (int i = 0; i < n+1; i++){
                 if (i == j)
                     System.out.print(i);
                 else if (i == n)
                     System.out.println(-1);
                 else if (i > j)
                     System.out.print(-1);
                 else
                     System.out.print(0);
             }
         }
    }
}
