def Hanoi_tower(a,b,c,n):
    if n == 1:
        print ('num:',n,'from',a,'->',c)
 
    else:
        Hanoi_tower(a,c,b,n-1)
        print ('num:',n,'from',a,'->',c)
        Hanoi_tower(b,a,c,n-1)
 
#test for Hanoi tower
print(Hanoi_tower('a','b','c',3))

def FIBO(n):
    if n == 1 or n == 2:
        return 1
    elif n == 0:
        return 0
    else:
        return int(FIBO(n-1))+int(FIBO(n-2))
 
#test for FIBO
for i in range(10):
    print(FIBO(i))
