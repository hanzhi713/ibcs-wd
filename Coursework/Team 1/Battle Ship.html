package battleShipSample;
import java.util.Scanner;
 
public class Player {
    int[][] grids = new int[8][8];
 
    public static int[][] initialGrids(){
        int[][] grids = new int[8][8];
        int[] row = new int[8];
        for(int i = 0;i<8;i++){
            row[i]=0;
        }
        for(int i = 0;i<8;i++){
            grids[i] = row;
        }
        return grids;
    }
 
    public static int[][] shipPosition(int i,int rawNum1,int rawNum2,int colNum1,int colNum2){
        int[][] position = new int[2][i+1];
        int raw = rawNum1;
        int col = colNum1;
        int rawDifference;
        int colDifference;
        if (rawNum1 != rawNum2){
            rawDifference = (rawNum2-rawNum1)/(i);
            }
        else{
                rawDifference = 0;
        }
        if (colNum1 != colNum2){
            colDifference = (colNum2-colNum1)/(i);
        }
        else{
            colDifference = 0;
        }
        position[0][0]=raw;
        position[1][0]=col;
        for(int j=1;j<=i;j++){
            raw = raw + rawDifference;
            col = col + colDifference;
            position[0][j]=raw;
            position[1][j]=col;
        }
        return position;
    }
 
    public static int[][] changeGrids(int[][] grids){
        Scanner in = new Scanner(System.in);

package battleShipSample;
import java.util.Scanner;
 
public class ClientBettle {
 
    public static void main(String[] args){
        Player a = null;
        Player b = new Player1();
        b.grids=Player.initialGrids();
        System.out.println("Player 1");
        b.grids=Player.changeGrids(b.grids);
        Player c = new Player2();
        c.grids=Player.initialGrids();
        System.out.println("Player 2");
        c.grids=Player.changeGrids(c.grids);
        boolean controller = true;
 
        while(true){
 
            if (controller == true){
                a = b;
                System.out.println("Player1 ");
 
                if(a.hit(c)!=true){
                    controller = false;
                    System.out.println("missed");
                }
                else{
                    System.out.println("boom!");
                }
            }
            else if(find(a.grids,7,7,1)==false){
                System.out.println(a.getName()+" is loss!");
                break;
            }
            else{
                a = c;
                System.out.println("Player2 ");
 
                if(a.hit(b)!=true){
                    controller = true;
                    System.out.println("missed");
                }
                else{
                    System.out.println("boom!");
                }
            }
        }
    }
 
    public static boolean find(int[][] matrix,int rows,int columns,int number){
        if(matrix!=null && rows>0 && columns>0){
            int row = 0;
            int column = columns-1;
            while(row<rows && column>=0){
                if(matrix[row][column]==number){
                    return true;
                }
                if(matrix[row][column]>number){
                    column--;
                }else{
                    row++;
                }
            }
        }
        return false;
    }
}
 


Firsly, users should input staring row, ending row, starting coloum and ending coloum seperatly. The computer will ouput the position of their ships by calculating. All the value of this point change to 1.

After inputting, the player start to input the position they want to hit. If the value of this point is 1, there will be a output "boom", if it is 0, the ouput will be "missed". And another player begin to play.

If all values of one player are 0, the other player win this game.

The main idea of my code is Polymorphism. Player 1 and Player 2 is a subclass of Player.Which one is executing depends on the controller. So, this code is also suitable for situations if the number of players is greater than 2. But, there also lots of unsolving problem of this code. Such as I didn't consider the situation if the input of client is invalid. I think these problems can be developed in the future.
