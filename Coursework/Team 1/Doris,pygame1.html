import pygame
import sys
import random
 
 
 
 
 
class object(pygame.sprite.Sprite):
    def __init__(self,object_position,v):
        pygame.sprite.Sprite.__init__(self)
        self.image=pygame.image.load('damge0.png')
        self.rectangle=self.image.get_rect()
        self.rectangle.topleft=object_position
        self.v=v
    def move(self):
        self.rectangle=self.rectangle.move(self.v)
 
 
class rect():
    def __init__(self,picture_filename,p):
        self.image = pygame.image.load(picture_filename)
        self.rectangle=self.image.get_rect()
        self.rectangle.topleft=p
 
 
 
 
def loadtext(levelnum,score):
    my_font=pygame.font.SysFont(None,30)
    levelstr='Level:'+str(levelnum)
    scores=my_font.render(levelstr, True, (255, 255, 255))
    bakscreen.blit(scores, (50,20))
    scorestr='Score:'+str(score)
    scores=my_font.render(scorestr, True, (255, 255, 255))
    bakscreen.blit(scores, (50,50))
 
def over(scorenum):
    my_font=pygame.font.SysFont(None,80)
    levelstr='GAME OVER! hhh!'
    over_screen=my_font.render(levelstr, True, (255, 255, 255))
    bakscreen.blit(over_screen, (200,300))
