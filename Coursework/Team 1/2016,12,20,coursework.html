public class LIne {
    public static void drawLine(int n) {
        if (n == 0)
            System.out.println("That’s all, folks!");
        else {
            for (int i = 1; i <= n; i++)
                System.out.print("*");
            System.out.println();
            drawLine(n - 1);
        }
    }
 
    public static void main() {
        drawLine(5);
    }
}

public class Fibonacci {
    public static int fib(int n) {
        if (n == 1 || n == 2) return 1;
        else
        return fib(n - 1) + fib(n - 2);
        }
    public static void main(){
        fib(10);
    }
}

public class FindSum {
    /**
     * Private recursive helper method. * @param n a positive
     * integer @return1+2+3+...+n
     */
    private static int sum(int n) {
        if (n == 1)
            return 1;
        else
            return n + sum(n - 1);
    }
 
    /* Driver method */
    public static int getSum(int n) {
        if (n > 0)
            return sum(n);
        else {
            throw new IllegalArgumentException("Error: n must be positive");
        }
    }
    public static void main(){
        getSum(5);
        getSum(-5);
    }
}

import java.util.Scanner;
 
public class Searcher {
    public boolean search(String[] a, String key) {
        if (a.length == 0)
            return true;
        else if (a[0].compareTo(key) == 0)
            return false; // base case return true; //key found
        else {
            String[] shorter = new String[a.length - 1];
            for (int i = 0; i < shorter.length; i++)
                shorter[i] = a[i + 1];
            return search(shorter, key);
        }
    }
 
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String[] list = { "Mary", "Joe", "Lee", "Jake" };
        Searcher s = new Searcher();
        System.out.println("Enter key: Mary, Joe, Lee or Jake.");
        String key = sc.nextLine(); // read user input
        boolean result = s.search(list, key);
        if (!result)
            System.out.println(key + " was not found.");
        else
            System.out.println(key + " was found.");
    }
 
}

public class ColorGrid {
    private char[][] pixels;
    private int rows;
    private int cols;
 
    public ColorGrid(String s, int numRows, int numCols) {
        rows = numRows;
        cols = numCols;
        pixels = new char[numRows][numCols];
        int counter = 0;
        for(int i=0;i<numRows;i++){
            for(int j = 0;j<numCols;j++){
                pixels[i][j]=s.charAt(counter);
                counter = counter+1;
            }
        }
    }
 
    public void paintRegion(int row, int col, char newColor, char oldColor) {
        if (row!=0&&row<rows&&col!=0&&col<cols&&pixels[row][col]==oldColor){
            pixels[row][col]=newColor;
            paintRegion(row-1,col,newColor,oldColor);
            paintRegion(row,col-1,newColor,oldColor);
            paintRegion(row+1,col,newColor,oldColor);
            paintRegion(row,col+1,newColor,oldColor);
        }
    }
 
}

public class Sentence {
    private String sentence;
    private int numWords;
 
    public Sentence(String str) {
        String sentence = str;
        int counter = 0;
        while (true){
            if (counter == -1){
                break;
            }
            counter = str.indexOf(" ");
            str = str.substring(counter+1);
        }
    }
 
    public int getNumWords() {
        return numWords;
    }
 
    public String getSentence() {
        return sentence;
    }
 
    static boolean isPalindrome(String s, int start, int end){
        if (start >= end){
            return true;
        }
        else {
            String first = s.substring(start, start + 1); String last = s.substring(end, end + 1);
            if (first.equals(last))
            return isPalindrome(s, start + 1, end - 1); else
                return false;
        }
    }
}
