TicTacToe：
package battleShipSample;
 
public class TicTacToe {
    private String[][] board;
    private static final int ROWS = 3;
    private static final int COLS = 3;
 
    public TicTacToe() {
        board = new String[ROWS][COLS];
        for (int r = 0; r < ROWS; r++)
            for (int c = 0; c < COLS; c++)
                board[r][c] = " ";
    }
 
    public void makeMove(int r, int c, String symbol) {
        board[r][c] = symbol;
    }
 
    public boolean checkPoint(int row, int col){
        if( board[row][col]==" "){
            return false;
        }
        return true;
    }
    public String toString() {
        String s = "";
        for (int r = 0; r < ROWS; r++) {
            s = s + "|";
            for (int c = 0; c < COLS; c++)
                s = s + board[r][c];
            s = s + "|\n";
        }
        return s;
    }
    public boolean controller(){
        for(int i = 0;i<3;i++){
            if(board[i][0]==board[i][1]&&board[i][0]==board[i][2]&&board[i][0]!=" "){
                System.out.println(board[i][0]+"is win");
                return true;
            }
            if(board[0][i]==board[1][i]&&board[0][i]==board[2][i]&&board[0][i]!=" "){
                System.out.println(board[0][i]+"is win");
                return true;
            }
            if(board[0][0]==board[1][1]&&board[0][0]==board[2][2]&&board[0][0]!=" "){
                System.out.println(board[0][0]+"is win");
                return true;
            }
            if(board[0][2]==board[1][1]&&board[0][2]==board[2][0]&&board[0][2]!=" "){
                System.out.println(board[0][2]+"is win");
                return true;
            }
        }
        return false;
    }
    public boolean emptyPoint(){
        boolean controller = false;
        for (int r = 0; r < ROWS; r++)
            for (int c = 0; c < COLS; c++)
                if(board[r][c]==" "){
                    controller = true;
                }
        return controller;
    }
}
