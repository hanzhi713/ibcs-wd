import cv2
import numpy as np
from matplotlib import pyplot as plt
img = cv2.imread('/Users/xueyuxin/Desktop/testImg2.png')
 
 
def salt(img,n):
    for i in range(n):
        x = int(np.random.random() * img.shape[0])
        y = int(np.random.random() * img.shape[1])
        img[x,y]=[255,255,255]
    return img
 
n = int(img.shape[1]*img.shape[0]*0.1)
print(img.shape[1]*img.shape[0])
print(n)
saltImage = salt(img,n)
b,g,r = cv2.split(saltImage)
rgb_img = cv2.merge([r,g,b])
plt.imshow(rgb_img)
plt.show()
