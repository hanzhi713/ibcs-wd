def nostalgia(img_path):
        img = cv2.imread(img_path)
        b, g, r = cv2.split(img)
        nrows, ncols, a = img.shape
        for i in range(nrows):
            for j in range(ncols):
                red = r[i][j]
                green = g[i][j]
                blue = b[i][j]
                r[i][j] = int(0.393 * red + 0.769 * green + 0.189 * blue)
                g[i][j] = int(0.349 * red + 0.686 * green + 0.168 * blue)
                b[i][j] = int(0.272 * red + 0.534 * green + 0.131 * blue)
                if b[i][j] < 0:
                    b[i][j] = 0
                if b[i][j] > 255:
                    b[i][j] = 255
                if g[i][j] < 0:
                    g[i][j] = 0
                if g[i][j] > 255:
                    g[i][j] = 255
                if r[i][j] < 0:
                    r[i][j] = 0
                if r[i][j] > 255:
                    r[i][j] = 255
        img = cv2.merge((r, g, b))
        plt.imshow(img)
        plt.show()
        plt.savefig("result.png")
def salt(img_orig_path):
    img = cv2.imread(img_orig_path)
    n = int(img.shape[1]*img.shape[0]*0.1)
    for i in range(n):
        x = int(np.random.random() * img.shape[0])
        y = int(np.random.random() * img.shape[1])
        img[x,y]=[255,255,255]
    b,g,r = cv2.split(img)
    rgb_img = cv2.merge([r,g,b])
    plt.imshow(rgb_img)
    plt.show()
    plt.savefig("result.png")
def glass(img_orig_path):
    img_orig=cv2.imread(img_orig_path)
    nrows, ncols, a = img_orig.shape
    b, g, r = cv2.split(img_orig)
    for i in range(1, nrows - 1):
        for j in range(1, ncols - 1):
            tmp = int(random.uniform(0, 9))
            resulti = i - 1 + tmp // 3
            resultj = j - 1 + tmp % 3
            if resulti < 0:
                resulti = 0
            if resulti > nrows - 2:
                resulti = nrows - 2
            if resultj < 0:
                resultj = 0
            if resultj > ncols - 2:
                resultj = ncols - 2
            b[i][j] = b[resulti][resultj]
            r[i][j] = r[resulti][resultj]
            g[i][j] = g[resulti][resultj]
    img = cv2.merge((r, g, b))
    plt.imshow(img)
    plt.savefig("result.png")
def mask(img_path):
    img = cv2.imread(img_path)
    nrows, ncols, a = img.shape
    row, col = np.ogrid[:nrows, :ncols]
    cnt_row, cnt_col = nrows / 2, ncols / 2
    outer_disk_mask = ((row - cnt_row) ** 2 + (col - cnt_col) ** 2 > (ncols / 2) ** 2)
    img[outer_disk_mask] = 0
 
    b, g, r = cv2.split(img)
    img = cv2.merge((r, g, b))
    plt.imshow(img)
    plt.savefig("result.png")
