public class MagicSquare {
    public static int[][] generator(int n){
        int[][] result = new int[n][n];
        int x = 0;
        int y = n/2;
        result[x][y] = 1;
        for(int i = 2;i<=n*n;i++){
            int a = x;
            int b = y;
            if (x == 0){
                x = n-1;
            }
            else{
                x --;
            }
            if (y == 0){
                y = n-1;
            }
            else{
                y --;
            }
            if (result[x][y]!=0){
                x = (a+1)%n;
                y = b;
            }
            result[x][y]=i;
        }
        return result;
    }
