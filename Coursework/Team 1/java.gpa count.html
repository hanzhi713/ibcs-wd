import java.util.Scanner;
import java.io.File;
import java.io.IOException;
 
public class HistoArray {
    public static void main(String[] args) {
        Scanner data = null;
        int[] count = {0,0,0,0,0};
        char[] a1 = {'A','b','c','d','e'};
        int idx;
 
        try {
                data = new Scanner(new File("gpa.txt"));
        }
        catch ( IOException e) {
            System.out.println("Sorry but I was unable to open your data file");
            e.printStackTrace();
            System.exit(0);
        }
 
        while(data.hasNextInt()) {
            idx = data.nextInt();
            if (idx > 100 || idx < 0)
                System.out.println("invalid grade");
            else if (idx >= 90)
                count[0] = count[0] + 1;
            else if (idx >= 80 && idx < 90)
                count[1] = count[1] + 1;
            else if (idx >= 70 && idx < 80)
                count[2] = count[2] + 1;
            else if (idx >= 60 && idx < 70)
                count[3] = count[3] + 1;
            else
                count[4] = count[4] + 1;
 
 
 
        }
 
        idx = 0;
        for(int i : count) {
            System.out.println(a1[idx] + " occured " + i + " times.");
            idx++;
        }
    }
}
