public class Date {
    private int day;
    private int month;
    private int year;
    public Date(){
        day = 0;
        month = 0;
        year = 0;
    }
    public Date(int mo,int da,int yr){
        day = da;
        month = mo;
        year = yr;
    }
    public int day(){
        return day;
    }
    public int year(){
        return year;
    }
    public String toString(){
        String a = month + "/" + day + "/" + year;
        return a;
    }
}

public class Rational {
    private int numerator;
    private int denominator;
    public Rational() {
        numerator = 0;
        denominator = 1;
    }
    public Rational(int num){
        numerator = num;
        denominator = 1;
    }
    public Rational(int num, int den){
        numerator = num;
        denominator = den;
    }
    public int numerator(){
        return numerator;
    }
    public int denominator(){
        return denominator;
    }
    public Rational plus(Rational r){
        this.numerator = this.numerator+r.numerator;
        this.denominator = this.denominator + r.denominator;
        return this;
    }
}

public class TemperatureFT {
    private String scale;
    private double degrees;
    static int Valid;
    public TemperatureFT(double tempDegrees, String tempScale){
        scale = tempScale;
        degrees = tempDegrees;
    }
    public TemperatureFT toFahrenheit(){
        TemperatureFT.isValidTemperature(this);
        if (Valid == 1){
            this.degrees=this.degrees+273;
            this.scale = "Faherenheit";
            return this;
        }
        else{
            System.out.println("Temperature is invlid");
            return null;
        }
    }
    public TemperatureFT toCelsius(){
        TemperatureFT.isValidTemperature(this);
        if (Valid == 1){
            this.degrees=this.degrees-273;
            this.scale = "Celsius";
            return this;
        }
        else{
            System.out.println("Temperature is invlid");
            return null;
        }
    }
    public TemperatureFT lower(double a){
        this.degrees = this.degrees - a;
        return this;
    }
    public TemperatureFT raise(double a){
        this.degrees = this.degrees + a;
        return this;
    }
    public void printTemperatur(){
        System.out.println(this.scale+"   "+this.degrees);
    }
    static void isValidTemperature(TemperatureFT r){
        if (r.scale == "Celsius"){
            if (r.degrees>=-273){
                Valid = 1;
            }
            else{
                Valid = 0;
            }
        }
        if (r.scale == "Faherenheit"){
            if (r.degrees >= 0){
                Valid = 1;
            }
            else{
                Valid = 0;
            }
        }
        System.out.println(Valid);
    }
 
}
