def num_zero_and_five_digits(n):
    count = 0
    while n > 0:
        digit = n % 10
        if digit == 0 or digit == 5:
            count = count + 1
        n = n // 10
    return count

def is_prime(n):
    if n < 2:
        return False
    else:
        for i in range(2,n):
            if n % i == 0:
                return True
            return False

    digit = 0
    while not  -2 < n < 2:
        if n % 2 == 0 and n % 10 != 0:
            digit += 1
        n = n // 10
    return digit
