import turtle
alex = turtle.Turtle()
i = 110
alex.penup()
alex.goto(0,-100)
alex.pendown()
alex.speed(20)
step=0
while i > 10:
    step += 1
    i -= 10
    if step % 2 == 1:
        alex.fillcolor('red')
    else:
        alex.fillcolor('white')
    alex.begin_fill()
    alex.circle(i)
    alex.write(step)
    alex.end_fill()
    alex.penup()
    alex.left(90)
    alex.forward(10)
    alex.pendown()
    alex.right(90)
