def is_anagram(word1, word2):
    if len(word1) != len(word2):
        return False
    a = []
    for i in word1:
        a.append(i)
    b = []
    for j in word2:
        b.append(j)
    a.sort()
    b.sort()
    if a != b:
        return False
    else:
        return True
 
print(is_anagram('asd','sae'))

    a=[]
    for i in x:
        if i not in a:
            a.append(i)
    return a
