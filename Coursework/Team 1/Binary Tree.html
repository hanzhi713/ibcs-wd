import java.util.ArrayList;
 
public class Tree {
    Station station;
    Tree left;
    Tree right;
    Tree a;
    static ArrayList<Tree> Nodes = new ArrayList<Tree>() ;
 
    public Tree(Station station, Tree left, Tree right) {
        this.station = station;
        this.left = left;
        this.right = right;
    }
 
    public Tree(Station station) {
        this.station = station;
        this.right = null;
        this.left = null;
    }
 
    public Tree(Station station, Tree left, String a) {
        this.station = station;
        this.left = left;
        this.right = null;
    }
 
    public Tree(Station station, String a, Tree right) {
        this.station = station;
        this.right = right;
        this.left = null;
    }
 
    public static int getZone(Tree rootNode, String stationName) {
        Tree treeNode = Tree.search(rootNode, stationName);
        return treeNode.station.getZone();
    }
 
    public static int getLineNumber(Tree rootNode, String stationName) {
        Tree treeNode = Tree.search(rootNode, stationName);
        return treeNode.station.getLineNumber();
    }
 
    public static Tree search(Tree treeNode, String stationName) {
        if (stationName.equals(treeNode.station.getName())) {
            return treeNode;
        } else if (stationName.compareTo(treeNode.station.getName()) < 0) {
            return Tree.search(treeNode.left, stationName);
        } else {
            return Tree.search(treeNode.right, stationName);
        }
    }
    public static Tree findSmallest(Tree rootNode){
        Tree root = rootNode.right;
        if (root.left==null){
            Nodes.add(root);
            return root;
        }
        else{
            while(root.left!=null){
                Nodes.add(root);
                root = root.left;
            }
            return root;
        }
 
    }
    public static void delete2(Tree delNode){
        Tree repNode = findSmallest(delNode);
        Tree parentNode = Nodes.remove(Nodes.size()-1);
        parentNode.left = null;
        delNode.station = repNode.station;
    }
      public static void inOrderTraverse(Tree node) {
            if (node == null)
                return;
            inOrderTraverse(node.left);
            System.out.print(node.station.getName() + " ");
            inOrderTraverse(node.right);
        }
 
}
