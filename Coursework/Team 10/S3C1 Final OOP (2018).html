package com.company;
 
public class TrainCompany {
    private String companyName;
    private String companyCode;
    private int numberOfJourneys;
    private Journey[] journeyHistory = new Journey[1000000];
 
    public TrainCompany(String x, String y){
        this.companyName = x;
        this.companyCode = y;
        this.numberOfJourneys = 0;
    }
 
    public String getCompanyName(){
        return companyName;
    }
    public String getCompanyCode(){
        return companyCode;
    }
    public int getNumberOfJourneys(){
        return numberOfJourneys;
    }
//    public Journey getJourneyHistory(){
//        return journeyHistory;
//    }
    public void setCompanyName(String companyName1){
        companyName = companyName1;
    }
    public void setCompanyCode(String companyCode1){
        companyCode = companyCode1;
    }
    public void setNumberOfJourneys(int numberOfJourneys1){
        numberOfJourneys = numberOfJourneys1;
    }
 
    public void addJourney(Journey j){
        journeyHistory[numberOfJourneys] = j;
        numberOfJourneys++;
    }
 
    public double averageDelay(){
 
        int sum = 0;
        int num = 0;
        int ad = 0;
 
        for(int k = 0; k<numberOfJourneys; k++){
            if(journeyHistory[k].getWeatherRelated() == false){
                sum = sum + journeyHistory[k].getDelay();
                num++;
            }
        }
        ad = sum/num;
        return ad;
    }
 
    public String longestDelay(Codes[] c){
        int max = -1;
        String rc = " ";
        String name = " ";
        for(int k = 0; k < numberOfJourneys; k++){
            if(!journeyHistory[k].getWeatherRelated() && journeyHistory[k].getDelay()>max){
                max = journeyHistory[k].getDelay();
                rc = journeyHistory[k].getRouteCode();
            }
        }
        for(int n = 0; n < c.length; n++){
            if(c[n].getRouteCode().equals(rc)){
                name = c[n].getRouteName();
            }
        }
        return name;
    }
 
    public String toString(Codes[]c){
        return companyName+" Average Delay = "+averageDelay()+" minutes : Longest Delay = "+longestDelay(c);
    }
}
