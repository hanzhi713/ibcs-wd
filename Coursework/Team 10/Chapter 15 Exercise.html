class Point:
    def __init__(self,x,y):
        self.x=x
        self.y=y
    def reflect_x(self):
        x1=self.x
        y1=(-1)*(self.y)
        return Point(x1,y1)
    def __str__(self):
        return "{0},{1}".format(self.x,self.y)
k=Point(3,5)
print(k.reflect_x())

class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    def get_line_to(self,target):
        a=(target.y-self.y)/(target.x-self.x)
        b=self.y-a*self.x
        return a,b
    def __str__(self):
        return "{0},{1}".format(a,b)
print(Point(4,11).get_line_to(Point(6,15)))
