package com.company;
 
public class Dates {
    private int day;
    private int month;
    private int year;
 
    public Dates(int day,int month,int year){
        this.day = day;
        this.month = month;
        this.year = year;
    }
 
    public int getDay(){
        return day;
    }
    public int getMonth(){
        return month;
    }
    public int getYear(){
        return year;
    }
    public static int StayDays(Dates x, Dates y){
        int stay = 0;
        if(x.year == y.year){
            if(x.month != y.month){
                stay = (y.month - x.month - 1)*30 + (30 - x.day) + y.day;
            }
            else{
                stay = y.day - x.day;
            }
        }
        else{
            stay = StayDays(x, new Dates(30,12,x.year))+StayDays(new Dates(1,1,y.year),y);
        }
        return stay;
    }
}
