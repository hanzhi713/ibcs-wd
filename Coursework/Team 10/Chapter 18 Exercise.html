def Cesaro(t, order, size):

        t.forward(size)
    else:

def r_min(nxs):
 
    smallest = None
    first_time = True
    for e in nxs:
        if type(e) == type([]):
            val = r_min(e)
        else:
            val = e
 
        if first_time or val < smallest:
            smallest = val
            first_time = False
 
    return smallest

def count(list1,target):
    c=0
    for i in list1:
        if i==target:
            c+=1
        else:
            continue
    return c

def flatten(list1):
    list2=[]
    for i in list1:
        if type(i)==type([]):
            flatten(i)
        else:
            list2.append(i)
    return list2
