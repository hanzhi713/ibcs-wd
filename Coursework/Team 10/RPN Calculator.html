import math
 
def postfix_evaluation(s):
    s = s.split()
    n = len(s)
    stack = []
 
    for i in range(n):
        if s[i].isdigit():
            stack.append(int(s[i]))
        elif s[i]=="+":
            a = stack.pop()
            b = stack.pop()
            stack.append(int(a)+int(b))
        elif s[i] =="*":
            a = stack.pop()
            b = stack.pop()
            stack.append(int(a)*int(b))
        elif s[i] =="-":
            a = stack.pop()
            b = stack.pop()
            stack.append(int(a) - int(b))
        elif s[i] =="/":
            a = stack.pop()
            b = stack.pop()
            stack.append(int(a)/int(b))
        elif s[i] =="%":
            a = stack.pop()
            b = stack.pop()
            stack.append(int(a) % int(b))
        elif s[i]=="**":
            a = stack.pop()
            b = stack.pop()
            stack.append(int(a) ** int(b))
        elif s[i]=="sin":
            a = stack.pop()
            stack.append(math.sin(math.radians(int(a))))
        elif s[i]=="cos":
            a = stack.pop()
            stack.append(math.cos(math.radians(int(a))))
        elif s[i]=="tan":
            a = stack.pop()
            stack.append(math.tan(math.radians(int(a))))
 
    return stack.pop()
 
def infix_evaluation(s):
    s = s.split()
    n = len(s)
    stack = []
 
    for i in range(n):
        if s[i].isdigit():
            stack.append(str(s[i]))
        elif s[i].islower() or s[i].isupper():
            stack.append(str(s[i]))
        else:
            a = stack.pop()
            b = stack.pop()
            if s[i]=="+":
                stack.append("(" + str(b) + "+" + str(a) + ")")
            elif s[i] =="*":
                stack.append("(" + str(b) + "*" + str(a) + ")")
            elif s[i] =="-":
                stack.append("(" + str(b) + "-" + str(a) + ")")
            elif s[i] =="%":
                stack.append("(" + str(b) + "%" + str(a) + ")")
            elif s[i]=="**":
                stack.append("(" + str(b) + "**" + str(a) + ")")
            elif s[i]=="sin":
                stack.append("sin" + "(" + str(a) + ")")
            elif s[i]=="cos":
                stack.append("cos" + "(" + str(a) + ")")
            elif s[i]=="tan":
                stack.append("tan" + "(" + str(a) + ")")
 
    return stack.pop()
 
s = "10 2 8 * + 3 -"
val = postfix_evaluation(s)
cal = infix_evaluation(s)
print(val)
print(cal)
