import pygame
import random
 
color1 = (255, 0, 255)
color2 = (0, 0, 0)
color3 = (100,200,10)
WHITE = (255, 255, 255)
RED = (255,0,0)
 
 
class Element:
 
    def __init__(self,x1,y1,width1,height1):
        self.x1 = x1
        self.y1 = y1
        self.width1 = width1
        self.height1 = height1
 
    def Rectangle(self, color2, screen):
        pygame.draw.rect(screen, color2, (self.x1, self.y1, self.width1, self.height1), 0)
 
 
 
class Circle:
 
    def __init__(self,x2,y2,radius):
        self.x2 = x2
        self.y2 = y2
        self.radius = radius
 
    def draw1(self, color1, screen):
        pygame.draw.circle(screen, color1, (self.x2, self.y2), self.radius, 0)
        pygame.draw.circle(screen, WHITE, (self.x2, self.y2), self.radius, 0)
        self.x2 += 5
        pygame.draw.circle(screen, color1, (self.x2, self.y2), self.radius, 0)
 
 
class Block:
 
     def __init__(self,x3,y3,width3,height3):
         self.x3 = x3
         self.y3 = y3
         self.width3 = width3
         self.height3 = height3
 
     def draw2(self,color3,screen):
         pygame.draw.rect(screen,color3,(self.x3, self.y3, self.width3, self.height3), 0)
 
pygame.init()
screen_width = 700
screen_height = 400
screen = pygame.display.set_mode([screen_width, screen_height])
pygame.display.set_caption("My Game")
 
my_font = pygame.font.SysFont("Courier", 50)
 
screen.fill([255, 255, 255])
a = Element(10, 10, 40, 20)
a.Rectangle(color2, screen)
 
b_list = []
for i in range(5):
    b = Circle(40,20,8)
    b.draw1(color1, screen)
    b_list.append(b)
c = Block(650,random.randrange(50,150),30,30)
c.draw2(color3, screen)
 
 
while True:
    ev = pygame.event.poll()
    if ev.type == pygame.QUIT:
        break
    elif ev.type == pygame.KEYDOWN:
        if ev.key == pygame.K_UP:
            pygame.draw.rect(screen, WHITE, (a.x1, a.y1, a.width1, a.height1), 0)
            pygame.draw.circle(screen, WHITE, (b.x2, b.y2), b.radius, 0)
            a.y1 += -5
            b.y2 += -5
        if ev.key == pygame.K_DOWN:
            pygame.draw.rect(screen, WHITE, (a.x1, a.y1, a.width1, a.height1), 0)
            pygame.draw.circle(screen, WHITE, (b.x2, b.y2), b.radius, 0)
            a.y1 += 5
            b.y2 += 5
    elif ev.type == pygame.KEYUP:
        if ev.key == pygame.K_UP or ev.key == pygame.K_DOWN:
            a.y1 += 0
            b.y2 += 0
 
    if (c.x3 + c.width3) - (b.x2-b.radius) < b.radius + c.width3 and c.y3 < b.y2 < c.y3 + c.height3:
        pygame.draw.rect(screen, WHITE, (c.x3, c.y3, c.width3, c.height3), 0)
        the_text = my_font.render("You Won!", True, RED)
        screen.blit(the_text, (250, 180))
 
    a.Rectangle(color2, screen)
    for i in b_list:
        if b.x2 > 800:
            b.x2 = 80
        b.draw1(color1, screen)
 
    pygame.display.flip()
 
pygame.quit()
