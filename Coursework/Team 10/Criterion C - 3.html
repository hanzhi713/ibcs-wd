import javax.imageio.ImageIO;
import javax.swing.*;
import java.awt.*;
import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;
 
public class addImage {
    private String pathway;
    private JPanel panel;
    private Image image,image1;
    private BufferedImage image2;
 
    public addImage(String pathway) {
        this.pathway = pathway;
    }
 
    public Image load() throws IOException {
        image = Toolkit.getDefaultToolkit().getImage(pathway);
        return image;
    }
 
    public void changeColor(int num) {
        image1 = Toolkit.getDefaultToolkit().getImage(pathway);
        image2 = new BufferedImage(image1.getWidth(null), image1.getHeight(null), image2.TYPE_INT_RGB);
        image2.getGraphics().drawImage(image1, 0, 0, null);
        int[] rgb = new int[3];
        int w = image2.getWidth(null);
        int h = image2.getHeight(null);
        for (int col = 0; col < w; col++) {
            for (int row = 0; row < h; row++) {
                int pixel = image2.getRGB(col,row);
                rgb[0] = (pixel & 0xff0000) >> 16;
                rgb[1] = (pixel & 0xff00) >> 8;
                rgb[2] = (pixel & 0xff);
                if(rgb[0]<20 && rgb[1]<20 && rgb[2]<20){
                    image2.setRGB(col, row, num);
                }
            }
        }
    }
}

public class createSlider extends JFrame{
    public JSlider  s;

    public createSlider(){
        s = new JSlider();
        s.setMajorTickSpacing(20);
        s.setMinorTickSpacing(5);
        s.setPaintTicks(true);
        s.setPaintLabels(true);
        s.setOrientation(SwingConstants.VERTICAL);
        s.getExtent();
    }
 
    public void addSlider(JSlider slider, String description){
        JPanel sliderPanel = new JPanel();
        sliderPanel.setLayout(new FlowLayout(FlowLayout.LEFT));
        JPanel panel = new JPanel();
        panel.add(slider);
        panel.add(new JLabel(description));
        sliderPanel.add(panel);
    }
}
