package com.company;
 
public class ColorGrid {
    private String[][]pixels;
    private int rows;
    private int cols;
 
    public ColorGrid(String s, int numRows, int numCols){
        rows = numRows;
        cols = numCols;
        String[][] pixels = new String[rows][cols];
        int index = 0;
        for(int r=0; r<rows; r++){
            for(int c=0; c<cols; c++){
                pixels[r][c]=s.substring(index,index+1);
                index++;
            }
        }
    }
    public void paintRegion(int row, int col, String newColor, String oldColor){
        if(row>=0 && row<rows && col>=0 && col<cols){
            if((pixels[row][col]).equals(oldColor)&&(!(newColor.equals(oldColor)))){
                pixels[row][col]=newColor;
                if(pixels[row+1][col].equals(oldColor)){
                    row=row+1;
                    paintRegion(row, col, newColor, oldColor);
                }
                else if(pixels[row-1][col].equals(oldColor)){
                    row=row-1;
                    paintRegion(row, col, newColor, oldColor);
                }
                if(pixels[row][col+1].equals(oldColor)){
                    col= col+1;
                    paintRegion(row, col, newColor, oldColor);
                }
                if(pixels[row][col-1].equals(oldColor)){
                    col=col-1;
                    paintRegion(row, col, newColor, oldColor);
                }
            }
        }
    }
}

package com.company;
 
public class Sentence {
 
    private String sentence;
    private int numWords;
 
    public Sentence(String str) {
        sentence = str;
        int blankCount = 0;
        for (int i = 0; i < sentence.length(); i++) {
            if (sentence.substring(i, i + 1).equals(" ")) {
                blankCount += 1;
            }
        }
        numWords = blankCount + 1;
    }
 
    private static boolean isPalindrome(String s, int ini, int end){
        if(s.length()<=1){
            return true;
        }
        else{
            ini=0;
            end=s.length();
            String iniString=s.substring(ini,ini+1);
            String endString=s.substring(end,end+1);
            if(iniString.equals(endString)){
                ini=ini+1;
                end=end-1;
                return isPalindrome(s,ini,end);
            }
            else{
                return false;
            }
        }
    }
}
