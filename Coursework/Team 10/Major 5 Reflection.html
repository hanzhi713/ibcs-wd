public class station {
    String name;
    int line;
    int zone;
 
    public station(String name, int line, int zone){
        this.name = name;
        this.line = line;
        this.zone = zone;
    }
 
    public int getLine(){
        return line;
    }
 
    public int getZone(){
        return zone;
    }
 
    public String getName(){
        return name;
    }
 
}
 
 
public class Tree {
    station cargo;
    Tree left, right;
 
    public Tree(station cargo, Tree left, Tree right){
        this.cargo = cargo;
        this.left = left;
        this.right = right;
    }
 
    public static int getLine(Tree node, String target){
        return TreeSearch(node,target).cargo.getLine();
    }
 
    public static int getZone(Tree node, String target){
        return TreeSearch(node,target).cargo.getZone();
    }
 
    public static Tree TreeSearch(Tree node, String target){
        Tree result = null;
        if(node.cargo.getName().compareTo(target)>0){
            result = TreeSearch(node.left,target);
        }
        else if(node.cargo.getName().compareTo(target)<0){
            result = TreeSearch(node.right,target);
        }
        else if(node.cargo.getName().equals(target)){
            result = node;
        }
        return result;
    }
}

     int[][] Fares = new int[12][12];
 
 
     for(int i=0;i<11;i++){
         station station1 = station[i];

         int AL = Tree.getLine(centro,station1.getName());
         for(int j=0;i<12;j++){
             station station2 = station[j];
             int BZ = Tree.getZone(centro,station2.getName());
             int BL = Tree.getLine(centro,station2.getName());
             if(AL==BL){
                 x = AZ - BZ;
                 if(x<0){
                     x = -x;
                 }
                 x+=1;
             }
             else{
                 x = AZ+BZ+1;
             }
             Fares[i][j] = x;
             Fares[j][i] = x;
         }
         Fares[i][i] = 0;
     }
