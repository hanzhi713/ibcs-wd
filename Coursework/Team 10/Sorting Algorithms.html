import java.util.ArrayList;
 
public class SortingAlgorithm {
    private int[]arr;
 
    public SortingAlgorithm(int[]arr){
        this.arr = arr;
    }
 
    public int[] SelectionSort(){
        for(int i=0;i<arr.length;i++){
            int min = i;
            for(int j=0;j<arr.length;j++){
               if(arr[j]< arr[min]){
                   arr[min]=arr[j];
               }
            }
            int temp = arr[i];
            arr[i] = arr[min];
            arr[min] = temp;
        }
        return arr;
    }
 
    public int[] InsertionSort(){
        for(int i=1;i<arr.length;i++){
            int num = arr[i];
            int j = i - 1;
            while(j>0 && num<arr[j]){
                arr[j+1]=arr[j];
                j = j-1;
            }
            arr[j+1] = num;
        }
        return arr;
    }
 
 
    public int[] MergeSort(int low,int high){
        if(low<high){
            int mid = (low+high)/2;
            MergeSort(low,mid);
            MergeSort(mid+1,high);
            Merge(low,mid,high);
        }
        return arr;
    }
 
    public int[] Merge(int low, int mid, int high) {
        int i = 0;
        int j = 0;
        int count = 0;
        int[] lower = new int[mid - low];
        int[] higher = new int[high - (mid + 1)];
        int[] merged = new int[arr.length];
        for (int a = 0; a < (mid - low); a++) {
            lower[a] = arr[low + a];
        }
        for (int b = 0; b < (high - (mid + 1)); b++) {
            higher[b] = arr[mid + 1 + b];
        }
        while (i < lower.length && j < higher.length) {
            if (lower[i] < higher[j]) {
                arr[count + i] = lower[i];
                i++;
                count++;
            } else if (lower[i] > higher[j]) {
                arr[count + j] = higher[j];
                j++;
                count++;
            }
        }
        return arr;
    }
 
}
