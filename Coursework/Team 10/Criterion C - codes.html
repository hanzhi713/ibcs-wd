class AnalogPlot:
#constr
    def __init__(self, strPort, maxLen):
 
        self.ser = serial.Serial(strPort, 9600)
        self.ax = deque([0.0]*maxLen)
        self.ay = deque([0.0]*maxLen)
        self.maxLen = maxLen
# add to buffer
    def addToBuf(self, buf, val):
        if len(buf) < self.maxLen:
            buf.append(val)
        else:
            buf.pop()
            buf.appendleft(val)
# add data
    def add(self, data):
        assert(len(data) == 2)
        self.addToBuf(self.ax, data[0])
        self.addToBuf(self.ay, data[1])
# update plot
    def update(self, frameNum, a0, a1):
        try:
            line = self.ser.readline()
            data = [float(val) for val in line.split()]
            if(len(data) == 2):
                self.add(data)
                a0.set_data(range(self.maxLen), self.ax)
                a1.set_data(range(self.maxLen), self.ay)
        except KeyboardInterrupt:
            print('exiting')
        return a0
# clean up
    def close(self):
# close serial
        self.ser.flush()
        self.ser.close()

public class addImage {
    private String pathway;
    private JPanel panel;
    private Image image,image1;
    private BufferedImage image2;
 
    public addImage(String pathway) {
        this.pathway = pathway;
//        this.panel = panel;
    }
 
    public Image load() throws IOException {
        image = Toolkit.getDefaultToolkit().getImage(pathway);
        return image;
    }
 
    public void changeColor(int num) {
        image1 = Toolkit.getDefaultToolkit().getImage(pathway);
        image2 = new BufferedImage(image1.getWidth(null), image1.getHeight(null), image2.TYPE_INT_RGB);
        image2.getGraphics().drawImage(image1, 0, 0, null);
        int[] rgb = new int[3];
        int w = image2.getWidth(null);
        int h = image2.getHeight(null);
        for (int col = 0; col < w; col++) {
            for (int row = 0; row < h; row++) {
                int pixel = image2.getRGB(col,row);
//                if (pixel < 20) {
                rgb[0] = (pixel & 0xff0000) >> 16;
                rgb[1] = (pixel & 0xff00) >> 8;
                rgb[2] = (pixel & 0xff);
                if(rgb[0]<20 && rgb[1]<20 && rgb[2]<20){
                    image2.setRGB(col, row, num);
                }
            }
        }
    }
}
