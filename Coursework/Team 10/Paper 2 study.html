public class Patient {
    private int id;
    private String name;
    private int priority;
    private String doctor;
 
    public Patient(int i, String n, int p){
        id=i;
        name=n;
        priority=p;
        doctor=null;
    }
 
    public void setId(int i){id=i;}
    public void setName(String n){name = n;}
    public void setPriority(int p){priority = p;}
    public int getId(){return id;}
    public String getName(){return name;}
    public int getPiroirty(){return priority;}
    public String getDoctor(){return doctor;}
    public String toString(){return id + " " + name + " " + priority + " " + doctor;}
}
 
public class WaitingRoom {
    private Patient[] patients = new Patient [10];
 
    public WaitingRoom(){
    }
 
    public void add(Patient newPatient){
        int i = 0;
        while(((patients[i])!=null)&&(i<10)){
            i = i+1;
        }
        if(i==10){System.out.println("No moew space in the waiting room.");}
        else{patients[i]=newPatient;}
    }
 
    public void callNextPatient(){
        int index = 0;
        if(patients[0]==null){
            System.out.println("The waiting room is empty.");
        }
        else{
            index = findNextPatientIndex();
            remove(index);
        }
    }
 
    private int findNextPatientIndex(){
        int i=1;
        int max=0;
        while ((i<10)&&(patients[i]!=null)){
            if(patients[i].getPiroirty() > patients[max].getPiroirty()){
                max = i;
            }
            i = i+1;
        }
        return max;
    }
 
    public void remove(int n){
        System.out.println(patients[n].toString());
        for(int i=n; i<9; i=i+1){
            patients[i] = patients[i+1];
        }
        patients[9] = null;
    }
}
 
 
public class Treatment {
    String date;
    Integer patientId;
    String doctor;
    String actions;
    String medication;
 
    public Treatment(String date, int id, String d, String a, String m){
        this.date = date;
        patientId = id;
        doctor = d;
        actions = a;
        medication = m;
    }
    public Treatment(){
    }
    public void setDate(String date){
        this.date = date;
    }
    public void setPatientId(Integer id){
        patientId = id;
    }
    public void setDoctor(String doctor){
        this.doctor = doctor;
    }
    public void setMedication(String medication){
        this.medication = medication;
    }
    public String getdate(){
        return date;
    }
    public Integer getPatientID(){
        return patientId;
    }
    public String getDoctor(){
        return doctor;
    }
    public String getMedication(){
        return medication;
    }
    public String toString(){
        return date+" "+patientId+" "+doctor+" "+actions+" "+medication;
    }
}
 
public class Ambulatory extends Treatment {
    private String medication;
 
    public Ambulatory(String med){
        super();
        medication = med;
    }
}
 
public class Referral extends Treatment{
    private String hospital;
    private Boolean ambulance;
 
    public Referral(String hos, Boolean amb){
        super();
        hospital = hos;
        ambulance = amb;
    }
}
public class InPatient extends Treatment {
    private String medication;
    private String roomNumber;
 
    public InPatient(String med, String rn){
        super();

    }
}
