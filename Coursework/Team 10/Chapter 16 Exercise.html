class Point:
    def __init__(self,x,y):
        self.x=x
        self.y=y
 
class Rectangle:
    def __init__(self,posn, w, h):
        self.corner = posn
        self.width = w
        self.height = h
 
    def area(self):
        Area=self.width*self.height
        return Area
 
    def perimeter(self):
        Perimeter= 2*(self.width+self.height)
        return Perimeter
 
    def flip(self):
        new_width=self.height
        new_height=self.width
        new=[]
        new.append(new_width)
        new.append(new_height)
        return new
 
    def point_falls(self,x,y):
        if x<10 and y<5:
            return True
        else:
            return False
 
 
    def collision(self,target,width1,height1):
        if self.corner.x < target.x < self.corner.x + self.width and self.corner.y - self.height < target.y < self.corner.y:
            return True
        elif self.corner.x<target.x<self.corner.x+self.width and self.corner.y-self.height<target.y-height1<self.corner.y:
                return True
        elif self.corner.x<target.x+width1<self.corner.x+self.width and self.corner.y-self.height<target.y<self.corner.y:
                return True
        elif self.corner.x<target.x+width1<self.corner.x+self.width and self.corner.y-self.height<target.y-height1<self.corner.y:
                return True
        else:
                return False
