Bank Account code:
import java.util.*;
 
public class BankAccount {
    public String password;
    public double balance;
    public static final double OVERDRAWN_PENALTY=20.00;
 
    public BankAccount(){
 
    }
    public BankAccount(String acctPassword,double acctBalance){
 
    }
    public double getBalance(){
        return balance;
    }
 
    public double deposit(String acctPassword,double amount){
        if(!acctPassword.equals(password)){
            System.out.println("Please enter your password again.");
        }
        else{
            balance += amount;
        }
        return balance;
    }
 
    public double withdraw(String acctPassword,double amount){
        if(!acctPassword.equals(password)){
            System.out.println("Please enter your password again.");
        }
        else{
            balance -= amount;
            if(balance<0){
                balance-=OVERDRAWN_PENALTY;
            }
        }
        return balance;
    }
 
    public String changePassword(String acctPassword,String newPassword){
        if(!acctPassword.equals(password)){
            System.out.println("Please enter your password again.");
        }
        else{
            password = newPassword;
        }
        return password;
    }
 
    public static double getInterestRate(){
        System.out.println("Enter your interest rate in decimal form please:");
        Scanner in = new Scanner(System.in);
        double interestRate=in.nextDouble();
        return interestRate;
    }
 
    public static void main(String[]args){
        Scanner scanner = new Scanner(System.in);
        BankAccount n = new BankAccount("Anacoda",200);
        n.password="Anacoda";
        n.balance=500;
        System.out.println(n.getBalance());
        System.out.println("How much do you want to deposit in");
        int amount=scanner.nextInt();
        System.out.println("Now your balance is " + n.deposit("Anacoda",amount));
        System.out.println("How much do you want to withdraw");
        int amount1=scanner.nextInt();
        System.out.println("Now your balance is "+n.withdraw("Anacoda",amount1));
        n.changePassword("Anacoda","Carl");
        System.out.println(n.password);
        System.out.println("Now your balance is "+n.balance);
        System.out.println(n.getInterestRate());
 
    }
}
