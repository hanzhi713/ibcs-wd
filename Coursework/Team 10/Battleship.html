public class Point {
    public int xCoord;
    public int yCoord;
 
    public Point(int xCoord, int yCoord){
        this.xCoord=xCoord;
        this.yCoord=yCoord;
    }
 
    public int get_x(){
        return xCoord;
    }
 
    public int get_y(){
        return yCoord;
    }
 
    public void set_x(int x){
        xCoord=x;
    }
 
    public void set_y(int y){
        yCoord=y;
    }
 
    public String toString(){
        return "("+xCoord+" , "+yCoord+")";
    }
}

public abstract class Ship {
 
    private String name;
    private int length;
    private Point[] position;
    public List<Point> hit = new ArrayList<>();
 
    public Ship(String name, int length){
        this.name=name;
        this.length=length;
    }
 
    public Ship(String name, int length, Point[] position){
        this.name=name;
        this.length=length;
        this.position=position;
    }
 
    public String getName(){
        return name;
    }
 
    public int getLength(){
        return length;
    }
 
    public Point[] getPosition(){
        return position;
    }
 
    public abstract boolean isHit(int x, int y);
    public abstract boolean isSank();
 
}
