public class BinarySearch {
    public static void main(String[] args){
        int key = 9;
        int[] a = new int[]{1,2,3,4,5,6,7,8,9,0};
        System.out.println(search(a,0,a.length-1,key));
    }
    public static int search(int[] a,int l,int r, int key){
        if (l>r){
            return -1;
        }
        else{
            int mid = (l+r)/2;
            if(key==a[mid]){
                return mid;
            }
            else if(key<a[mid]){
                return search(a,0,mid,key);
            }
            else{
                return search(a,mid+1,r,key);
            }
        }
    }
}
QuickSort:
public class QuickSort {
    public void quickSort(int[] a, int low, int high) {
        int i = low;
        int j = high;
        if (i >= j) {
            for (int z : a) {
                System.out.println(z);
            }
            return ;
        }
        if (i < j) {
            int value = a[i];
            while (i < j) {
                while (i < j && a[j] >= value) {
                    j -= 1;
                }
                while (i < j && a[j] < value) {
                    int x = a[i];
                    int y = a[i+1];
                    a[i] = a[j];
                    a[i+1]=a[j];
                    a[j] = x;
                    a[j]=y;
                    i+=1;
                }
            }
            a[i] = value;
        }
        quickSort(a, low, i);
        quickSort(a, i + 1, high);
    }
 
    public static void main(String[] args) {
        int[] x = new int[] { 7, 32, 28, 3, 1, 8, 0, 4, 3 };
        QuickSort y = new QuickSort();
        y.quickSort(x, 0, 8);
    }
}

public class MergeSort {
    public static void sort(int[] x, int l, int r) {
        if (l >= r) {
            return;
        }
        int m = (l + r) / 2;
        sort(x, l, m);
        sort(x, m + 1, r);
        mergeTwo(x, l, m, r);
    }
 
    public static void mergeTwo(int a[], int l, int m, int r) {
        int[] newList = new int[a.length];
        int rIndex = m + 1;
        int lIndex = l;
        int counter = l;
        while (l <= m && rIndex <= r) {
            if (a[l] <= a[rIndex]) {
                newList[lIndex++] = a[l++];
            }
            else {
                newList[lIndex++] = a[rIndex++];
            }
        }
            while (l <= m) {
                newList[lIndex++] = a[l++];
            }
            while (rIndex <= r) {
                newList[lIndex++] = a[rIndex++];
            }
 
        while (counter <= r) {
            a[counter] = newList[counter];
            System.out.print(a[counter] + "  ");
            counter++;
        }
        System.out.println();
    }
 
    public static void main(String[] args) {
        int a[] = new int[] { 26, 5, 98, 108, 28, 99, 100, 56, 34, 1 };
        MergeSort.sort(a, 0, a.length - 1);
    }
 
}
